#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        char a[1002];
        int n;
        cin>>n;
        scanf("%s",a);
        int y=0;
        int in=0;
        for(int i=0;i<n;i++)
        {
            if(a[i]=='Y')
                y=1;
            if(a[i]=='I')
                in=1;
        }
        if(y==0 && in==0)
            cout<<"NOT SURE
";
        else if(y==1 && in==0)
            cout<<"NOT INDIAN
";
        else
            cout<<"INDIAN
";
    }

    return 0;
}
