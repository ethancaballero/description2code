#include <iostream>
#include<stdio.h>

using namespace std;

int main()
{
        int t, n, in = 0, ni = 0;
        char arr[1001];

        scanf("%d", &t);
        while(t--){
                scanf("%d", &n);
                scanf("%s", &arr[0]);
                in=0, ni=0;
                for(int i = 0; i<n; i++){
                        if(arr[i] == 'Y'){
                                ni++;
                                break;
                        }

                        if(arr[i] == 'I'){
                                in++;
                                break;
                        }

                }

                if (!ni && !in) printf("NOT SURE
");
                else if (in) printf("INDIAN
");
                else printf("NOT INDIAN
");
        }
    return 0;
}
