#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>

using namespace std;

int main()
{
    int t,n;
    char str[1000];
    scanf("%d",&t);
    while(t--)
    {
        int count=0,curr=0;
        scanf("%d",&n);
        scanf("%s",&str);
        for(int i=0;i<strlen(str);i++)
        {
            if(str[i]=='I')
                count++;
            else if(str[i]=='Y')
               curr++;
        }
        if(count>0)
            printf("INDIAN
");
        else if(curr>0)
            printf("NOT INDIAN
");
        else
            printf("NOT SURE
");

    }
    return 0;
}
