#include <bits/stdc++.h>
long long int t,r,a1,a2,b1,b2,c1,c2,cou;;
using namespace std;
int main()
{
	scanf("%lld",&t);
	while (t--) {
		cou = 0;
		scanf("%lld%lld%lld%lld%lld%lld%lld",&r,&a1,&a2,&b1,&b2,&c1,&c2);
		
		if (sqrt((a1-b1)*(a1-b1) + (a2-b2)*(a2-b2)) <= r ) {
			cou++;
		}
 		if (sqrt((a1-c1)*(a1-c1) + (a2-c2)*(a2-c2)) <= r) {
			cou++;
		}
		if (sqrt((c1-b1)*(c1-b1) + (c2-b2)*(c2-b2)) <= r) {
			cou++;
		}
		if (cou < 2) {
			printf("no
");
		}
		else {
			printf("yes
");
		}
	}
	return 0;
}
