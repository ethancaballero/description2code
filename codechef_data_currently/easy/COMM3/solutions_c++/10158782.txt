#include<iostream>
#include<math.h>
using namespace std;

float dist(float *a,float *b)
{
    return sqrt(pow(a[0]-b[0],2)+pow(a[1]-b[1],2));
}

int main()
{
    int t;
    float r,m[2],c1[2],c2[2],d[2];
    cin>>t;
    while(t-->0)
    {
        cin>>r;
        cin>>m[0]>>m[1]>>c1[0]>>c1[1]>>c2[0]>>c2[1];
        d[0]=dist(m,c1);
        d[1]=dist(m,c2);
        if(d[0]>r||d[1]>r)
        {
            if(dist(c1,c2) <= r)
                cout<<"yes
";
            else
                cout<<"no
";
        }
        else
            cout<<"yes
";

    }
    return 0;
}
