#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int tot[10007]={0};
		long long int a,b,p,q;
		scanf("%lld %lld %lld %lld",&a,&b,&p,&q);
		int i,j;
		for(i=0;i<a;i++)
		{
			long long int tem;
			if(i<a-1)
				for(j=0;j<p;j++)
						scanf("%lld",&tem), tot[i]=tot[i]+tem;
			else
				for(j=0;j<p-1;j++)
						scanf("%lld",&tem), tot[i]=tot[i]+tem;
		}
		sort(tot,tot+a-1);
		long long int ans=tot[a-b-1]-tot[a-1]+1;
		if(ans<=q)
			if(ans>=0)
				printf("%lld
",ans);
			else
				printf("0
");
		else
			printf("Impossible
");
	}
	return 0;
}
