    #include<iostream>
    #include<cstdio>
    #include<algorithm>
    using namespace std;
    void sort(long long *,int);
    int main()
    {
    	int T,N,K,E,N1,K1;          //N->no. of students 
    	long int M;
    	long long sum,M1,p[100001];					//K->no. of students to get passed
    	cin>>T;								//E->no. of entrance exams
    	while(T--)							//M->max. marks in each exam
    	{
    		cin>>N>>K>>E>>M;
    		for(N1=0;N1<(N-1);N1++)
    		{
    			K1=0;
    			sum=0;
    			while(K1!=E)
    			{
    				cin>>M1;
    				sum+=M1;
    				K1++;
    			}
    			p[N1]=sum;
    		}
    		sum=0;
    		while(--E)
    		{
    			cin>>M1;
    			sum+=M1;
    		}
    		sort(p,p+N-1);
    		M1=p[N-1-K]-sum;
    		if(M1>=M)
    		cout<<"Impossible
";
    		else
    		{
    			if(M1>=0)
    				cout<<M1+1<<endl;
    			else 
    				cout<<"0"<<endl;
    		}
    		
    	}
    	return 0;
    }
