#include<stdio.h>
#include<bits/stdc++.h>
#define MAX 10001
using namespace std;
long long int a[MAX][5],b[MAX],c[MAX];
int main()
{
    long long int m,sum,re,ans=0,t,e,k,n,j,i,d;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld %lld %lld %lld",&n,&k,&e,&m);
        for(i=0;i<n-1;i++)
        {
            sum=0;
            for(j=0;j<e;j++)
            {
                scanf("%lld",&a[i][j]);
                sum=sum+a[i][j];
            }
            c[i]=sum;
        }
         //for(i=0;i<n-1;i++)
        //printf("%lld",c[i]);
        sort(c,c+n-1);
        //reverse(c,c+n-1);

        re=0;
          for(i=0;i<e-1;i++)
        {
            scanf("%lld",&b[i]);
            re=re+b[i];
        }
        //printf("%lld",re);
        d=n-k-1;
        if(c[d]>=re){
        ans=c[d]+1;
        ans=ans-re;
        //printf("%lld",ans);
        }
        else
            ans=0;
            if(ans<0)
                ans=0;
        if(ans>m)
             printf("Impossible
");
        else
         printf("%lld
",ans);

    }
    return 0;
}
