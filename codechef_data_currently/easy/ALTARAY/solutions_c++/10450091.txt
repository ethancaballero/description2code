#include <iostream>
using namespace std;
typedef long long int l;
l a[100000];
l dp[100000];
int main() {
	l t,n,f;
	cin>>t;
	while(t--)
	{	cin>>n;
		for(int i=1;i<=n;i++)
		{cin>>a[i];
		dp[i]=1;
		}
	
		for(int i=n-1;i>0;i--)
		{
			if(a[i+1]>0)
			f=1;
			else
			f=2;
			if(f==1&&a[i]<0||f==2&&a[i]>0)
			dp[i]=dp[i]+dp[i+1];
	
		}
			for(int i=1;i<=n;i++)
			cout<<dp[i]<<" ";
			cout<<"
";
	}
	
	
	return 0;
}
