#include <bits/stdc++.h>

using namespace std;

 

int t,n;

int a[100005];

int b[100005];

void solve(){

cin >> n;

for(int i=0;i<n;i++) scanf("%d",a+i);

b[n-1] = 1;

for(int i=n-2;i>=0;i--){

if((a[i+1] > 0 && a[i] < 0) || (a[i+1] < 0 && a[i] > 0)){

b[i] = 1+b[i+1];

} else

b[i] = 1;

}

printf("%d",b[0]);

for(int i=1;i<n;i++) printf(" %d",b[i]);

printf("
");

}

 

 

int main(){

cin >> t;

while(t--) solve();

return 0;

}
