#include<stdio.h>
#include <math.h>
int main(){
    long long int n,t;
    scanf("%lld",&t);
    while(t--){
        scanf("%lld",&n);
      long long int num,b[n+1],i,j=0,d[n+1];
      for(i=1;i<=n;i++){
        scanf("%lld",&num);
        if(num>=0)
            d[i]=1;
        else
            d[i]=-1;
        if(d[i]==d[i-1]){
          b[j]=i;
        j++;}
      }
      b[j]=n+1;
      j=0;
        for(i=1;i<=n;i++)
        {
            if(i==b[j])
                j++;
            printf("%lld ",b[j]-i);
        }
            printf("
");
       }

return 0;
}
