#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll n;
        cin>>n;
        ll a[n],count=1,ans[n];
        for(ll i=0;i<n;i++)
            cin>>a[i];
        for(ll i=n-2;i>=0;i--)
        {
            if((a[i]*a[i+1])<0)
            {
                count++;
                ans[i]=count;
            }
            else
            {
                count=1;
                ans[i]=count;
            }
        }

    }

}
