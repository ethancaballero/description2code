#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%i",&t);
    while(t--)
    {
        int n,a,c[100000],x=0,p=0;
        scanf("%i",&n);
        for(int i=0;i<n;i++)
            c[i]=0;
        for(int i=0;i<n;i++)
        {
            scanf("%i",&a);
            if(p!=1&&a>0)
            {
                p=1;
                c[x]++;
            }
            else if(p!=-1&&a<0)
            {
                p=-1;
                c[x]++;
            }
            else
            {
                x++;
                c[x]++;
                if(a>0)
                p=1;
                else if(a<0)
                p=-1;
                else
                p=0;
            }
        }
        int i=0;
        while(i<=x)
        {
            if(c[i]>0)
            {
                printf("%i ",c[i]);
                c[i]--;
            }
            else
                i++;
        }
        printf("
");
    }
	return 0;
}
