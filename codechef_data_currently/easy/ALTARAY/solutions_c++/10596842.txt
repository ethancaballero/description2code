#include <iostream>
#include<cstdio>
using namespace std;
int main()
{
    long long int t,i,j;
    long long int n;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        long long int a[n],b[n];
        for(i=0;i<n;i++)
            scanf("%lld",&a[i]);
        std::fill(b,b+n,1);
        for(i=n-2;i>=0;i--)
        {
            if((a[i+1]>=0)&&(a[i]<0))
                b[i]=b[i+1]+1;
            else if((a[i]>=0)&&(a[i+1]<0))
                b[i]=b[i+1]+1;
            else
                b[i]=1;
        }
        for(i=0;i<n;i++)
            printf("%lld ",b[i]);
        printf("
");
    }
    return 0;
}
