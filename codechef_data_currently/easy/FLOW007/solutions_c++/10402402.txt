int main()
{

int T,N,p[5],x;

long *result=new long[T];

{ num=0;
  cin>>N;
  for(int j=0;j<5;j++)
   {
     if(N/10!=0)
      {
	p[j]=N%10;
	N/=10;
	x=j;
      }
     else

   }

}

 {
