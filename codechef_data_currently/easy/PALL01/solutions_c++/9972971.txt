#include<stdio.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,rem,rev=0,temp;
		scanf("%d",&n);
		temp = n;
		while(n != 0)
		{
			rem = n%10;
			rev = rev*10+rem;
			n = n/10;
		}
//		printf("%d",rev);
		if(rev == temp)
			printf("wins
");
		else
		printf("losses
");
	}
	return 0;
}
