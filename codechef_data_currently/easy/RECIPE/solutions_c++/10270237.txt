#include <iostream>
#include <vector>

using namespace std;

int main() {
  int t;
  cin >> t;
  while(t--) {
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];
    int d = 1;
    for (int i = 2; i <= 1000; ++i) {
      bool div = true;
      for (int j = 0; j < n; ++j) div &= (a[j]%i==0);
      if (div) d = i;
    }
    for (int i = 0; i < n; ++i)
      cout << a[i]/d << (i+1==n?'
':' ');
  }
}
