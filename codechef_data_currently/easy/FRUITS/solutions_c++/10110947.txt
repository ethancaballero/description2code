#include <iostream>
using namespace std;
#define ll long long

int main()
{   ll t;
    cin>>t;
    while(t--)
    {   ll a,b,c,d,temp;
        cin>>a>>b>>c;
        if(b>a)
        {   temp=a;
            a=b;
            b=temp;
        }
        d=a-b;
        d-=c;
        if(d<0)
            cout<<"0
";
        else
            cout<<d<<"
";
    }
    return 0;
}
