#include <iostream>
#include<stdio.h>
 
 
int main(){
    int t,n,k,m;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d %d", &n,&m,&k);
        int diff;
        if(n > m)
            diff = n-m;
        else
            diff = m-n;
        if(k >= diff)
            printf("0
");
        else
            printf("%d
",diff-k);
    }
    return 0;
}
