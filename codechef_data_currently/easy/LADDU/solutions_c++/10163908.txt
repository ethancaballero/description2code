#include <iostream>
#include <string>

using namespace std;

int main() {
    int T;

    cin >> T;

    while(T--) {
        int activities, points = 0;
        string origin;
        cin >> activities >> origin;

        while(activities--) {
            string activity;
            int activityData;

            cin >> activity;

            if(activity == "CONTEST_WON") {
                cin >> activityData;
                points += 300 + (activityData < 20 ? 20 - activityData : 0);
            } else if(activity == "TOP_CONTRIBUTOR") {
                points += 300;
            } else if(activity == "BUG_FOUND") {
                cin >> activityData;
                points += activityData;
            } else if(activity == "CONTEST_HOSTED") {
                points += 50;
            }
        }

        if(origin == "INDIAN") {
            cout << points/200 << "
";
        } else {
            cout << points/400 << "
";
        }
    }
}
