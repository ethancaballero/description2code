#include<bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	int t, activities;
	string origin;
	int count=0;
	int rank=0,severity=0;
	cin>>t;
	int ans[t+32],ct=0;
	while(t--){
		cin>>activities;
		cin>>origin;
		count=0;
		for(int i=0;i<activities;++i)
		{
			string activity;
			cin>>activity;
			if(activity=="CONTEST_WON")
			{
				cin>>rank;
				if(rank>20)
					count+=300;
				else
					count=count+300+20-rank;
			}
			else if(activity=="TOP_CONTRIBUTOR")
			{
				count+=300;
			}
			else if(activity=="BUG_FOUND")
			{
				cin>>severity;
				count=count+severity;
			}
			else if(activity=="CONTEST_HOSTED")
			{
				count=count+50;
			}
		}
		if(origin=="INDIAN")
			ans[ct]=count/200;
		else if(origin=="NON_INDIAN")
			ans[ct]=count/400;
		ct++;
	}
	for(int i=0;i<ct;++i)
	{
		cout<<ans[i]<<'
';
	}
	return 0;
}
