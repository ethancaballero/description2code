#include <bits/stdc++.h>
using namespace std;

int main()
{
	/* code */
	int t,act,laddu,rank,sev;
	int ans[500],z=0;
	string nation;
	cin>>t;
	while(t--){
		act=0;
		laddu=0;
		cin>>act>>nation;
		while(act--){
			string data;
			cin>>data;
			if(data=="CONTEST_WON"){
				cin>>rank;
				laddu+=(300);
				if(rank<=20)
					laddu+=(20-rank);
			}
			else if(data=="TOP_CONTRIBUTOR")
				laddu+=300;
			else if(data=="BUG_FOUND"){
				cin>>sev;
				laddu+=sev;
			}
			else if(data=="CONTEST_HOSTED")
				laddu+=50;
		}
		if(nation=="INDIAN")
			ans[z]=laddu/200;
		else if(nation=="NON_INDIAN")
			ans[z]=laddu/400;
		z++;
	}
	for(int j=0;j<z;++j)
		cout<<ans[j]<<'
';
	return 0;
}
