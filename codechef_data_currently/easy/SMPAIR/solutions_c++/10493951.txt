#include <iostream>
using namespace std;
void merge(int a[],int l,int m,int u)
{
if(l!=u)
{int n1,n2;
n1=m-l+1;
n2=u-m;
int c[n1],d[n2];
int i,j,k;
for( i=0;i<n1;i++)
{
c[i]=a[l+i];
}
for( j=0;j<n2;j++)
{
d[j]=a[m+j+1];
}
i=0;
j=0;
k=l;
while(i<n1 && j<n2)
{
if(c[i]<=d[j])
{
a[k]=c[i];
i++;
}
else 
{
a[k]=d[j];
j++;
}
k++;
}
while(i<n1)
{
a[k]=c[i];
i++;
k++;
}
while(j<n2)
{
a[k]=d[j];
j++;
k++;
}}
}
