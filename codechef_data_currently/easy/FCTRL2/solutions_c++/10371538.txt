#include<cstdio>
#include<cstring>
int ans[1111],digit;
void init(){
    memset(ans,0,sizeof(ans));
    ans[1]=1;
    digit=1;
}
int main(){
    int cs;
    scanf("%d",&cs);
    while(cs--){
        int n;
        scanf("%d",&n);
        init();
        for(int i=2;i<=n;i++){
                int carry=0;
            for(int j=1;j<=digit;j++){
                int tmp=ans[j]*i+carry;
                ans[j]=tmp%10;
                carry=tmp/10;
            }
            while(carry){
                ans[++digit]=carry%10;
                carry/=10;
            }
        }
        for(int i=digit;i>=1;i--)
             printf("%d",ans[i]);
        printf("
");
    }
    return 0;
}
