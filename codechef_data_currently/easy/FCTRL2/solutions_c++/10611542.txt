#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);

	while(t--){
		int n;
		int res[500];
		scanf("%d",&n);
		res[0] = 1;
		int res_size = 1;
		for(int i=2; i<=n; i++){
			int prod;
			int carry = 0;
			for(int j=0; j<res_size; j++){
				prod = res[j]*i + carry;
				res[j] = prod % 10;
				carry = prod / 10;
			}

			while(carry){
				res[res_size] = carry % 10;
				carry = carry / 10;
				res_size++;
			}
		}

		for(int i=res_size-1; i>=0; i--){
			printf("%d",res[i]);
		}
		printf("
");
	}
	return 0;
}
