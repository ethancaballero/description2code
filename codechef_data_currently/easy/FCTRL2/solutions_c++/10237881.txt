#include<iostream>
using namespace std;
void factorial(int n){
	int i,size=1,carry=0,j,temp;
	int arr[200];
	arr[0]=1;
	if(n==1){
		printf("1
");
		return;
	}
	for(i=2;i<=n;i++){
		for(j=0;j<size;j++){
			temp=arr[j]*i+carry;
			arr[j]=temp%10;
			carry=temp/10;
		}
		while(carry!=0){
			arr[j++]=carry%10;
			carry=carry/10;
		}
		size=j;	
	}
	for(i=size-1;i>=0;i--){
		printf("%d",arr[i]);
	}
	printf("
");
}
int main()
{
	int i ,j, n;
	cin>>n;
	int num;
	for(i=0;i<n;i++)
	{
		cin>>num;
		factorial(num);
	} 
return 0;
}
