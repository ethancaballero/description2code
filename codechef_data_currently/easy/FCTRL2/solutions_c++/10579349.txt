#include <iostream>
#include<vector>
#include<stdio.h>

using namespace std;

int main()
{
        int t, n, carry, q, temp;
        long long j;
        vector<int> fact;

        scanf("%d", &t);
        while(t--){

                fact.push_back(1);
                scanf("%d", &n);

                for(int i = 2; i<=n; i++){

                        j = fact.size();
                        carry = 0;

                        while(j-1>-1){
                                j = j-1;
                                temp = fact[j] * i + carry;
                                q = temp%10;
                                carry = temp >= 10? temp/10: 0;

                                fact[j] = q;
                                //if (i == 19)
                                        //printf("%d ", fact[j]);

                        }
                        while(carry){
                                temp = carry%10;
                                carry /= 10;
                                fact.insert(fact.begin(), temp);
                        }

                }

                for(int i = 0; i < fact.size(); i++)
                        printf("%d", fact[i]);

                printf("
");


                fact.clear();
        }
    return 0;
}
