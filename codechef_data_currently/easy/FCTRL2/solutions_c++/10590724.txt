#include<iostream>

    using namespace std;


    int main()
    {
      int n;
      cin>>n;

      while(n--)
      {
          int factorial;
    	  cin>>factorial;
    	  int array[200];
    array[199]=1;

    		int k=0;
    		  while(k<199)
    	  {
    	     array[k++]=-1;
    	  }

    		int p1;
    		int temp;
    		int index;
    		if(factorial==1)
    		  {
    		  cout<<"1"<<endl;
    		   continue;
    		}
    		while(factorial>1)
    		{
    		     temp=0;
    		     index=199;
    			while(array[index]>=0)
    			{

    			   temp=((factorial*array[index]+temp));

    			    array[index]=temp%10;
    			      temp=temp/10;
    				  index--;

    				  }

    				while(temp)
    		{
    		   array[index]=temp%10;
    		   temp/=10;
    		   index--;

    		}

    				factorial--;
    		}

    		p1=index+1;

    	  while(p1<=199)
    	  {
    	     cout<<array[p1++];
    	  }
    	  cout<<endl;
      }

    return 0;
    }
