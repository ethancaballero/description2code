#include <iostream>
using namespace std;

int curIndex=0;
void vFactorial(int val, int arResult[])
{
    if(val==0)
        return;
        
    int tempVal = 0;
    for(int i = 0; i<=curIndex; i++)
    {
        int curVal = val * arResult[i] + tempVal;
        arResult[i] = curVal%10;
        tempVal = curVal/10;
    }
    while(tempVal != 0)
    {
		curIndex++;
        arResult[curIndex] = tempVal%10;
		tempVal = tempVal/10;
    }
    return vFactorial(val-1, arResult);
}
