int mul(int x,int a[],int as)
{
int c=0;

    for(int i=0;i<as;i++)
    {
        int pro=(a[i]*x)+c;
        a[i]=pro%10;
        c=pro/10;
    }
    while(c)
    {
        a[as]=c%10;

    }
