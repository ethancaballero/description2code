#include<stdio.h>
int main()
{
    int n,i,count;
    char ch;
    scanf("%d",&n);
    scanf("%c",&ch);
    i=0;
    while(i<n)
    {
        count = 0;
         while(scanf("%c",&ch)!=EOF)
        {
            if(ch==10)
            {
                break;
            }
            else
            {   
                if(ch=='4')
                {
                    count++;
                }
            }
        }
        printf("%d
",count);
        i++;
    }
    return(0);
}
