#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
	string x,y,ans;
	cin >> x >> y;
	for(int i=0;i<x.length();i++)
	{
		if(x.at(i)==y.at(i))
			ans.push_back('W'==x.at(i) ? 'B' : 'W');
		else
			ans.push_back('B');
	}
	cout << ans << '
';
	}
	return 0;
}
