#include <bits/stdc++.h>
using namespace std;

int main()
{
#ifdef gsdt
    freopen("input.txt","r",stdin);
#endif // gsdt
//#define debug

    int n,k;
    int a[100005], trace[100005];
    long long ans;
    const long long oo=1000000007;
    deque<pair<double,int> > q;

    scanf("%d%d",&n,&k);
    for(int i=1; i<=n; i++) scanf("%d",a+i);

    q.push_back(make_pair(log(a[1]),1));
    trace[1]=0;
    ans=a[1];
    for(int i=2; i<=k; i++)
    {
        while(!q.empty() && q.back().first>log(a[1]*a[i])) q.pop_back();
        q.push_back(make_pair(log(a[1])+log(a[i]),i));
        trace[i]=1;
    }

    #ifdef debug
        for(auto x:q)
        {
            cout<<x.first<<" "<<x.second<<endl;
        }
        cout<<"---------------------------
";
    #endif // debug

    for(int i=k+1; i<=n; i++)
    {
        while(!q.empty() && i-q.front().second>k) q.pop_front();
        trace[i]=q.front().second;
        while(!q.empty() && q.back().first>q.front().first+log(a[i])) q.pop_back();
        q.push_back(make_pair(q.front().first+log(a[i]),i));
        //q.pop_front();

        #ifdef debug
        cout<<i<<endl;
        for(auto x:q)
        {
            cout<<x.first<<" "<<x.second<<endl;
            printf("trace[%d]=%d
",i,q.front().second);
        }
        cout<<".............
";
        #endif // debug
    }
    ans=a[n];
    int i=n;
    while(trace[i])
    {
        i=trace[i];
        ans=ans*1LL*a[i]%oo;
    }
    printf("%I64d",ans);
    return 0;
}
