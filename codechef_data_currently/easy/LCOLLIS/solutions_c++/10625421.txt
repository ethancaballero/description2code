#include <bits/stdc++.h>
using namespace std;
 
int main() {
	int t;
	cin >> t;
        while (t--) {
	int n,m,i,j,ans=0;
	cin >> n >> m;
	char a[n][m];
	for (i=0;i<n;i++)
	    cin >> a[i];

	for (i=0;i<m;i++)
	{
	    int c=0;
	    for (j=0;j<n;j++)
	    {
	        if (a[j][i] == '1')
	            c++;
	    }
	    ans = ans + (c*(c-1))/2;
	 }
	 cout << ans << "
";
}
	return 0;
}
