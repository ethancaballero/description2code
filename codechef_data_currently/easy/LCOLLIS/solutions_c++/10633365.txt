ll power(ll a,ll b,ll m)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a, b/2, m);
	temp=(temp*temp)%m;
	if(b&1)
		temp=(temp*a)%m;
	return temp;
}

		{
			adj[u].pb( v);
