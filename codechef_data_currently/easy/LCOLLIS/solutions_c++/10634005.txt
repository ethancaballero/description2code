int main()
{
    ll int t;
    Sl(t);
    while(t--) {
        ll int n, m;
        Sl(n); Sl(m);
        string a[n];
        ll int i, j;
        ll int count = 0;
        for (i = 0; i < n; i++)
        {
            cin >> a[i];
        }
        ll int c2 = 0;
        for (j = 0; j < m; j++)
        {
            c2 = 0;
            for (i = 0; i < n; i++)
            {
                if(a[i][j] == '1')
                    c2++;
            }
            if (c2 > 1)
            {
                c2 = c2 * (c2 - 1);
                c2 = c2 / 2;
                count += c2;
            }
        }
        Pl(count);
    }
    return 0;
}
