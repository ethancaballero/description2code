#include<iostream>
#include<vector>
typedef long long int ll;
using namespace std;
int main()
{
	ll t;
	cin>>t;
	
	while(t--)
	{
		ll N,M,i,j;
		cin>>N>>M;
		//vector< vector<bool> > matrix(n, vector<bool>(m, false));
		char Matrix[N][M];
		for(i=0;i<N;i++)
		for(j=0;j<M;j++)
		cin>>Matrix[i][j];
		ll sum=0;
		for(j=0;j<M;j++)
		{
			ll count=0;
			for(i=0;i<N;i++)
			{
				if(Matrix[i][j]=='1')
					count++;
			}
			sum+=(count*(count-1)/2);
		}
		cout<<sum<<"
";
	}
	return 0;
}
