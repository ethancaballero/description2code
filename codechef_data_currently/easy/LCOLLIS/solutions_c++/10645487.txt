#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
	  char a[n][m];
		long int sum=0;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				cin>>a[i][j];
			}
		}
		for(int i=0;i<m;i++)
		{
			int count=0;
			for(int j=0;j<n;j++)
			{
				if(a[j][i]=='1')
				count++;
			}
			if(count>=2)
			sum=sum+(count*(count-1))/2;
		}
		cout<<sum<<"
";
	}
}
