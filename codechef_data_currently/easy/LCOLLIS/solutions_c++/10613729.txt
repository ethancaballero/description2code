#include <iostream>
#include <cstdio>
 
using namespace std;
 
int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
		int n,m;
		char a[11][11];
		scanf("%d %d",&n,&m);
 
		int sum = 0;
		int colsum[11] = {0};
		for(int i=0; i<n; i++){
			for(int j=0; j<m; j++){
				cin >> a[i][j];
				if(a[i][j] == '1'){
					colsum[j]++;
				}
			}
		}
 
		for(int j=0; j<m; j++){
			if(colsum[j] > 1){
				sum += colsum[j] * (colsum[j]-1)/2;
			}
		}
		printf("%d
",sum);
	}
	return 0;
}
