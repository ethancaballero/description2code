#define ll long long

#define MIN(a, b) a < b ? a : b

using namespace std;

int main(int argc, char *argv[]) {
    int t;
    scanf("%d", &t);

    while( t-- ) {
        ll int a, b, c, answer;
        scanf("%lld%lld%lld", &a, &b, &c);

        if( (a > b && b > c) || (c > b && b > a) ) {
            answer = b;
        }
        else if( (a > c && c > b) || (b > c && c > a) ) {
            answer = c;
        }
        else {
            answer = a;
        }

        printf("%lld
", answer);
    }
    return 0;
}
