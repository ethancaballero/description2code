#include <iostream>
using namespace std;
int main()
{
	int a,b,c,t,second_max;
	cin >> t;
	while(t--)
	{
		cin >> a >> b >> c ;
		    if ( a < b )
    {
        if ( b < c ) second_max = b;
        else second_max = ( a < c ? c : a );
    }
    else
    {
        if ( a < c ) second_max = a;
        else second_max = ( b < c ? c : b );
    }
		cout << second_max << "
" ;
	}	
	 return 0;
}
