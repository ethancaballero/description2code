#include <iostream>
#include <string>

#define VALIDATE_POS(pos)         {                 if (pos == std::string::npos)                 {                         break;                  }         }
using namespace std;

int main()
{
        string s;
        cin >> s;
        int count = 0;

        size_t pf, pe, ph, pc;

        while(true)
        {
                pf = s.rfind('F');
                VALIDATE_POS(pf);

                pe = s.rfind('E', pf);
                VALIDATE_POS(pe);

                ph = s.rfind('H', pe);
                VALIDATE_POS(ph);

                pc = s.rfind('C', ph);
                VALIDATE_POS(pc);

                count++;

                s.erase(pc, 1);
                s.erase(ph-1, 1);
                s.erase(pe-2, 1);
                s.erase(pf-3);
        }

        cout << count << endl;
        return 0;
}
