#include<iostream>
using namespace std;
int b[100003],ch,ce,cf,cc;
int main()
{
   string str;
   cin>>str;
   int l = str.length();
   int i;

   for(i=l-1;i>=0;i--)
   { //cout<<i<<'
';
     if(str[i]=='F')
        cf++;
     else if(str[i]=='E')
       {
          if(ce<cf)
            {
              ce++;
              b[i]=1;
            }
       }
     else if(str[i]=='H')
     {
          if(ch<ce)
            {
               ch++;
               b[i]=1;
            }
     }
     else
      {
          if(cc<ch)
          {
            cc++;
            b[i]=1;
          }
      }
   }
   int ct=0;
   for(i=0;i<l;i++)
     {
       if(str[i]=='C' && b[i]==1)
          ct++;
     }
     cout<<ct<<'
';
}
