#include<stdio.h>
void sort(int arr[],int num)
{
	int i,j,temp;
	for(i=1;i<num;i++)
	{
		j=i;
		while(j>0 && arr[j] < arr[j-1])
		{
			temp = arr[j];
			arr[j] = arr[j-1];
			arr[j-1] = temp;
			j--;
		}
	}
}
