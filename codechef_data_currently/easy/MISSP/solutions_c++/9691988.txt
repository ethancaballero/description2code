#include<iostream>
using namespace std;

void insert(int a[],int n)
{
    int i,j,temp;
    for(i=1;i<n;i++)
    {
        temp=a[i];
        j=i-1;
        while( j>=0 && temp < a[j] )
        {
            a[j+1]=a[j];
            j--;
        }
        a[j+1]=temp;
    }
}


int main()
{
  int n,i,t;
  int a[100010];
  cin>>t;
  while(t--)
  {
        cin>>n;
        for(i=0;i<n;i++)
        cin>>a[i];
        insert(a,n);
        for(i=0;i<n;i+=2)
        {
            if(i+1==n)
            {
               cout<<a[i]<<"
";
               break;
            }
            if(a[i]!=a[i+1])
            {
                cout<<a[i]<<"
";
                break;
            }
        }
  }
}
