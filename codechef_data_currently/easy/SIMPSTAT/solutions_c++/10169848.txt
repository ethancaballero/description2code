#include <bits/stdc++.h>
using namespace std;

typedef long long int lli;

int main(int argc, char const *argv[])
{
	ios::sync_with_stdio(false);
	cout.precision(10);

	int tc;
	cin>>tc;
	while(tc--) {
		lli a[10005];
		lli n,k;
		cin >> n >> k;
		for (int i = 0; i < n ; ++i)
		{
			cin >> a[i];
		}

		sort(a,a+n); // arguements of sort are [first,last)	
		long double ans=0;
		for (int i = k; i < n-k; ++i) // because starting from 0
		{
			ans += a[i];
		}

		
		ans /= (n-(2*k));
		cout << fixed  << ans << "
";
		
	}
	return 0;
}
