#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
        int t, num, first, last;

        scanf("%d", &t);
        while(t--){
                scanf("%d", &num);

                first = num%10;

                while(num/10 != 0)
                        num = num/10;
                last = num%10;

                printf("%d
", first+last);

        }
    return 0;
}
