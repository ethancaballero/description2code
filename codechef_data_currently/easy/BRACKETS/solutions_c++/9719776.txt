#include<iostream>
using namespace std;
int main()
{
   int t,bal,max_bal,i;
   char ch[100010];
   cin>>t;
   while(t--)
   {
      bal=i=max_bal=0;
      cin>>ch;
       while(ch[i])
       {
           if(ch[i]=='(')
                bal++;
           if(ch[i]==')')
            bal--;
           if(bal>max_bal)
            max_bal=bal;
            i++;
       }
       for(i=0;i<max_bal;i++)
        cout<<'(';
       for(i=0;i<max_bal;i++)
        cout<<')';
       cout<<"
";
   }

}
