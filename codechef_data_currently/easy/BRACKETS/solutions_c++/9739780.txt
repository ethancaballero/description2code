#include<iostream>
#include<string.h>
using namespace std;
char s[100001];
int main()
{
	ios::sync_with_stdio(false);
	long int t,i;
	cin>>t;
	while(t--)
	{
		cin>>s;
		int balance=0,max_balance=0;
		for(i=0;i<strlen(s);i++)
		{
		    
			if(s[i]=='(')
			balance=balance+1;
			else if(s[i]==')')
			balance=balance-1;
			max_balance=max(max_balance,balance);
			
		}
		for(i=0;i<max_balance;i++)
		cout<<"(";
			for(i=0;i<max_balance;i++)
		cout<<")";
		cout<<"
";
	}
}
