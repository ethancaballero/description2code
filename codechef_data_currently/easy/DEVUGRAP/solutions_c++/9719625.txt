#include<iostream>
#define abs(n) (n<0?-n:n)

int min1(int a, int b)
{
    return a<b?a:b;
}
using namespace std;

int main()
{
    int n,t,k,i;
  unsigned long int sum,data;
   cin>>t;
   while(t--)
   {
       sum=0;
       cin>>n>>k;
       for(i=0;i<n;i++)
        {
            cin>>data;
            if(data<k)
                sum+=(k-data);
            else
            sum+=min1(data%k,abs(k-data%k));
        }
      cout<<sum<<"
";
   }

}
