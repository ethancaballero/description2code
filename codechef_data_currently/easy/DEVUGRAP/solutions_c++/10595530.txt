#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
	    int n,k,a;
	    long long int sum=0;
	    scanf("%d%d",&n,&k);
	    for(int i=0;i<n;i++){
	        scanf("%d",&a);
	        int val=a%k;
	        if(a>=k){
	            int plus=(k-val<val)?k-val:val;
	            sum+=plus;
	        } else {
	            sum+=k-val;
	        }
	    }
	    printf("%lld
",sum);
	}
	return 0;
}
