#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,a,b,c;
    scanf("%d",&t);
    while(t--)
    {
        c=0;
        scanf("%d %d",&a,&b);

        while(a!=b)
       {
        if(a%2==1 && a!=1)
            {
                a=(a-1)/2;
                c++;
            }
        else
        if(a%2==0 || a==1)
        {
          if(a>b || (a<b &&(b%a!=0)))
            {
                a=a/2;
                c++;
            }
            else
                if(a<b && (b%a==0))
            {
                a=2*a;
                c++;
            }
        }
        }
        printf("%d
",c);
    }
    return 0;
}
