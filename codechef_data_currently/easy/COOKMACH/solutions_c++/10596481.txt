#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,a,b,c;
    scanf("%d",&t);
    while(t--)
    {
        c=0;
        scanf("%d %d",&a,&b);
        while(a!=b)
       {
        if(a%2==1 && a!=1)
            {
                a=(a-1)/2;
                c++;
            }
        else
        {
          if(a>b)
            {
                a=a/2;
                c++;
            }
            else
            {
                b/=2;
                c++;
            }
        }
        }
        printf("%d
",c);
    }
    return 0;
}
