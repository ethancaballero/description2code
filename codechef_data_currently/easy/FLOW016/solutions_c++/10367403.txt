#include <iostream>
#include <stdio.h>
using namespace std;

long int gcd(long int a, long int b) {
    long int t;
    if(a < b) {
        t=a;
        a=b;
        b=t;
    }
    while(!(b==0)) {
        t=a;
        a=b;
        b=t%b;
    }
    return a;
}

int main() {
	long int t, a, b, g;
	scanf("%d", &t);
	while(t--){
	    scanf("%ld %ld", &a, &b);
	    g = gcd(a, b);
	    printf("%ld %ld
", g, a*b/g);
	}
	return 0;
}
