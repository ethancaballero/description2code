#include<iostream>
#include<cstring>
#include<cstdio>
#include<vector>
#include<cmath>

using namespace std;

int gcd(int m,int n)
{
    if(n==0)
        return m;
    else
        return gcd(n,m%n);
}

int main()
{
    int t;
    int a,b,ans,ans1;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&a,&b);
        ans=gcd(a,b);
        ans1=(a*b)/ans;
        printf("%d %d
",ans,ans1);
    }
    return 0;
}
