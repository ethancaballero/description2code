#include<iostream>
#include<algorithm>
using namespace std;
int gcd(int a, int b)
{
	if (b == 0)
		return a;
	return gcd(b, a % b);
}
int main()
{
	int t;
	int a, b, g;
	cin >> t;
	while (t--)
	{
		cin >> a >> b;
		g = gcd(max(a, b), min(a, b));
		cout << g << " " << (a * b) / g << "
";
	}
    return 0;
}
