#include<bits/stdc++.h>
using namespace std;
int main()
{
 int t,a,b;
 cin>>t;
 while(t--)
 {
     int gcd=1;
     cin>>a>>b;
     for(int i=1;i<1000;++i)
                {
                                if((a%i==0)&&(b%i==0))
                                gcd=i;
                }

                cout<<gcd<<"	"<<(a*b)/gcd<<endl;
 }
 return 0;
}
