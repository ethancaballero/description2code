#include<iostream>]
#include<string.h>
using namespace std;
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
      char str1[101],str2[101];
      cin>>str1;
      cin>>str2;
      int min=0,max=0;
      for(int i=0;i<strlen(str1);i++)
	  {
	  	  if(str1[i]=='?'||str2[i]=='?')
	  	  max++;
	  	  else if(str1[i]!=str2[i])
	  	  min++,max++;
		 }   
		 cout<<min<<" "<<max<<"
";
	}
}
