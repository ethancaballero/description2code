#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
#include<vector>

using namespace std;

int main()
{
    int t;
    char str1[105],str2[105];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",&str1);
        scanf("%s",&str2);
    int n=strlen(str1);
    int count=0,curr=0;
    for(int i=0;i<n;i++)
    {
        if(str1[i]=='?' || str2[i]=='?')
            count++;
        else if(str1[i]==str2[i])
            curr++;
    }
    printf("%d %d
",n-curr-count,n-curr);
   }
return 0;
}
