#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
	    char a[102],b[102];
	    int min=0,max=0;
	    scanf("%s%s",a,b);
	    int len=strlen(a);
	    for(int i=0;i<len;i++) {
	       if (a[i]=='?' || b[i]=='?') {
	           max++;
	       } else if(a[i]!=b[i]) {
	           min++;
	           max++;
	       }
	    }
	    printf("%d %d
",min,max);
	}
	return 0;
}
