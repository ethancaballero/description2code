T = input()
for cas in xrange(1,T+1):
    N, K = map(int, raw_input().strip().split())

    forgotten = raw_input().split()

    allWordsfromPhrases = []
    for i in xrange(K):
     allWordsfromPhrases += raw_input().split()

    ans = ""
    for i in xrange(N):
     if forgotten[i] in allWordsfromPhrases: 
      ans += "YES "
     else: 
      ans += "NO "
    print ans
