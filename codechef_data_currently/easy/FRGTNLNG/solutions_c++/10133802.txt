#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,j,l;
    scanf("%d",&t);
    while(t--)
    {
        char a[100][6];
        int b[100]={0};
        char c[6];
        int ls;
        int n,k;
        cin>>n>>k;
        for(i=0;i<n;i++)
        {
            scanf("%s",a[i]);
        }
        for(i=0;i<k;i++)
        {
            cin>>ls;
            for(j=0;j<ls;j++)
            {
                scanf("%s",c);
                for(l=0;l<n;l++)
                {
                    if(strcmp(a[l],c)==0)
                        b[l]=1;
                }
            }
        }
        /*for(i=0;i<n;i++)
        {
            printf("%s %d
",a[i],b[i]);
        }*/
        for(i=0;i<n;i++)
        {
            if(b[i]==1)
                cout<<"YES ";
            else
                cout<<"NO ";
        }
        cout<<endl;
    }
    return 0;
}
