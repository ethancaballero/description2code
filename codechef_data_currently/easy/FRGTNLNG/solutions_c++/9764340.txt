#include<iostream>
#include<string.h>
using namespace std;

bool check(string a,string b)
{
    int l1=a.length();
    int l2=b.length();
    if(l1!=l2)
        return false;
    for(int i=0;i<l1;i++)
        if(a[i]!=b[i])
        return false;
    return true;
}
int main()
{
  int t,n,k,l,m,j;
  cin>>t;
  while(t--)
  {
      cin>>n>>k;
      string words[n];
      for(int i=0;i<n;i++)
        cin>>words[i];
        string phrase[5020];
        int ind=0;
      while(k--)
      {
          cin>>l;

          for(j=0;j<l;j++){
            cin>>phrase[ind];
            ind++;
          }
      }

          for(int k=0;k<n;k++)
          {
              for(m=0;m<ind;m++)
              {
                  if(!check(words[k],phrase[m]))
                    continue;
                  if(check(words[k],phrase[m]))
                  {
                      cout<<"YES ";
                      break;
                  }
              }
              if(m>=ind)
              cout<<"NO ";
          }
          cout<<"
";

    }

}
