#include<iostream>
#include<string.h>
using namespace std;
int main()
{
   int n,t;
   cin>>t;
   while(t--)
   {
       int k;
       cin>>n>>k;
       int flag[n];
       for(int i=0;i<n;i++)
        flag[i]=-1;
       string words[n];
       for(int i=0;i<n;i++)
        cin>>words[i];
       while(k--)
       {
          int l;
          cin>>l;
          string para[l];
          for(int i=0;i<l;i++)
            cin>>para[i];
          for(int i=0;i<n;i++)
          {
              if(flag[i] == -1)
              {
                  for(int j=0;j<l;j++)
                  {
                      if(words[i].compare(para[j])==0)
                      {
                          flag[i]=1;
                           break;
                      }
                  }
              }
          }

       }

       for(int i=0;i<n;i++)
        if(flag[i]==1)
        cout<<"YES ";
       else
        cout<<"NO ";
       cout<<"
";
   }

}
