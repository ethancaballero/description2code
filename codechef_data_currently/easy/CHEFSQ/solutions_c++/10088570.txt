#include<stdio.h>
int main()
{
	int T,n,n1,i,j;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&n);
		int a[n];
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		scanf("%d",&n1);
		int b[n1];
		for(i=0;i<n1;i++)
			scanf("%d",&b[i]);
		for(i=0,j=0;i<n && j<n1;i++)
		{
			if(a[i]==b[j])
			{
			   // i=-1;
			    j++;
			}
		}
		if(j==n1)
			printf("Yes
");
		else
			printf("No
");
    }
	return 0;
}
