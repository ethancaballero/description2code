#include<iostream>
using namespace std;
int main()
{
   int t,n,m,i,j,flag;
   int a[100100],b[100100];
   cin>>t;
   while(t--)
   {
       flag=1;
       cin>>n;
       for(i=0;i<n;i++)
        cin>>a[i];
       cin>>m;
       for(i=0;i<m;i++)
        cin>>b[i];
      i=j=0;
       while(i<n)
       {
           if(a[i]==b[j])
            j++;
           i++;
           if(j==m)
           {
               flag=0;
               break;
           }

       }

        if(flag)
            cout<<"No
";
        else
            cout<<"Yes
";
   }
}
