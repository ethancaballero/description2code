#include <iostream>

using namespace std;

int main()
{
    int t,n,i,j,f,c;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        cin>>f;
        int b[f];
        for(i=0;i<f;i++)
            cin>>b[i];
        c=0;
        for(i=0;i<f;i++)
        {
            for(j=0;j<n;j++)
                if(b[i]==a[j])
                 {
                    c++;
                    break;
                 }

        }
        if(c==f)
            cout<<"Yes
";
        else
            cout<<"No
";

    }
    return 0;
}
