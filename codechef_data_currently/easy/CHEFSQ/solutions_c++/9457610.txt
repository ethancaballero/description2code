#include<iostream>
using namespace std;
typedef long long ll;
int main(){

    ll i,n,t,f,j;
    scanf("%lld",&t);
    while(t--){
        scanf("%lld",&n);
        ll a[n];
        for(i=0;i<n;++i) scanf("%lld",&a[i]);

        scanf("%lld",&f);
        ll b[f];
        for(i=0;i<f;++i) scanf("%lld",&b[i]);
        ll count=0;
        j=0;
        for(i=0;i<f;++i) {
            for(;j<n;++j){
                int fd=0;
                if(b[i]==a[j]){
                    fd=1;
                    count++;
                    break;
                }
                if(fd==1) break;
            }
        }
        if(count==f) cout<<"Yes
";
        else cout<<"No
";
    }


    return 0;
}
