#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%i",&t);
    while(t--)
    {
        int l1,n[1000000],l2,f[1000000],c=0;
        scanf("%i",&l1);
        for(int i=0;i<l1;i++)
        {
            scanf("%i",&n[i]);
        }
        scanf("%i",&l2);
        for(int i=0;i<l2;i++)
        {
            scanf("%i",&f[i]);
        }
        int i,j;
        for(j=0,i=0;i<l1;i++)
        {
            if(n[i]==f[j])
                j++;
            if(j==l2)
            {
                c=1;
                break;
            }
        }
        if(c==1)
            printf("Yes
");
        else
            printf("No
");
    }
	return 0;
}
