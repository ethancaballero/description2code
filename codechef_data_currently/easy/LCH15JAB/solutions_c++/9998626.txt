#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
    while(t--)
	{
	    char a[52];
	    int max=0, index;
	    int b[256] = {0};
	    scanf("%s",a);
	    int len = strlen(a);
	    for(int i=0;i<len;i++){
	        b[a[i]]++;
	        if(b[a[i]]>max){
	            max=b[a[i]];
	            index=a[i];
	        }
	    }
	    int sum=0;
	    for(int i=0;i<256;i++){
	        if(index!=i){
	            sum+=b[i];
	        }
	    }
	    if(sum==max){
	        printf("YES
");
	    } else {
	        printf("NO
");
	    }
	}
	return 0;
}
