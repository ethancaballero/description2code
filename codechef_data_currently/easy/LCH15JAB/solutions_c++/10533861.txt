#include <bits/stdc++.h>
using namespace std;
 
int main() 
{
	int t;
	cin>>t;
	for(int x=0;x<t;x++)
	{
	    char ar[55];
	    cin>>ar;
	    int n=strlen(ar);
	    int freq[26]={0};
	    for(int i=0;i<n;i++)
	    {
	        freq[ar[i]-97]++;
	    }
	    int flag=0;
	    for(int i=0;i<26;i++)
	    {
	        if(freq[i]!=0)
	        {
	            long long sum=0;
    	        for(int j=0;j<26;j++)
    	        {
    	            if(freq[j]!=0)
    	            {
    	                if(i!=j)
    	                sum=sum+freq[j];
    	            }
    	        }
    	        if(freq[i]==sum)
    	        {
    	            flag=1;
    	            cout<<"YES
";
    	            break;
    	        }
	        }
	    }
	    if(flag==0)
	    {
	        cout<<"NO
";
	    }
	}
	return 0;
}
