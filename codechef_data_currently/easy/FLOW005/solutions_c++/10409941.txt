#include <iostream>
#include<math.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	int money[6]={1,2,5,10,50,100};
	for(int i=0;i<t;i++)
	{
	    int n,j=5,count=0;
	    cin>>n;
	    while(n!=0)
	    {
	        if(n>=money[j])
	        {
	            n=n-money[j];
	            count++;
	        }
	        else{j--;}
	    }
	    cout<<count<<"
";
	}
	return 0;
}
