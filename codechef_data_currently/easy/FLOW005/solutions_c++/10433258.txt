#include <bits/stdc++.h>
long long int t, n, a[6], i, cou;
using namespace std;
int main()
{
	a[0] = 1;
	a[1] = 2;
	a[2] = 5;
	a[3] = 10;
	a[4] = 50;
	a[5] = 100;
	scanf("%lld",&t);
	while(t--) {
		scanf("%lld",&n);
		i = 5;
		cou = 0;
		while(n != 0) {
			if (n >= a[i]) {
				n = n - a[i];
				i = 5;
				cou++;
			}else{
				i--;
			}
		}
		printf("%lld
",cou);
	}
	return 0;
}
