#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
   int x,y,m,t,a[15],b[25],mul,sum,i,j;
   cin>>t;
   while(t--)
   {
       cin>>m>>x>>y;
       mul=x*y;
       sum=0;
       for(i=0;i<m;i++)
        cin>>a[i];
        sort(a,a+m);
       b[0]=0;
       for(i=0,j=1;i<m;i++,j+=2)
       {
           b[j]=a[i]-mul;
           b[j+1]=a[i]+mul;
       }
       b[j]=100;
       for(i=0;i<j;i+=2)
       {
           if((b[i+1]-b[i]-1)>0)
            sum=sum+(b[i+1]-b[i]-1);

            if(((i+1)==j) &&((b[i+1]-b[i])>0))
            {
                sum+=1;
            }
       }
       cout<<sum<<"
";
   }

}
