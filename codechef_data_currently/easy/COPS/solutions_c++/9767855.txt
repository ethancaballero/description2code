#include<iostream>
using namespace std;
int t,i, cop[110],m,x,y,p,j,safe;

int main()
{
	cin>>t;
	while(t--)
	{
		for(i=1;i<=100;i++) cop[i]=0;
		cout<<"
";
		cin>>m>>x>>y;
		cout<<"
";
		for(i=1;i<=m;i++)
		{
			cin>>p;
			cop[p]=1;
			for(j=p-1;j>=p-(x*y);j--)
			{
				if(j>=1) cop[j]=1;
			}
			for(j=p+1;j<=p+(x*y);j++)
			{
				if(j<=100) cop[j]=1;
			}
		}
		safe=0;
		for(i=1;i<=100;i++)
		{
			if(!cop[i]) safe++;
		}
		cout<<"
"<<safe;
	}
}
