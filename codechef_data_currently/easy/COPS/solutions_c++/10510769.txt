#include<stdio.h>
#include<string.h>
#include<assert.h>
#include<vector>
#include<set>
#include<algorithm>
#include<iostream>

#define pb push_back
using namespace std;

void solve()
{
	int number_of_cops,x,y;
	vector<int> location;
	scanf("%d%d%d",&number_of_cops,&x,&y);
	for(int i=0;i<number_of_cops;i++){
		int x1;
		scanf("%d",&x1);
		location.pb(x1);
	}
	sort(location.begin(),location.end());
	int answer[110];
	memset(answer,0,sizeof(answer));
	for(int i=0;i<number_of_cops;i++){
		for(int j=location[i];(j<=(location[i]+x*y)) && j<=100;j++)	answer[j] = 1;
		for(int j=location[i];(j>=(location[i]-x*y)) && j>=1;j--)	answer[j] = 1;
	}
	int print  = 0;
	for(int i=1;i<=100;i++){
		if( answer[i] == 0 ){
			print++;
		}
	}
	printf("%d
",print);
}

int main()
{
	int T;
	scanf("%d",&T);
	for(int i=1;i<=T;i++){
		solve();
	}
	return 0;
}
