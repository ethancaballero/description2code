#include<bits/stdc++.h>
#include<iostream>
using namespace std;
int main()
{
    int t,a[105]={0},ar[20]={0};
	cin>>t;
	while(t--)
	{
	    int i,j,m,x,y,count=0;
	    cin>>m;
	    cin>>x;
	    cin>>y;
        for(i=0;i<m;i++)
        cin>>ar[i];
        for(i=0;i<m;i++)
        {
        if(ar[i]-(x*y)>0&&ar[i]+(x*y)<101)
         {
         for(j=ar[i]-(x*y);j<=ar[i]+(x*y);j++)
         a[j]++;
         //printf("a ");
         //printf("%d %d
",ar[i]-(x*y),ar[i]+(x*y));
         }
        else if(ar[i]-(x*y)<=0&&ar[i]+(x*y)<101)
         {
         for(j=1;j<=ar[i]+(x*y);j++)
         a[j]++;
         //printf("b ");
         //printf("1 %d
",ar[i]+(x*y));
         }
         else if(ar[i]-(x*y)>0&&ar[i]+(x*y)>100)
         {
         for(j=ar[i]-(x*y);j<=100;j++)
         a[j]++;
         //printf("c ");
         //printf("%d 100
",ar[i]-(x*y));
         }
         else
         {
         for(j=1;j<=100;j++)
         a[j]++;
         //printf("d ");
         //printf("1 100
");
         }
        }
        for(i=1;i<=100;i++)
        {
        //printf("%d=%d
",i,a[i]);
        if(a[i]==0)
        count++;
        a[i]=0;
	    }
	    cout<<count<<endl;
	    count=0;
	}
return 0;
}
