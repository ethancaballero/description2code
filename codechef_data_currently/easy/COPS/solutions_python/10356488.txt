t=int(raw_input())
while t>0 :
    t-=1
    m,x,y = map(int,raw_input().strip().split())
    a = map(int,raw_input().strip().split())
    a.sort()
    counter = 0
    if a[0] > x*y + 1 :
        counter += a[0] - x*y - 1
    for i in xrange(m-1) :
        if a[i+1] - a[i] > 2*x*y + 1 :
            counter += a[i+1]-a[i] - 2*x*y - 1
    if a[m-1] + x*y < 100 :
        counter += 100 - a[m-1] - x*y 
    print counter
