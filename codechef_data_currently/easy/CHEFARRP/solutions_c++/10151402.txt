#include<iostream>
using namespace std;
main()
{
    int t,product,sum,c,i,j,k,n;
    cin>>t;
    for(i=0;i<t;i++)
    {
        c=0;
        cin>>n;
        int a[n];
        for(j=0;j<n;j++)
        cin>>a[j];
        for(j=0;j<n;j++)
        {

            sum=0;
            product=1;
            for(k=j;k<n;k++)
            {
                product*=a[k];
                sum+=a[k];
                if(sum==product)
                    c++;
            }
        }
            cout<<"
"<<c;


    }


}
