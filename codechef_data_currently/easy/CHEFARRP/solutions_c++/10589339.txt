#include<bits/c++io.h>
#include<iostream>
#define max 100000
using namespace std;
int main()
{
    int t,n,sum=0,pro=1,count=0;
    cin>>t;
    while(t--)
    {
        int a[max],i,j;
        cin>>n;
        for(i=0;i<n;i++)
        cin>>a[i];
        for(i=0;i<n;i++)
        {
            for(j=i;j<n;j++)
            {
                //printf("a[j]=%d ",a[j]);
                sum+=a[j];
                pro=pro*a[j];
                //printf("sum=%d pro=%d
",sum,pro);
                if(sum==pro)
                count++;
            }
        sum=0;
        pro=1;
        //printf("
");
        }
        cout<<count<<endl;
        count=0;
    }
}
