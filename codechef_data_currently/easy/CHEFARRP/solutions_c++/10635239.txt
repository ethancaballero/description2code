#include <iostream>
using namespace std;
#include <vector>
#define ll long long

ll su(vector<ll> z,ll x,ll y)
{   ll s=0;
    for(ll i=x;i<=y;++i)
        s+=z[i];
    return s;
}

ll pro(vector<ll> z,ll x,ll y)
{   ll s=1;
    for(ll i=x;i<=y;++i)
        s*=z[i];
    return s;
}

int main()
{   ll t;
    cin>>t;
    while(t--)
    {   ll n,i,j;
        vector<ll> z;
        cin>>n;
        for(i=0;i<n;++i)
        {   ll q;
            cin>>q;
            z.push_back(q);
        }
        ll s=0;
        for(i=0;i<n;++i)
            for(j=i;j<n;++j)
            {   if(su(z,i,j)==pro(z,i,j))
                    s++;
            }
        cout<<s<<"
";
    }
    return 0;
}
