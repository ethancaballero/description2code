#include <bits/stdc++.h>
using namespace std;

int solve(int coins, int people) {
	int mod = 0;
	for (int i = 1; i <= people; ++i)
	{
		if(coins % i > mod) {
			mod = coins%i;
		}
	}
	return mod;
}

int main(int argc, char const *argv[])
{
	ios::sync_with_stdio(false);

	int tc;
	cin>>tc;
	while(tc--) {
		int n,k;
		cin>> n >> k;
		cout << solve(n,k) << "
";
	}
	return 0;
}
