#include<stdio.h>
#include<iostream>
using namespace std;

int main(void)
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int a[n],b[n],i,count=0;
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		for(i=0;i<n;i++)
		{
			scanf("%d",&b[i]);
			if(i==0)
			{
				if(b[i]<=a[i])	count++;
			}
			else
			{
				if(b[i]<=(a[i]-a[i-1]))	count++;
			}
			
		}
		
		printf("%d
",count);
			
	}
	return 0;
}
