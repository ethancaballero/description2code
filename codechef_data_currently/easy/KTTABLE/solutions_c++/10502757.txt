#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,count=0;
		cin>>n;
		ll a[n+1],b[n+1];
		for(ll i=1;i<=n;i++)
		{
			cin>>a[i];
		}
		a[0]=0;
		for(ll i=1;i<=n;i++)
		{
			cin>>b[i];
			if(b[i]<=(a[i]-a[i-1]))
				count++;
		}
		cout<<count<<"
";
	}
	return 0;
}
