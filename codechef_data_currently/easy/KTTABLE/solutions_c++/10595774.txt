#include<iostream>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while (t--)
    {
        int n, i, out = 0;
        long long int *a, *b;
        cin >> n;
        a = new long long int [n];
        for (i=0; i<n; i++)
        cin >> a[i]; 
        b = new long long int [n];
        for (i=0; i<n; i++)
        cin >> b[i];
        for (i=0; i<n; i++)
        {
            if (i == 0)
            {
                if (a[0]>=b[0])
                out += 1;
            }
            else if (i>0)
            {
                if (b[i]<=(a[i]-a[i-1]))
                out += 1;
            }
        }
        cout << out << "
";
    }
}
