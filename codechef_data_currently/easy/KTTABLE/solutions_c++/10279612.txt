#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    int t,n,i;
    long long int c;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        long long int a[n],b[n];
        for(i=0;i<n;i++)
        {
            scanf("%lld",&a[i]);
        }
        for(i=0;i<n;i++)
        {
            scanf("%lld",&b[i]);
        }
        c=(a[0]>=b[0])?1:0;
        for(i=1;i<n;i++)
        {
            if(a[i]-a[i-1]>=b[i])
                c++;
        }
        printf("%lld
",c);

    }

}
