#include<iostream>
#include<cstdio>
int main()
{
    int t;
    long long int sal;
    float hra,da;
    scanf("%d",&t);
    while(t--)
    {

        scanf("%lld",&sal);
        if(sal<1500)
        {
            hra=sal*0.1;
            da=sal*0.9;
            printf("%g
",sal+hra+da);
        }
        else
        {
            hra=500.0;
            da=sal*0.98;
            printf("%g
",sal+hra+da);
        }
    }
    return 0;
}
