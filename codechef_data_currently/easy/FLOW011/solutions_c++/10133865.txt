//gross salary
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
            int b;
            cin>>b;
            if(b<1500)
            {
                printf("%g
",b+b*0.9+b*0.1);
            }
            else
            {
                printf("%g
",b+b*0.98+500);
            }
    }
    return 0;
}

/*If his basic salary is less than Rs. 1500, then HRA = 10% of base salary and DA = 90% of basic salary.
If his salary is either equal to or above Rs. 1500, then HRA = Rs. 500 and DA = 98% of basic salary*/
