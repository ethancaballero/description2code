#include <iostream>
#include<stdio.h>
using namespace std;

int findNum(int arr[], int lb, int ub, int p){
        int m = (lb+ub)/2;

        if(lb==ub || arr[m] == p)
                return arr[m];

        else if(arr[m]<p && arr[m+1] <= p && m+1 <= ub)
                return findNum(arr, m+1, ub, p);

        else if (arr[m]> p && m-1>=lb)
                return findNum(arr, lb, m-1, p);
        else
                return arr[m];
}

int main()
{
        int t, p, d, c = 0;
        int arr[] = {1, 2, 4, 8, 16, 32, 64, 128, 256, 512, 1024, 2048};

        scanf("%d", &t);
        for(int i = 0; i<t; i++){
                scanf("%d", &p);
                c = 0;
                while(p){
                        d = findNum(arr, 0, 11, p);
                        c += p/d;
                        p = p%d;
                }

                printf("%d
", c);

                //printf("%d
", findNum(arr, 0, 11, p));

        }
    return 0;
}
