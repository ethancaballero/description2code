#include <iostream>
#include<stdio.h>

using namespace std;

int main()
{
        int t, num, sum;

        scanf("%d", &t);
        while(t--){
                sum = 0;
                scanf("%d", &num);

                while(num){
                    sum = sum + num%10;
                    num/=10;
                }

                printf("%d
", sum);
        }
    return 0;
}
