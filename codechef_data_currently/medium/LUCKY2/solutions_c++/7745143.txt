LL dp[1005][1005][3],mod=1000000007,lch[1005];

int check(int x)
{
    if(x == 0)return 0;
    while(x)
    {
        int my = x%10;
        if(my!=4 && my!=7)return 0;
        x/=10;
    }
    return 1;
}

void prework()
{
    int i;
    for(i=0; i<=1000; i++)lch[i] = check(i);
}

{

    prework();
    scanf("%d", &t);
    while(t--)
    {
        scanf("%s %s", l, r);
        ll = strlen(l);
        rl = strlen(r);
        int len = strlen(l);
        for(i=len-1; i>=0; i--)
        {
            if(l[i] == '0')

                l[i] = '9';
            }
            else
            {
                l[i] = l[i]-1;

        LL ans = mx - mn;

}
