#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,flag=0,flag1=0,i,j;
	char r[1001],s[1001];
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		int a[26]={0},b[26]={0};
		flag=0,flag1=0;
		scanf("%s%s",r,s);
		for(j=0;j<strlen(r);j++)
			a[r[j]-97]+=1;
		for(j=0;j<strlen(s);j++)
			b[s[j]-97]+=1;
		for(j=0;j<26;j++)
		{
		    if(a[j]!=b[j] && a[j]>0 && b[j]>0)
				flag1=1;
			else if((a[j] && !b[j] ) || (!a[j]&&b[j]))
				flag=1;
		}
		if(flag1 == 1 && flag!=1)
			cout<<"NO
";
		else
			cout<<"YES
";
	}	
return 0;
}
