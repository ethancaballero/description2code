from sys import stdin, stdout
t = int(stdin.readline())
while t:
	t -= 1
	n = int(stdin.readline())
	m = map(int, stdin.readline().strip().split(' '))
	m.sort()
	i = n
	num = 0
	ans = 0
	while i:
		if num%4 == 0 or num%4 == 1:
			ans += m[i-1]
		i -= 1
		num += 1
		num %= 4
	stdout.write(str(ans)+'
')
