#include <iostream>
using namespace std;
#include <vector>
#include <algorithm>
#define ll long long

int main()
{   ll t;
    cin>>t;
    while(t--)
    {   ll n,q,s=0,i;
        cin>>n;
        vector<ll> z;
        for(i=0;i<n;++i)
        {   cin>>q;
            z.push_back(q);
        }
        sort(z.begin(),z.end());
        for(i=0;i<n;++i)
        {   if(i%4==0 || i%4==1)
                s+=z[n-1-i];
        }
        cout<<s<<"
";
    }
    return 0;
}
