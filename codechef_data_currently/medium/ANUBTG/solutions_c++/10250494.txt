#include<stdio.h>
#include<vector>
#include<algorithm>

using namespace std;

int main()
{
    int test,i,sum,temp,n;
    vector <int> A;

    scanf("%d",&test);

    while (test > 0) {

        scanf("%d",&n);

        for (i = 0; i < n; i++) {
            scanf("%d",&temp);
            A.push_back(temp);
        }

        sort(A.begin(),A.end());

        i = n-1;
        sum = 0;

        while (i > 0) {
            sum += A[i] + A[i-1];
            i-=4;
        }

        if (i == 0) {
            sum += A[0];
        }

        printf("%d
",sum);

        A.clear();

        test--;
    }
   return 0;
}
