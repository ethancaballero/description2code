#include<iostream>
#include<cstring>
#include<cstdio>
#include<vector>
#include<cmath>
#include<cstdlib>
#include<algorithm>

using namespace std;

int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        vector<int> vec;
        int num;
        for(int i=0;i<n;i++)
        {
            scanf("%d",&num);
            vec.push_back(num);
        }
        sort(vec.begin(),vec.end());
        long long int cost=0;
        for(int i=n-1;i>=0;i=i-4)
          {
              cost=cost+vec[i];
              if(i)
                cost=cost+vec[i-1];

          }
        printf("%lld
",cost);
    }
    return 0;
}
