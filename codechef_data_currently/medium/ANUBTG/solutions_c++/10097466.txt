#include<iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
void bubble(int *a, int n)
{
    int j,k,temp;
    for(k=0;k<n-1;k++)
        for(j=0;j<n-k-1;j++)
        if(a[j]<a[j+1])
    {
        temp=a[j];
        a[j]=a[j+1];
        a[j+1]=temp;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,*a,i,cost=0;
        cin>>n;
        a=(int*)malloc(sizeof(int)*n);
        for(i=0;i<n;i++)
            cin>>a[i];
        bubble(a,n);
        if(n>4)
        {
            for(i=0;i<n;i+=4)
                {
                    cost+=(a[i]);
                    if(i+1<n)
                        cost+=a[i+1];
                }
        }
        else if(n<=4 &&n>=2)
            cost+=(a[0]+a[1]);
        else
            cost+=a[0];
        cout<<cost<<"
";
    }
    return 0;
}
