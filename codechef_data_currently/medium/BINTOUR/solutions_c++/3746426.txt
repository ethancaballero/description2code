#include<cstdio>
#define MOD 1000000009ll
using namespace std;
int n , k;
long long extendedEuclid( long long a, long long b, long long *x, long long *y)	{
    long long t, d;
    if(b == 0)	{
        *x = 1; *y = 0; return a;
    }
    d = extendedEuclid(b, a % b, x, y);
    t = *x; *x = *y; *y = t - (a/b)*(*y);
    return d;
}
int main(){
    // Input
    // freopen("input.txt", "r", stdin);
    // freopen("output.txt", "w", stdout);

    long long base, c, x, y;
    int i, m, r;

    scanf("%d", &k);
    n = 1 << k;
    long long *f = new long long[(n>>1)+1];
    f[1] = 1;
    for(i=2; i<=(n>>1); ++i){
        f[i] = (i*f[i-1])%MOD;
    }
    for(i=1; i<n/2; ++i)
        printf("0
");
    base = (f[n>>1] * f[n>>1])%MOD;
    base = (base<<1)%MOD;
    r = (n>>1) - 1;
    c = 1;
    for(m=(n>>1); m<=n; ++m){
        printf("%lld
", (c*base)%MOD);
        extendedEuclid(m-r, MOD, &x, &y);
        c = (m * c)%MOD;
        c = (c * (x + MOD)%MOD)%MOD;
    }
}
