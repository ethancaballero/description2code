#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll mod=1e9+9;
ll fact[1100000],ifact[1100000];
ll exp(ll a,ll b)
{
    ll result=1;
    while(b)
    {
        if(b&1)
        {
            result*=a;
            result%=mod;
        }
        a*=a;
        a%=mod;
        b>>=1;
    }
    return result;
}
void cal_fac(ll n)
{
    fact[0]=1;ifact[0]=1;
    for(ll i=1;i<=n;i++)
    {
        fact[i]=(fact[i-1]*i)%mod;
        ifact[i]=exp(fact[i],mod-2);
       // cout<<fact[i]<<" "<<ifact[i]<<endl;
    }
}

int main()
{
	//freopen("i.txt","r",stdin);
    ll k;
    scanf("%lld",&k);
    ll n;
    n=1<<k;
    ll ans;
    cal_fac(n);
    //cout<<n<<endl;
    for(int i=1;i<=n;i++)
    {
        if(i<(n/2-1))
        {
            printf("0
");
            continue;
        }
        //cout<<":P"<<endl;
        ans=(fact[i-1]*ifact[i-n/2])%mod;
        ans=(ans*2)%mod;
        ans=(ans*ifact[n/2-1])%mod;
        ans=(ans*fact[n/2])%mod;
        ans=(ans*fact[n/2])%mod;
        printf("%lld
",ans);

    }
	return 0;
}
