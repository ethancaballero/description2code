#include<bits/stdc++.h>
#define M 1000000009

using namespace std;

int inverse[1000006];
int fact[1000006];
int res[1000006];


int degree(long long a, long long k) {
  long long res = 1;
  long long cur = a;

  while (k) {
    if (k%2) {
      res = (res * cur)%M;
    }
    k /= 2;
    cur = (cur * cur) % M;
  }
  return (int)res;
}

int calc_inv(int n)
{
    for(int i=1;i<=n;i++)
        inverse[i]=degree(i,M-2);
    return 0;
}

int ncr(int n)
{
    res[n/2-1]=1;
    int j=n/2;
    for(int i=n/2;i<n;i++)
    {
        res[j]=(((res[j-1]*(long long)inverse[i-n/2+1])%M)*i)%M;
        j++;
    }
}
