#include<iostream>
#include<cmath>
using namespace std;
typedef long long int LL;
//typedef __int64 LL;
 
LL MOD = 1000000009;
LL fact[1100000];
 
LL bigmod(LL a, LL b)
{
	if(b == 0) return 1;
	LL x = bigmod(a, b / 2);
	x = (x * x) % MOD;
	if(b & 1) x = (x * a) % MOD;
	return x;
}
 
int main()
{
	LL ans;
	int k, n, i;
 
	scanf("%d", &k);
	n = 1<<k;
 
	fact[0] = 1;
	for(i = 1; i <= n; i++)
		fact[i] = (fact[i - 1] * i) % MOD;
 
	for(i = 1; i <= n; i++)
	{
		if(i < n/2 - 1) 
		{
			printf("0
");
			continue;
		}
 
		ans = (fact[i - 1] * bigmod(fact[n/2 - 1], MOD - 2)) % MOD;
		ans = (ans * bigmod(fact[i - n/2], MOD - 2)) % MOD;
		ans = (ans * 2) % MOD;
		ans = (ans * fact[n/2]) % MOD;
		ans = (ans * fact[n/2]) % MOD;
 
		printf("%lld
", ans);
	}
 
	return 0;
}
