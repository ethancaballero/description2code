#include <cctype>
#include <cstdio>
#include <cmath>
 
using namespace std;
 
const int BUFSIZE = 4096;
const char MINUS = '-';
const char ZEROCHAR = '0';
const char DOT = '.';
const double TEN = 10;
 
char r_buffer[BUFSIZE];
int r_bufpos = BUFSIZE;
 
double readdouble()
{
    bool dotright = false;
    double n = 0.0;
    double precision = 1.0;
    int sign = 1;
    for (; r_bufpos < BUFSIZE && r_buffer[r_bufpos] != MINUS
            && r_buffer[r_bufpos] != DOT && !isdigit(r_buffer[r_bufpos]);
            r_bufpos++);
    if (r_bufpos == BUFSIZE)
    {
        r_bufpos = 0;
        fread(r_buffer, 1, BUFSIZE, stdin);
        for (; r_bufpos < BUFSIZE && r_buffer[r_bufpos] != MINUS
                && r_buffer[r_bufpos] != DOT && !isdigit(r_buffer[r_bufpos]);
                r_bufpos++);
    }
    if (r_buffer[r_bufpos] == MINUS)
    {
        sign = -1;
        r_bufpos += 1;
    }
    for (; r_bufpos < BUFSIZE; r_bufpos++)
    {
        if (isdigit(r_buffer[r_bufpos]))
        {
            n = TEN * n + r_buffer[r_bufpos] - ZEROCHAR;
            if (dotright)
                precision *= TEN;
        }
        else if (r_buffer[r_bufpos] == DOT)
        {
            dotright = true;
        }
        else
            break;
    }
    if (r_bufpos == BUFSIZE)
    {
        r_bufpos = 0;
        fread(r_buffer, 1, BUFSIZE, stdin);
        for (; r_bufpos < BUFSIZE; r_bufpos++)
        {
            if (isdigit(r_buffer[r_bufpos]))
            {
                n = TEN * n + r_buffer[r_bufpos] - ZEROCHAR;
                if (dotright)
                    precision *= TEN;
            }
            else if (r_buffer[r_bufpos] == DOT)
            {
                dotright = true;
            }
            else
                break;
        }
    }
    return sign * n / precision;
}
