typedef long long ll;
using namespace std;

const int wt[] = {10,18,16,14,12,10,8,6,4,2};
const int wtbase[] = {9,17,15,13,11,9,7,5,3,1};
#define PRIME 1000007
#pragma warning (disable:4996)

int main () {
	int t,N,W;
	ll x[22][250][10];
	scanf ("%d
", &t);

	memset (x, 0LL, sizeof x);

	for (int i=1; i<10; i++)
		for (int j=0; j<10; j++)  x[2][abs(i-j)][j]++;
		
	for (int n=3; n<=20; n++) 
		for (int w=0; w<=200; w++) {
			for (int d=0; d<10; d++) {
				for (int i=0; i<10; i++) {
					int diff = abs(d-i);
					x[n][w+diff][d] += x[n-1][w][i];
					x[n][w+diff][d] %= PRIME;
				}
			}
		}

	while (t--) {
		scanf ("%d %d
", &N, &W);

		ll ans=0;
		for (int i=0; i<10; i++) { ans+=x[N][W][i]; ans%=PRIME; }
		printf ("%lld
", ans);
	}

	return 0;
}
