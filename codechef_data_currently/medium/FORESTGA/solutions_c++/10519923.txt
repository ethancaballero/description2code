ll H[100001],R[100001],N,W,L;

int totalwood(ll m) {
	ll sum  = 0;
	for (ll i = 0; i < N; i++) {
		if(H[i]+m*R[i] >= L){
			sum += H[i] + m*R[i];
			}
		if(sum >= W)	return 1;

	}
	return 0;
}

	ll mid;
	while (right-left>1) {
		mid = (left+right)/2;
		if(totalwood(mid))	right = mid;
