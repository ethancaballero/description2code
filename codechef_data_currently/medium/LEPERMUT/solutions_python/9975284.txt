from sys import stdin,stdout
t = int(stdin.readline())
for _ in xrange(t):
	n = int(stdin.readline())
	a = map(int,stdin.readline().split())
	k = 0
	for i in xrange(n-1):
		for j in xrange(i+1,n):
			if a[i] > a[j]:
				k += 1
	for i in xrange(n-1):
		if a[i] > a[i+1]:
			k -= 1
	if k == 0:
		stdout.write("YES")
	else:
		stdout.write("NO")
	stdout.write("
")
