inline int lower_search(int K, int X) {
    
    int i = divisors[K].size();

    for (int lg = 1 << 17; lg > 0; lg >>= 1) {
        if (i - lg >= 0 && divisors[K][i - lg] >= X) {
            i -= lg;
        }
    }
    return i;
}
inline int upper_search(int K, int Y) {
    int i = 0;
    int n = divisors[K].size();
    
    for (int lg = 1 << 17; lg > 0; lg >>= 1) {
        if (i + lg < n && divisors[K][i + lg] <= Y)
            i += lg;
    }
    return i;
}
