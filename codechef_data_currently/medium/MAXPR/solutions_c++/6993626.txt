vlong gcd ( vlong a, vlong b ) {
    a = abs ( a ); b = abs ( b );
    while ( b ) {
        a = a % b;
        swap ( a, b );
    }
    return a;
}
 
vlong bigmod ( vlong a, vlong p, vlong m ) {
    vlong res = 1, x = a % m;
    while ( p ) {
        if ( p & 1 ) res = ( res * x ) % m;
        x = ( x * x ) % m;
        p >>= 1;
    }
    return res;
}
