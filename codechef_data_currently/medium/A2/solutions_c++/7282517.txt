#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
	int t;
	for(scanf("%d", &t); t--; )
	{
		int k, a[1000001];
		scanf("%d", &k);
		for(int i=0; i<k; i++)
			scanf("%d", &a[i]);
		int stems=1;
		bool ans=true;
		for(int i=0; i<k-1; i++){
			if(a[i]>stems){
				ans=false;
				break;
			}
			stems=2*(stems-a[i]);
		}
		if(a[k-1]==stems && ans)
			printf("Yes
");
		else
			printf("No
");
		
	}
	return 0;
}
