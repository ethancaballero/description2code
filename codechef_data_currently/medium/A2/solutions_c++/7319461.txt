#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,k,i,f,z,w;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&k);
        long long int a[k+1];
        for(i=0;i<k;i++)
        {
            scanf("%lld",&a[i]);
        }
        f=0;
        z=1;
        for(i=0;i<k;i++)
        {
            if(a[i]>z)
            {
                f=1;
                break;
            }
            w=z-a[i];
            z=2*w;
        }
        if(z!=0)
            f=1;
        if(f==0)
            printf("Yes
");
        else
            printf("No
");
    }
    return 0;
}
