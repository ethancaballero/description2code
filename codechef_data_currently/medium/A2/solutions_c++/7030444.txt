#include <iostream>
#include<stdio.h>
using namespace std;

int main() {
	int t,d;
	long long int i,n,a[1000005],stem,stem_left;
	
	scanf("%d",&t);
	while(t--) {
		d=0;
		scanf("%lld",&n);
		for(i=0;i<n;i++){
			scanf("%lld",&a[i]);
		}
		if(a[0]>1)
		d=1;
		
		stem=1;
		stem_left=stem-a[0];
		for(i=1;i<n;i++) {
			stem=2*stem_left;
			if(a[i]>stem) {
				d=1;
				break;
			}
			stem_left=stem-a[i];
			if(i==n-1) {
				if(a[i]!=stem)
				d=1;
			}
		}
		if(n==1 && a[0]!=1)
		d=1;
		if(d==1)
		printf("No
");
		else
		printf("Yes
");
	}
	return 0;
}
