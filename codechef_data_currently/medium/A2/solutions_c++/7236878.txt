#include <bits/stdc++.h>
using namespace std;
int main()
{
	int a=0,b,c,d,i,j,l,sum,result,n,m,k,s,t,flag;
    scanf("%d",&t);
    while(t--)
    {
    	flag=0;
    	scanf("%d",&n);
    	sum=1;
    	for(i=1;i<=n;i++)
    	{
    		scanf("%d",&a);
    		if(a>sum)
    		{
    			flag=1;break;
    		}
    		sum-=a;
    		sum*=2;
    	}
    	if(sum==0&&flag==0)
    	printf("Yes
");
    	else
    	printf("No
");
    }
	return 0;
}
