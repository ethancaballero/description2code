int main(){
	int t;
	scan(t);
	while(t--){
		int n;
		bool ans;
		scan(n);
		int i;
		rep(i,0,n){
			scan(arr[i]);
		}
		ll branch = 1;
		i=0;
		while(i<n&&branch>0){
			if(arr[i]==0){
				branch*=2;
			}
			else{
				branch-=arr[i];
				branch*=2;
			}
			i++;
		}
		if(i<n){
			ans = false;
		}
		else{
			ans = (branch==0);
		}
		ans?printf("Yes

	}

}
