#include <iostream>
#include<cstdio>
using namespace std;
 
int main() {
	long long int check,count=0,stems,t,n,leaf;
	scanf("%lld",&t);
	while(t--)
	{
		stems=1;
		count=0;
		scanf("%lld",&n);
		check=n;
		 while(n--)
		 {
		 	scanf("%lld",&leaf);
		 	if(leaf==0&&n!=0)
		 	{
		 		stems=stems*2;
		 		count++;
		 	}
		 	else
		 	{  
		 		stems=stems-leaf;
		 		stems=stems*2;
		 		if(n!=0){
		 		if(stems>=0)
		 		 count++;
		 	    }
		 	    else
		 	    {
		 	    	if(stems==0)
		 	    	 count++;
		 	    }
		 	}
		 }
		 if(count==check)
		  printf("Yes
");
		 else
		  printf("No
");
		
	}
	return 0;
}
