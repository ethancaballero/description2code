#include<bits/stdc++.h>
int main()
{
    int t,n,count;
    int total,stem;
    scanf("%d",&t);
    while(t--)
    {
        count=0;
        scanf("%d",&n);
        int a[n];
        for(int i=0;i<n;i++)
            scanf("%d",&a[i]);
        total=1;
        for(int i=0;i<n;i++)
        {
            if(a[i]>total)
            {
                count=1;break;
            }
            stem=total-a[i];
            total=2*stem;
        }
        if(total!=0)
            count=1;
        !count?printf("Yes
"):printf("No
");

    }
    return 0;
}
