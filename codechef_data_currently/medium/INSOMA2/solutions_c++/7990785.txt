			int n;

			int ans=0;
			void backtrack(int x,int y,int i)
			{
				if(x>n-1 or y>n-1 or x<0 or y<0 or finish[i]!=ch[x][y])return ;
				if(i==finish.size()-1){ans++;return ;}
				if(ch[x][y]==finish[i])
				  backtrack(x-1,y-1,i+1);
				if(ch[x][y]==finish[i])
				  backtrack(x+1,y+1,i+1);
				if(ch[x][y]==finish[i])
				  backtrack(x-1,y+1,i+1);
				if(ch[x][y]==finish[i])
				  backtrack(x+1,y-1,i+1);
			}
