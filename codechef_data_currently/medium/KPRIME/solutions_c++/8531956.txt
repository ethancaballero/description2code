int p[MAXN];
int kp[MAXN][6];
void precompute()
{
	FOR(i,2,MAXN)
		if(p[i]==0)
			for(int j=i;j<MAXN;j+=i)
				p[j]++;
	FOR(i,2,MAXN)
	{
		FOR(j,1,MAXK)
		{
			kp[i][j]=kp[i-1][j]+(p[i]==j?1:0);
			//kp[j][i]=p[j];
}
}
}
int main()
{

	precompute();
	int t,a,b,k;
	scanf("%d",&t);
	assert(t>=1 && t<=10000);
	while(t--)
	{
		scanf("%d%d%d",&a,&b,&k);
		assert(a>=2 && a<MAXN);
		assert(b>=a && b<MAXN);
		assert(k>=1 && k<=MAXK);
		printf("%d
",kp[b][k]-kp[a-1][k]);
	}
	return 0;
}
