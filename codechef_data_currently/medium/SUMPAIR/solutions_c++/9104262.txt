#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,d;
        long long int sum=0;
        cin>>n>>d;
        vector<long long int>x;
        for(int i=0;i<n;i++)
        {
            long long int a;
            cin>>a;
            x.push_back(a);
        }
        sort(x.begin(),x.end());
        for(int i=n-1;i>0;i--)
        {
            if(x[i]-x[i-1]<d)
            {
                sum+=x[i]+x[i-1];
                i--;
            }
        }
        cout<<sum<<"
";
    }
}
