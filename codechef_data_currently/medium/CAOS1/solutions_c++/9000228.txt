inline int min(int x,int y){
	return x>y?y:x;
}
int getMinLRTB(char arr[50][50],int x,int y,int rc,int c){
	int l=0,r=0,t=0,bm=0;
	int a=x,b=y;
	while(a<rc-1&&arr[a+1][b]=='^'){
		r++;
		a++;
	}
	a = x;
	b = y;
	while(a>0&&arr[a-1][b]=='^'){
		l++;
		a--;
	}
	a = x;
	b = y;
	while(b<c-1&&arr[a][b+1]=='^'){
		bm++;
		b++;
	}
	a = x;
	b = y;
	while(b>0&&arr[a][b-1]=='^'){
		t++;
		b--;
	}
	a = min(l,r);
	b = min(t,bm);
	return min(a,b);
}
