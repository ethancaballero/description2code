#include<iostream>
using namespace std;
char a[100][100];
int n,m;
int main()
{
	int test,i,j,k,flag,count;	
	cin>>test;
	while(test--)
	{
		count=0;
		cin>>n>>m;		
		for(i=0;i<n;++i)	
		cin>>a[i];
				
		for (j=2;j<n-2;++j)
		{
			for (k=2;k<m-2;++k)
			if (a[j][k]=='^')
			if (a[j-1][k]=='^' && a[j-2][k]=='^' && a[j+1][k]=='^' && a[j+2][k]=='^' && a[j][k-1]=='^' && a[j][k-2]=='^' && a[j][k+1]=='^' && a[j][k+2]=='^')
            count++;
            
		}
		cout<<count<<"
";
	}
	return 0;
}
