#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define MOD 1000000007
#define MAX 100005
#define pb(a) push_back(a)
int MIN(int a,int b,int c,int d)
{
return min(min(min(a,b),c),d);
}
int main()
{
	int i,p,r,c,j,k,m,n,t,c1,c2,c3,c4,count;
	si(t);
	while(t--)
	{
		si(r);
		si(c);
		char str[r][c];
		int mat[r][c];

		f(i,0,r)
			f(j,0,c)
			scanf(" %c",&str[i][j]);
		count=0;

		f(i,0,r)
		{
			f(j,0,c)
			{
				if(i==0 || i==r-1 || j==0 || j==c-1)
					mat[i][j]=0;

				else if(str[i][j]!='#')
				{
					c1=0;
					c2=0;
					c3=0;
					c4=0;
					for(p=i-1;p>=0;p--)
					{
						if(str[p][j]=='^')
							c1++;

						else
							break;
					}

					f(p,i+1,r)
					{
						if(str[p][j]=='^')
							c2++;

						else
							break;
					}
					f(p,j+1,c)
					{
						if(str[i][p]=='^')
							c3++;

						else
							break;
					}
					for(p=j-1;p>=0;p--)
					{
						if(str[i][p]=='^')
							c4++;

						else
							break;
					}
					if(MIN(c1,c2,c3,c4)>=2)
						count++;
				}
			}
		}
		printf("%d
",count);
	}
}
