#include <stdio.h>
#define MAX 52
 
int main() {
    int t, row, col, i, j, count;
    char c[MAX][MAX];
    scanf("%d", &t);
    while (t--) {
        scanf("%d %d", &row, &col);
        count = 0;
        for (i=0; i<row; i++) scanf("%s", c[i]);
        for (i=2; i<row-2; i++) {
            for (j=2; j<col-2; j++) {
                if (c[i][j] == '^' && c[i][j+1] == '^' && c[i][j+2] == '^'
                && c[i+1][j] == '^' && c[i+2][j] == '^' && c[i][j-1] == '^'
                && c[i][j-2] == '^' && c[i-1][j] == '^' && c[i-2][j] == '^')
                    count++;
            }
        }
        printf("%d
", count);
    }
    return 0;
}
