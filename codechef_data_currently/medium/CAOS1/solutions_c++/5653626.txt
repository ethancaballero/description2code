#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
	int t;	
	char s[100];
	scanf("%d",&t);
	
	while(t--)
	{
		int n,m;
		int ans=0;
		scanf("%d %d",&n,&m);
		
		char arr[n+1][m+1];
		
		for(int i=1;i<=n;i++)
		{
			scanf("%s",s);
			for(int j=1;j<=m;j++)
				arr[i][j]=s[j-1];
		}
	
		for(int i=3;i<=(n-2);i++)
		{
			for(int j=3;j<=(m-2);j++)
			{
				int l=0,r=0,t=0,b=0;
				
				if(arr[i][j]=='^')
				{
					for(int k=j-1,ctr=2; ctr; k--,ctr--)
					{
						if(arr[i][k]=='^')
						l++;
					}
				
					for(int k=j+1,ctr=2; ctr; k++,ctr--)
					{
						if(arr[i][k]=='^')
						r++;
					}
				
					for(int k=i-1,ctr=2; ctr; k--,ctr--)
					{
						if(arr[k][j]=='^')
						t++;
					}
				
					for(int k=i+1,ctr=2; ctr; k++,ctr--)
					{
						if(arr[k][j]=='^')
						b++;
					}
				
					if( min(r,min(l,min(t,b))) >= 2)
					ans++;
				}
			}
		}
		
		printf("%d
",ans);
	}
return 0;
}
