#include <bits/stdc++.h>

using namespace std;
int main()
{
	int n;
	vector<int> ar;

	scanf("%d",&n);
	for (int i = 0; i < n; i++) {
		int x;
		scanf("%d",&x);
		ar.push_back(x);
	}
	sort(ar.begin(),ar.end());
	vector<long long> sum(n+1);
	sum[0] = 0;
	for (int i = 0; i < n; i++) {
		sum[i+1] = ar[i]+sum[i];
	}
	int q;
	scanf("%d",&q);
	while (q--) {
		int k;
		scanf("%d",&k);
		k++;
		printf("%lld
",sum[(n+k-1)/k]);
	}
	return 0;
}
