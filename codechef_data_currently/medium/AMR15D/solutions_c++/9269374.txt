#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

int main() {
	int n;scanf("%d",&n);
	int arr[n],cum[n];
	for(int i=0;i<n;i++){
	    scanf("%d",&arr[i]);
	}
	sort(arr,arr+n);
	cum[0]=arr[0];
	for(int i=1;i<n;i++){
	    cum[i]=arr[i]+cum[i-1];
	    //printf("%d",cum[i]);
	}
	int q; scanf("%d",&q);
	for(int i=0;i<q;i++){
	    int k; scanf("%d",&k);
	    int m=ceil((float)n/(k+1));
	  //cout << m << " ";
	    printf("%d
",cum[m-1]);
	}
	return 0;
}
