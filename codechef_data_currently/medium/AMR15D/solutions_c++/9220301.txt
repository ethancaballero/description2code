#include <math.h>
#include <algorithm>
#include <iostream>
#include <cstdio>
using namespace std;

int read()
{
	int n=0;
	register char ch=getchar_unlocked();
	while(ch<'0' || ch>'9') ch=getchar_unlocked();
	while(ch>='0' && ch<='9')
	{
		n=(n<<3)+(n<<1)+(ch-'0');
		ch=getchar_unlocked();
	}
	return n;
}

int main()
{
	int n;
	scanf("%d",&n);
	
	int i,j,key;

	int a[100009]={0};
	for(i=0;i<n;i++)
		a[i]=read();

	sort(a,a+n);

	int q;
	scanf("%d",&q);
	int k,count;
	long int sum;
	while(q--)
	{
		count=0;
		sum=0;
		scanf("%d",&k);

		int m=ceil((double)n/(1+k));
		for(i=0;i<m;i++) sum+=a[i];
		printf("%ld
",sum);
	}

	return 0;
}
