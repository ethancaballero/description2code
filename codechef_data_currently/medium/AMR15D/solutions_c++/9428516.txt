#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,j;
    scanf("%d",&n);
    int a[n+1];
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    int q,k,x;
    sort(a,a+n);
    for(i=1;i<n;i++)
        a[i]+=a[i-1];
    scanf("%d",&q);
    while(q--)
    {
        scanf("%d",&k);
        if(k==0)
            printf("%d
",a[n-1]);
        else if(k==1)
            printf("%d
",a[(n-1)/2]);
        else
        {
            x=n-1;
            j=-1;
            while(x>=0)
            {
                x-=(k+1);
                j++;
            }
            printf("%d
",a[j]);
        }
    }
    return 0;
}
