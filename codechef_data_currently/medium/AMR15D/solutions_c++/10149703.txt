#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int a[100005],s[100005];
int main()
{
	int n,i,q,k,c;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	sort(a,a+n);
	s[0]=a[0];
	for(i=1;i<n;i++)
		s[i]=s[i-1]+a[i];
	scanf("%d",&q);
	while(q--)
	{
		scanf("%d",&k);
		if(k==0)
		{
			printf("%d
",s[n-1]);
			continue;
		}
		c=n/(k+1);
		if(n%(k+1)!=0)
			c++;
		printf("%d
",s[c-1]);
	}
	return 0;
}
