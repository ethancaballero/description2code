#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin >> t;
	while(t--)	{
		int N;
		string s;
		cin >> N >> s;
		for(int i = 0;i < (1 << N);++i){
			int orgI = 0;
			for(int k = 0;k < N;++k){
				if(i & (1 << k))
					orgI += 1;
				if(k < N - 1)
					orgI <<= 1;	
			}
			cout << s[orgI];
		}
		cout << "
";
	}
	return 0;
}
