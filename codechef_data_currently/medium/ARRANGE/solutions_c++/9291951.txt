#include <iostream>
#include <stdio.h>
#include <math.h>
using namespace std;

int updown(int,int);
char ch[131072];

int main() {
	int t,k,i,j;
	string s;
	scanf("%d",&t);
	while(t--){
	    scanf("%d",&k);
	    cin>>s;
	    for(i=0;i<s.length();i++){
	        j=updown(i,k);
	        //printf("%d %d ",i,j);
	        ch[j]=s[i];
	    }
	    for(i=0;i<s.length();i++) printf("%c",ch[i]);
	    printf("
");
	}
	return 0;
}

int updown(int n,int k){
    int ans=0,i=k-1;
    while(n>0){
        ans+=pow(2,i)*(n%2);
        n/=2;
        i--;
    }
    return ans;
}
