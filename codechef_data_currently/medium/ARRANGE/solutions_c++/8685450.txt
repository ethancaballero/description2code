#include<stdio.h>
#include<algorithm>
#include<math.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int k,x,f,r=0,i,j,n;
		scanf("%lld",&k);
        x = pow(2,k);
        char a[x+1],b[x+1];
        scanf("%s",a);
        for(i = 0; i < x; i++)
        {
          long long int ar[k];
          f = i;
          r = 0;
          while(f!=0)
          {
          	ar[r++] = f%2;
          	f=f/2;
          }
          for(j = r; j < k; j++)
          ar[j] = 0;
          r=0;n=0;
          for(j = k-1; j >=0;j--)
          {
          	n = n+ar[j]*pow(2,r);
          	r++;
          }
          b[n] = a[i];
        }
        
		for(i = 0; i < x; i++)
		printf("%c",b[i]);
		printf("
");
	}
	return 0;
}
