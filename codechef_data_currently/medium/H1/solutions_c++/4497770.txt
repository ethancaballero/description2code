    void moveit(int p1,int p2){
    int diff=board[p2]-board[p1];
    posn=fromposn+diff*powers[p1]-diff*powers[p2];
    int posndiv10=posn/10;
    if(!found[posndiv10]){
    found[posndiv10]=found[fromposn/10]+1;
    q[back++]=posn;
    }
    }
