//main

int main()
{
    int n=scan_d();
    if(n>=68)
    {
    	putchar_unlocked('Y'),putchar_unlocked('e'),putchar_unlocked('s');
    	return 0;
    }
    long long unsigned arr[n];
    for(int i=0;i<n;i++)
    	arr[i]=scan_llu();
    bool flag=0;
    for(int i=0;i<n-3;i++)
    	for(int j=i+1;j<n-2;j++)
    		for(int k=j+1;k<n-1;k++)
    			for(int l=k+1;l<n;l++)
    				if(!(arr[i]^arr[j]^arr[k]^arr[l]))
    				{
    					flag=1;
    					break;
    				}
    if(flag)
    	putchar_unlocked('Y'),putchar_unlocked('e'),putchar_unlocked('s');
    else
    	putchar_unlocked('N'),putchar_unlocked('o');
    return 0;
}
