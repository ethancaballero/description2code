#include<stdio.h>
    #include<string.h>
    int main()
    {
    	int f[5001],i,t,a;
    	char str[1001];
    	scanf("%d",&t);
    	f[0]=0;
    	f[1]=1;
    	for(i=2;i<=5000;i++)
    	{
    		f[i]=f[i-1]+f[i-2];
    	}
    	while(t--)
    	{
    		scanf("%s",str);
    		a=0;
    		for(i=0;i<strlen(str);i++)
    		{
    		a=a*10+(str[i]-'0');
    		}
    		int flag =0;
    		for(i=0;i<5000;i++)
    		{
    			if(a==f[i])
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag)
    		printf("YES
");
    		else
    		printf("NO
");
    	}
    	return 0;
    }
