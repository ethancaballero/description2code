#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;

unsigned long long a[6666];

unsigned long long stoint(string s){
         int n = s.length();
         unsigned long long num=0;
         for(int i=0;i<n;i++)
         num=(num*10)+(s[i]-'0');
         return num;
         }

int main(){
    a[0]=0;a[1]=1;
    for(int i=2;i<6666;i++)
    a[i]=a[i-1]+a[i-2];
    int t,flag;
    string s;
    unsigned long long num;
    scanf("%d",&t);
    while(t--){
               cin>>s;
               num = stoint(s);
               flag=0; 
               for(int i=0;i<6666;i++){
                       if(num==a[i]){
                                     flag=1;
                                     break;
                                     }
                       }
                       if(flag)
                       printf("YES
");
                       else
                       printf("NO
");
               }
}
