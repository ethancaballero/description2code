#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t,n;
	cin>>t;
	while(t--){
        int arr[100005],sum=0;
        cin>>n;
        for(int i = 0; i < n ; i++){
            cin>>arr[i];
        }
        sort(arr,arr+n);
        for(int i = 1 ; i <n; i++){
            sum+= arr[i]-arr[0];
        }
        cout<<sum<<endl;
	}
	return 0;
}
