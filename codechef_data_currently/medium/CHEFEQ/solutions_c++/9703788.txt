#include<iostream>
#include<stdlib.h>
#include<bits/stdc++.h>
using namespace std;
long long int a[100001];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		memset(a,0,100001*sizeof(long long int));
		long int n,i,b;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>b;
			a[b]++;
		}
		long int max=0;
		for(i=1;i<100001;i++)
		{
			if(a[i]>max)
			max=a[i];
		}
		cout<<(n-max)<<"
";
		
		
	}
}
