#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<math.h>
using namespace std;
typedef long long ll;
const int mod=1e8+7;

ll QP(ll a,ll n){
	ll tmp=a,ans=1;
	while(n){
		if(n&1)ans=ans*tmp%mod;
		tmp=tmp*tmp%mod;
		n>>=1;
	}
	return ans;
}

int main(){
	int T;
	scanf("%d",&T);
	while(T--){
		ll n;
		scanf("%lld",&n);
		printf("%lld ",(((QP(3,n)-2*QP(2,n)+1)%mod+mod)%mod)*QP(2,mod-2)%mod);
		printf("%lld
",(((QP(4,n)-3*QP(3,n)+3*QP(2,n)-1)%mod+mod)%mod)*QP(2,mod-2)%mod);
	}
	return 0;
}
