#include<cstdio>
#define M 50001
using namespace std;
 
inline int get()
{
int c,res;
res=0;
do{c=getchar_unlocked();}while(c<'0'||c>'9');
while(c>='0'&&c<='9')
{res=(res<<3)+(res<<1)+(c-'0');
c=getchar_unlocked();}
return res;
}
 
int main()
{
int test,sum,temp,temp1,temp2;
int a[M],b[M],c[M],d[M],f[M],f1[M];
int n,i,j,k,ans;
test=get();
while(test--)
{
	n=get();
	for(i=0;i<=n;i++)
	{a[i]=0;b[i]=0;}
	for(i=0;i<n;i++)
	{c[i]=get();
	d[i]=get();
	a[c[i]]++;
	b[d[i]]++;
	}
	sum=0;
	ans=0;
	for(i=n;i>=0;i--)
	{
		sum+=b[i];
		if(sum==i)
			f1[ans++]=i;
		sum-=a[i];
	}
	for(i=0,k=ans-1;i<ans;i++,k--)
		f[k]=f1[i];
	printf("%d
",ans);
	for(i=0;i<n;i++)
	{
		if(f[0]>=c[i]&&f[ans-1]<=d[i])
			printf("1");
		else
		{
			printf("0");
			for(j=0;j<ans;j++)
			if(f[j]>=c[i])
			break;
			temp=0;
			for(k=j;k<ans;k++)
			{if(f[k]>d[i])break;
			temp++;
			}
			temp1=j;
			temp2=k;
			for(;k<ans;k++,j++)
			f[j]=f[k];
			ans-=temp;
		}
	}
printf("
");
}
return 0;
}
