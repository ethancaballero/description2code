int query(int low,int high,int l,int h,int j)
{

{ r= query(low,mid,l,h,2*j+1);

else if((l>mid)&&(h>mid))
{ r= query(mid+1,high,l,h,2*j+2);}

{
 r=(query(low,mid,l,mid,2*j+1)

}
