const int N = 50050;
const int mod = (int)1e9 + 7;

int a[N], b[N];

int f[N];

int main() {
    int T; gi(T);
    rep(tt, T) {
        fill(f, 0);
        int n; gi(n);
        rep(i, n) {
            gi(a[i]), gi(b[i]);
            f[a[i]] += 1;
            f[b[i] + 1] -= 1;
        }
        fore(i, 1, n) f[i] += f[i - 1];

        set <int> valid;
        fore(i, 0, n) {
            if(f[i] == i) valid.insert(i);
        }

        assert(sz(valid));

        int nans = sz(valid);
        string ans;
        rep(i, n) {
            if(*valid.begin() < a[i] or *valid.rbegin() > b[i]) {
                ans.pb('0');
                if(*valid.rbegin() < a[i]) continue;
                int v = *valid.lower_bound(a[i]);
                while(v <= b[i]) {
                    valid.erase(v);
                    set <int>::iterator it = valid.upper_bound(v);
                    if(it == valid.end()) break;
                    v = *it;
                }
            } else {
                while(*valid.begin() < a[i]) {
                    assert(sz(valid));
                    valid.erase(valid.begin());
                }
                while(*valid.rbegin() > b[i]) {
                    assert(sz(valid));
                    valid.erase(--valid.end());
                }
                ans.pb('1');
            }
        }
        pin(nans);
        puts(ans.c_str());
    }
    return 0;
}
