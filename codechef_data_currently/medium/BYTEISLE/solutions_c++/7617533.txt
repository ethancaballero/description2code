#include<cstdio>
#include<iostream>
#include<set>
using namespace std;
#define MAX 50001
#define MOD 1000000007
using namespace std;
int main()
{
	int lo[MAX],hi[MAX],a,b;
	int test,n,i;
	cin>>test;
	while(test--)
	{
		int diff[MAX+1]={0};
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a>>b;
			lo[i]=a;
			hi[i]=b;
			diff[lo[i]]++;
			diff[hi[i]+1]--;
		}
		int count=0;
		int ans=0;
		set<int>good;
		for(i=0;i<=n;i++)
		{
			count+=diff[i];
			if(count == i)
			{
				ans++;
				good.insert(i);
			}
		}
		printf("%d
",ans);
		for(i=0;i<n;i++)
		{
			if(lo[i] <= *good.begin() && *good.rbegin() <= hi[i])
			{
				putchar('1');
			}
			else
			{
				putchar('0');
				good.erase(good.lower_bound(lo[i]),good.upper_bound(hi[i]));
			}
		}
		putchar('
');
	}
}
