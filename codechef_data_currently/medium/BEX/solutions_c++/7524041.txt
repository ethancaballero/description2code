#include<stdio.h>

long int ex[1000000],minimum[1000000];
char books[1000000][16];

int main()
{
    long int i,n,j=0;
    long int num;
    char str[15];

    scanf("%ld%ld%s",&n,&ex[0],&books[0]);

    minimum[0]=0;
    for(i=1;i<n;i++)
    {
        scanf("%ld",&num);
        if(num == -1)
        {
            printf("%ld %s
",j-minimum[j],books[minimum[j]]);//popping out books upto the one having minimum exercises in one go
            j=minimum[j]-1;
        }
        else if(num == 0)
        {
            scanf("%s",&str);//neglecting books with no exercises left to be unsolved
        }
        else
        {
            //pushing new book
            j++;
            ex[j]=num;
            scanf("%s",&books[j]);
            if(num<=ex[minimum[j-1]])//minimum[j] array calculates the index of smallest element in ex[] array with index less than j
                minimum[j]=j;
            else
                minimum[j]=minimum[j-1];
        }
    }
    return 0;
}
