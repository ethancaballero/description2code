        scanf("%d", &x);
        if(x == -1){
            int cnt = 0;
            while(mini[N] != r[N]){
                --N;
                ++cnt;
            }
            printf("%d %s
", cnt, s[N]);
            --N;
        } else{
            ++N;
            r[N] = x;
            scanf(" %s", s[N]);
            mini[N] = min(mini[N - 1], r[N]);
            if(x == 0) --N;
        }/*
