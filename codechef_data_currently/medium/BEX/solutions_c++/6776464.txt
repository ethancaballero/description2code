#include<bits/stdc++.h>
using namespace std;
#define ll long long int
struct book
{
    string name;
    ll x;
    ll ctr;
};
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    stack<book> s;
    string a;
    ll n,m;
    cin>>n;
    while(n--)
    {
        cin>>m;
        if(m==0)
        {
            cin>>a;
            continue;
        }
        else if(m!=-1)
        {
            cin>>a;
            if(s.empty())
            {
                book b;
                b.name=a;
                b.ctr=0;
                b.x=m;
                s.push(b);
            }
            else
            {
                if(m>s.top().x)
                {
                    (s.top().ctr)++;
                }
                else
                {
                    book b;
                    b.name=a;
                    b.ctr=0;
                    b.x=m;
                    s.push(b);
                }
            }
        }
        else
        {
            cout<<s.top().ctr<<" "<<s.top().name<<"
";
            s.pop();
        }
    }
    return 0;
}
