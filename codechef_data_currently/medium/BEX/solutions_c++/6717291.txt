#include <bits/stdc++.h>
#define mp  make_pair
using namespace std;
#define gc            getchar_unlocked

#ifndef ONLINE_JUDGE
    #define gc getchar
#endif

inline void sfast(string &str)
{
    register char c = 0;
    register int i = 0;
    while (c < 33)
        c = gc();

    str = "";
    while(c != '
')
    {
        str += c;
        c = gc();
    }
}

int main()
{
	int n;
	cin >> n;
	stack< pair<int, pair<int,string> > > s;
	s.push(mp(100001, mp(0, " ")));
	for(int i = 0; i < n; i++)
	{
		string sub, temp;
		int ex;
		scanf("%d", &ex);;
		if(ex > 0)	//push BOOK
		{

			if(ex <= s.top().first)
			{
			    sfast(sub);
				s.push(mp(ex, mp(0, sub)));

			}
			else
			{	sfast(temp);
				s.top().second.first++;
			}


		}
		else if(ex == -1)		//pop BOOK
		{
			printf("%d %s
", s.top().second.first, s.top().second.second.c_str());
			s.pop();
		}
        else
            sfast(temp);
	}
	return 0;
}
