#include<bits/stdc++.h>
using namespace std;
typedef struct book{
	long long count,ex;
	char name[16];
};
int main(){
	long long n,x;
	char str[16];
	scanf("%lld",&n);
	stack<book> s;
	while(n--){
		scanf("%lld",&x);
	
			
		if(x!=-1){
			book b;
			scanf("%s",str);
			if(x==0){
				continue;
			}
			if(s.empty()){
				strcpy(b.name,str);
				b.ex=x;
				b.count=0;
				s.push(b);
			}
			else{
				b=s.top();
				
				if(b.ex<x){
					s.pop();
					b.count++;
					s.push(b);	
				}
				else{
					book bx;
					strcpy(bx.name,str);
					bx.count=0;
					bx.ex=x;
					s.push(bx);
				}
			}
		}
		else{
			book b=s.top();
//			cout<<b.count<<" "<<b.name<<endl;
			printf("%lld %s
",b.count,b.name);
			s.pop();
		}
	}
	return 0;
}
