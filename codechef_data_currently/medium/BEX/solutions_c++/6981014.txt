#include<iostream>
#include<cstdio>
#include<stack>
using namespace std;
struct book
{
    long int ex,c;
    char name[17];
    book()
    {
        c=0;
    }
};
int main()
{
    stack<struct book>b;
    long int n;
    scanf("%ld",&n);
    while(n--)
    {
       struct book x;
        scanf("%ld",&x.ex);
        if(x.ex!=-1)
        {
            scanf("%s",x.name);
            if(x.ex==0)
                continue;
            if(b.empty())
                b.push(x);
            else
            {
                if(b.top().ex>=x.ex)
                b.push(x);
                else
               (b.top()).c++;
            }

        }
        else
        {
          printf("%ld %s
",(b.top()).c,(b.top()).name);
          b.pop();
        }
    }
    return 0;
}
