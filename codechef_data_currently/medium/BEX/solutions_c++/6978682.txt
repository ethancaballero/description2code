#include<iostream>
#include<cstdio>
#include<stack>
using namespace std;
struct book
{
    long ex,topbooks;
    char sub[17];
    book()
    {
        topbooks=0;
    }
};
int main()
{
    stack <book> num;
    long n;
    scanf("%ld",&n);
    while(n--)
    {
        book b;
        scanf("%ld",&b.ex);
        if(b.ex!=-1)
        {
            scanf(" %s",b.sub);
            if(b.ex==0)
                continue;
            if(num.size()==0)
                num.push(b);
            else
            {
                if((num.top()).ex>=b.ex)
                    num.push(b);
                else
                    (num.top()).topbooks++;
            }

        }
        else
        {
            b=num.top();
            printf("%ld %s
",b.topbooks,b.sub);
            num.pop();
        }

        }

    return 0;
}
