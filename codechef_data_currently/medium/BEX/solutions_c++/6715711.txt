#include <iostream>
#include<stack>
#include<string.h>
#include<stdio.h>
using namespace std;
struct book
{
    int ex;
    int count;
    char name[16];
};
int main()
{
	stack<book> s;
	book x;
	long int n;
	scanf("%ld",&n);
	int t;
	int min=100001;
	while(n--)
	{
	    scanf("%d",&t);
	    if(t>0)
	    {
	        if(t<min)
	        min=t;
	        char naam[20];
	        scanf("%s",naam);
	        strcpy(x.name,naam);
	        x.ex=t;
	        x.count=min;
	        s.push(x);
	    }
	    else if(t==-1)
	    {
	        int c=0;
	        while(s.top().ex!=min)
	        {
	            s.pop();
	            c++;
	        }
	        printf("
%d %s",c,s.top().name);
	        s.pop();
	        if(!s.empty())
	        min=s.top().count;
	        else 
	        min=100001;
	    }
	    //in case t=0
	    else 
	    {
	        char naam[20];
	        scanf("%s",naam);
	    }
	}
	return 0;
}
