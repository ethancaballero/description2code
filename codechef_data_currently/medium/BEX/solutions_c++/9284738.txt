#include<bits/stdc++.h>
using namespace std;
struct book
{
	char name[16];
	int booksabove;
	int remex;
	book()
	{
	}
	book(char title[],int ex)
	   :booksabove(0)
	{
		strcpy(name,title);
		remex=ex;
	}
};
int main()
{

	stack<book> books;
	int n;
	scanf("%d",&n);
	char name[16];
	int ex;
	
	while(n--)
	{
		
		scanf("%d",&ex);
		if(ex!=-1)
		{
			scanf("%s",name);
			if(ex==0)
		    continue;
		    if(books.size()==0)
		    books.push(book(name,ex));
		    else
		    {
		    	book& b=books.top();
		    	if(b.remex>=ex)
		    	{
		    		books.push(book(name,ex));
		    	}
		    	else
		    	{
		    		b.booksabove++;
		    	}
		    }
		}
		else
		{
			  book& b=books.top();
			  books.pop();
			  printf("%d %s
",b.booksabove,b.name);
		}
		
		
		
	}
	return 0;
}
