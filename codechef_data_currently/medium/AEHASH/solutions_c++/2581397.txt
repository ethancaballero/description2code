#include<stdio.h>
#include<iostream>
#include<ctime>
 
using namespace std;
 
#define MOD 1000000007
#define LL (long long)
 
int N,a,e,v,k,n,i,j;
long long f[51][51][1001];
 
int main()
{
    f[1][0][1]=1LL;      
    for(v=0;v<153;v++)
      f[0][0][v]=f[0][1][v]=1;
    for(v=1;v<153;v++)
      f[1][0][v]=1;   
    for(a=0;a<51;a++)
    for(e=0;e<51;e++) 
    {
      if(a+e>1)
      for(v=a;v<153;v++)
      {
        n=a+e;
        int mink=max(0,a-(n+1)/2);
        int maxk=min(a,n/2);
        for(k=mink;k<=maxk;k++)
          f[a][e][v]=(f[a][e][v]+LL(f[k][n/2-k][v-a])*f[a-k][(n+1)/2-(a-k)][v-a])%MOD;
      }
    }    
    int d,b,c,t;
    d=b=c=0;    
    scanf("%d",&t);
    while(t--)    
    {
        scanf("%d %d %d",&d,&b,&c);
        if(c>152) printf("0
");
        else 
        {
           int ans=f[d][b][c]- (c>0 ? f[d][b][c-1] : 0);
           if(ans<0) ans+=MOD;
           printf("%d
",ans);
        }
    }  
}
