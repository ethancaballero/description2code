#define DEBUG 1
#define MOD 1000000007
 
#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
 
long long unsigned dp[51][51][153];
int sum[51][51][153];
 
int main(void) {
  int tc, a, e, v, i;
 
  memset(dp, 0, sizeof(dp));
  memset(sum, 0, sizeof(sum));
  dp[1][0][1]=dp[0][0][0]=sum[0][0][0]=1;
  for(v=1;v<=152;v++)
    sum[1][0][v]=sum[0][0][v]=1;
 
  for(e=1;e<=50;e++) {
    sum[0][e][0]=dp[0][e][0]=1;
    for(v=1;v<=152;v++)
      sum[0][e][v]=sum[0][e][v-1]+dp[0][e][v];
  }
 
  for(a=1;a<=50;a++)
    for(e=0;e<=50;e++) {
      if(a==1 && e==0)
        continue;
      for(v=1;v<=152;v++) {
       if(v-a>0)
        for(i=0;i<=min(a,(a+e)/2);i++)
          if((a+e)/2-i<=e)
            dp[a][e][v]=(dp[a][e][v]+dp[i][(a+e)/2-i][v-a]*dp[a-i][e-((a+e)/2-i)][v-a]+dp[i][(a+e)/2-i][v-a]*sum[a-i][e-((a+e)/2-i)][v-a-1]+sum[i][(a+e)/2-i][v-a-1]*dp[a-i][e-((a+e)/2-i)][v-a])%MOD;
        sum[a][e][v]=(sum[a][e][v-1]+dp[a][e][v])%MOD;
      }
    }
 
  scanf("%d", &tc);
  while(tc--) {
    scanf("%d%d%d", &a, &e, &v);
    if(v>152)
      printf("0
");
    else
      printf("%llu
", dp[a][e][v]);
  }
  return 0;
}
