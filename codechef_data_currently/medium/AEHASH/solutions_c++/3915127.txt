#include<stdio.h>
#include<string.h>
#define LL long long
#define M 1000000007
LL dp[60][60][160];
 
LL S(int a,int e,int v)
{
int m=(a+e)/2,i;
LL p=0;
if(v<0||~dp[a][e][v]||a+e<=1)return(v<0)?0:((~dp[a][e][v])?dp[a][e][v]:(dp[a][e][v]=(a<=v)));
for(i=-1;++i<=m;p=(i<=a&&m-i<=e)?(p+S(i,m-i,v-a)*S(a-i,e-(m-i),v-a))%M:p);
return dp[a][e][v]=p;
}
 
main()
{
int f,a,e,v;
LL r;
for(scanf("%d",&f)&&memset(dp,-1,sizeof dp);f--;scanf("%d%d%d",&a,&e,&v),printf("%lld
",(r=v>152?0:(r=(S(a,e,v)-S(a,e,v-1))%M)<0?r+M:r)));
return 0;
}
