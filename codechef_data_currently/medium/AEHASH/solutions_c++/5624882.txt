#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
const int mod = 1e9+7;
ll dp[105][55][1005];
ll solve(int A,int E,int V){
  if(A<0 || E<0 || V<0) return 0;
  if(dp[A][E][V] != -1) return dp[A][E][V];
  int N = A+E;
  if(N<=1) return (dp[A][E][V] = (A<=V));
  if(V<A) return (dp[A][E][V] = 0);
  ll res = 0;
  int x = N/2;
  for(int i=0;i<=A;i++){
     ll a = solve(i,x-i,V-A);
     ll b = solve(A-i,E-(x-i),V-A);
     res = (res+((a*b)%mod))%mod;
  }
  dp[A][E][V] = res;
  return res;
}
int main(){
int t,A,E,V;
memset(dp,-1,sizeof dp);
scanf("%d",&t);
while(t--){
  scanf("%d %d %d",&A,&E,&V);
  ll ans = solve(A,E,V);
  if(V)
    ans = (ans-solve(A,E,V-1)+mod)%mod;
  printf("%lld
",ans);
}
return 0;
}
