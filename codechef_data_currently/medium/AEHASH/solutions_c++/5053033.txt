#include <cstdio>

#include <iostream>

#include <cstring>

 
using namespace std;

 
int DP[60][60][200];

 
const long long MOD=1000000007;

 
long long funct(int a, int e, int v)

{

    if (v<0) return 0;

    if (DP[a][e][v]!=-1) return DP[a][e][v];

    if (a+e <= 1) return DP[a][e][v]=(a<=v);

    long long sol=0LL;

    int t=(a+e)/2;

    for (int i=0;  i < t+1; i++)

    {

        if ((i<=a) && (t-i<=e))

        {
          int g = v-a;
          int w = t-i;

            sol+=funct(i,w,g)*funct(a-i,e-w,g);

            sol%=MOD;

        }

    }

    return DP[a][e][v]=sol;

}

 
int main()

{

    int t;

    memset(DP,-1,sizeof(DP));

    scanf("%d",&t);

    while (t--)

    {

        int A,E,V;

        scanf("%d%d%d",&A,&E,&V);

        long long temp=0;

        temp=(funct(A,E,V)-funct(A,E,V-1)+MOD)%MOD;

        printf("%lld
",temp);

    }

 
 
    return 0;

}
