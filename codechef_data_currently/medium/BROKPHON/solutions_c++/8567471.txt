#include<bits/stdc++.h>
using namespace std;
int main(){
	int t,n;
	long arr[100002];
	bool ans[100002];
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(int i=0;i<n;i++){
			scanf("%ld",&arr[i]);
		}
		for(int i=0;i<n;i++){
			ans[i]=0;
		}
		for(int i=1;i<n;i++){
			if(arr[i]!=arr[i-1]){
				ans[i]=1;
				ans[i-1]=1;
			}
		}
		long cnt=0;
		for(int i=0;i<n;i++){
			if(ans[i]==1){
				cnt++;
			}
		}
		printf("%ld
",cnt);
	}
	return 0;
}
