#include <stdio.h>
#define gc getchar_unlocked
 
inline int getn(){
  int n = 0, c = gc();
  while(c < '0' || c > '9') c = gc();
  while(c >= '0' && c <= '9')
    n = (n<<3) + (n<<1) + c - '0', c = gc();
  return n;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i,count=0;
		n=getn();
		int p=0,m=0,x=0;
		p=getn();
		x=getn();
		m=p;
		for(i=0;i<n-2;i++)
		{
			if(p!=m||m!=x) count++;
			p=m;
			m=x;
			x=getn();
		}
		if(p!=m||m!=x) count++;
		if(x!=m) count++;
		printf("%d
",count);
	}
	return 0;
}
