#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,i,sum=0,d=1;
		scanf("%lld",&n);
		long long int a[n];
		for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
		for(i=0;i<n-1;i++)
		{
			if(a[i]!=a[i+1]&&d==1)
			{
				sum+=2;
				d=0;
			}
			else if(d==0&&a[i]!=a[i+1])
			{
				sum+=1;
			}
			else if(a[i]==a[i+1])
			d=1;
		}
		printf("%lld
",sum);
	}
	return 0;
}
