#include<bits/stdc++.h>
using namespace std;

int a[100010];
bool b[100010];

int main()
{
	int t,i,n,cnt;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		cnt=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			if(i!=0)
			{
				if(a[i]!=a[i-1])
				{
					if(!b[i] && !b[i-1])
					cnt+=2;
					else
					cnt++;
					b[i]=b[i-1]=1;
					//printf("%d ",i);
				}
			}
		}
		for(i=0;i<n;i++)
		b[i]=0;
		printf("%d
",cnt);
		
	}
	return 0;
}
