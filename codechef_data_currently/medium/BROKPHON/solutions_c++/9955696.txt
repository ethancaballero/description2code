#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    int t,n,i,c;

     scanf("%d",&t);
     while(t--)
     {
         c=0;
           scanf("%d",&n);
           int f[n],a[n];
           for(i=0;i<n;i++)
            {
                 scanf("%lld",&a[i]);
                 f[i]=0;
            }

            for(i=1;i<n;i++)
            {
                if(a[i-1]!=a[i])
                {
                    if(f[i-1]==1)
                        c++;
                    else
                    c=c+2;
                    f[i]=1;
                }
            }

            printf("%d
",c);


     }
}
