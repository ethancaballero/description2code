#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
       int T;
       scanf("%d",&T);
       while(T--)
       {
              long int i,n,count=0;
              scanf("%ld",&n);
              long int a[n];
              for(i=0;i<n;i++)
                     scanf("%ld",&a[i]);
              for(i=0;i<n-1;i++)
              {
                     if(i==0)
                     {
                            if(a[i]!=a[i+1])
                                   count+=2;
                     }
                     else
                     {
                            if(a[i]!=a[i+1]&&a[i]==a[i-1])
                                   count+=2;
                            if(a[i]!=a[i+1]&&a[i]!=a[i-1])
                                   count+=1;
                     }
              }
              printf("%ld
",count);
       }
       return 0;
}
