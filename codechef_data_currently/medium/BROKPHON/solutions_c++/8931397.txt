#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,j;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        int arr[n],ans=0;
        int wrong[n];
        for(i=0;i<n;i++)
            wrong[i]=0;
        scanf("%d",&arr[0]);
        for(i=1;i<n;i++)
        {
            scanf("%d",&arr[i]);
            if(arr[i]!=arr[i-1])
               {
                   if(wrong[i-1])
                    ans+=1;
                   else
                    ans+=2;
                   wrong[i-1]=1;
                   wrong[i]=1;
               }
        }
        printf("%d
",ans);
    }
}
