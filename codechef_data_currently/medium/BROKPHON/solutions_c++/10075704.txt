#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	long long int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long long int f=0;
		long long int a[10000];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		for(int i=0;i<n;i++)
		{
			if(i>0)
			{
				if(a[i-1]!=a[i])
				{
					f++;
					continue;
				}
			}
		    if (i<n-1)
			{
				if(a[i+1]!=a[i])
				{
					f++;
				}
			}
		}
		cout<<f<<"
";
	}
}
