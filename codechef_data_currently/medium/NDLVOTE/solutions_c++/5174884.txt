#include<stdio.h>
#include<iostream>
#include<stdlib.h>
#include<string.h>
 
#define INF 1000000000
#define get getchar_unlocked()
 
using namespace std;
 
inline int read ()
{
        char c;
        int n = 0;
 
        while ((c = get) < 48);
        n += (c - '0');
        while ((c = get) >= 48)
        n = n * 10 + (c - '0');
        return n;
}
 
int main(){
	int n;
 
	while(true){
		scanf("%d", &n);
		if(n == 0 )  break;
		
		int tc = n;
		int user = 0 , votes, temp, diff;
		char type;
		while(tc--)
		{	
			getchar();
			scanf("%c",&type);
			scanf("%d", &votes);
			if(type == 'P')
				diff = 1;
			else
				diff = -1;
			
			if((votes - diff) > 0 ){
				temp = votes - diff;
			}else{
				temp = diff - votes;
			}			
			
			if(temp >= user)
				user = temp;
			else if(( user - temp ) % 2 == 1)
				user++;
		}
		printf("%d", user);
		printf("
");
		user = 0;
	}
	return 0;
}
