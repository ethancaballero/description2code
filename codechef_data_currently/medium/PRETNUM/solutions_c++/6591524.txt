bool isprime[100005];
int a[limit];
bool isPrime(int n)
{
	if(n < 2) return false;
	if(n == 2) return true;
	if(n % 2 == 0) return false;
	for(int i = 3; i * i <= n; i++) {
		if(n % i == 0) return false;	
	} 	
	return true;
}
 
void pre()
{
	rep(i, 100000) {
		isprime[i] = isPrime(i);
	}
}
 
void solve()
{
	LL L,R;
	sl(L); sl(R);
	for(LL i = 0; i <= R - L; i++) a[i] = 0;
	for(LL i = 1; i * i <= R; i++) {
		LL pro = i * i;
		for(LL j = (L / i) * i; j <= R; j += i) if( j >= L && pro <= j ) a[j - L] += 1 + (pro != j);
	}
	int ans = 0;
	for(LL i = L; i <= R; i++) if(isprime[a[i - L]]) ans++;
	pi(ans);
	printf("

}
