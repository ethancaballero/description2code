vector<int> Prime;
int mark[1000002];

void sieve(int n)
{
	int i,j,limit=sqrt(n*1.)+2;

	mark[1] = 1;
	for(i=4;i<=n;i+=2) mark[i]=1;

	Prime.push_back(2);
	for(i=3;i<=n;i+=2)
		if(!mark[i])
		{
			Prime.push_back(i);

			if(i<=limit)
			{
				for(j=i*i;j<=n;j+=i*2)
				{
					mark[j] = 1;
				}
			}
		}
}

LL num[1000006];
int divs[1000006];

int main()
{
	sieve(1000000);

	int T;
	LL L, R, i, n, p;
	int j, cnt, sz;

	LL LIM = 1000000;
	LIM *= LIM;

	scanf("%d", &T);

	while(T--)
	{
		scanf("%lld %lld", &L, &R);
		AIN(L, 1, R);
		AIN(R, L, LIM);
		AIN(R-L, 0, 1000000);

		if(L == 1) L++;
		
		for(i = L; i <= R; i++)
		{
			num[i - L] = i;
			divs[i - L] = 1;
		}

		//segmented sieve style
		sz = SZ(Prime);
		for(j = 0; j < sz; j++)
		{
			p = Prime[j];
			if(p > R) break;

			for(i = ((L - 1)/p + 1) * p; i <= R; i += p)
			{
				n = num[i - L];
				cnt = 0;
				while(n % p == 0)
				{
					n /= p;
					cnt++;
				}

				divs[i - L] *= cnt + 1;
				num[i - L] = n;
			}
		}

		cnt = 0;
		for(i = L; i <= R; i++)
		{
			if(num[i - L] != 1)
				divs[i - L] *= 2;
			cnt += mark[ divs[i - L] ] == 0;
		}

		AIN(cnt, 0, 1000000);

		printf("%d
", cnt);
	}

	return 0;
}
