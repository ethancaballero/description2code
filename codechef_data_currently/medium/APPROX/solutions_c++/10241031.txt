#include<bits/stdc++.h>
using namespace std;
 
int main() {
	int t;
	scanf("%d", &t);
	while(t--) {
		int q;
		scanf("%d", &q);
		int num = 103993, den = 33102;
		for (int i = 0; i <= q; ++i) {
			printf("%d", num / den);
			if (i == 0 && q > 0) {
				printf(".");
			}
			num = num % den * 10;
		}
		printf("
");
	}
	return 0;
}
