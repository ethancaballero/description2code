double ans[100001][35];
 
int main()
{
	ans[0][1]=-0.5;
    for(int i=1;i<100001;i++)
    {
        ans[i][1]=ans[i-1][1]+0.5;
        ans[i][0]=i;
    }
    for(int i=2;i<35;i++)
    {
        double sum=0;
        for(int j=i+1;j<100001;j++)
            ans[j][i]=(sum+=ans[j-1][i-1])/j;
    }
    sp='.';
    int t=scan_d();
    while(t--)
    {
       int n=scan_d(),k=scan_d();
        if(k<n&&k<35)
        {
        	printsp_d((int)ans[n][k]);
        	double temp=ans[n][k]-(int)ans[n][k];
        	int te=6;
        	while(te--)
        	{
        		temp*=10;
        		putchar_unlocked((int)temp+'0');
        		temp-=(int)temp;
        	}
        	putchar_unlocked('
');
        }
        else
            putchar_unlocked('0'),putchar_unlocked('
');
    }
	return 0;
}
