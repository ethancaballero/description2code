int mod = 1000000007LL;

/**************************Code****************************/

int power( int a, int b )
{
	if( !b )
		return 1;
	int r = power( a, b/2 );
	r = r * 1LL * r % mod;
	if( b&1 )
		r = r * 1LL * a % mod;
	return r;
}

{

	}

}
