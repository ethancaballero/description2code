#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int maxn = 1e6 + 10;
const ll mod = 1e9 + 7;
int A[maxn];
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		for(int i = 1;i <= n;++i){
			cin>>A[i];
		}
		ll stems = 1;
		bool can = true;
		for(int i = 1;i <= n;++i){
			if(stems < A[i]){
				can = false;
				break;
			} else{
				stems -= A[i];
				stems <<= 1;
			}
		}
		if(can && stems == 0){
			cout<<"Yes
";
		} else{
			cout<<"No
";
		}
	}
	return 0;
}
