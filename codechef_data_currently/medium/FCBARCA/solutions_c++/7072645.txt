#include <bits/stdc++.h>

#define mod 1000000007

using namespace std;

long long dp[1001][11];

int k,n,N,K,T;

int main(int argc, char **argv)
{
	for(k=0;k<=10;k++)
	{
		dp[0][k]=0;
		dp[1][k]=k;
	}
	
	for(n=2;n<=1000;n++)
	{
		for(k=1;k<=10;k++)
		{
			dp[n][k]=(((dp[n-1][k]*(k-1))%mod)+((dp[n-2][k]*k)%mod))%mod;
		}
	}
	
	for(scanf("%d",&T);T>0;T--)
	{
		scanf("%d%d",&N,&K);
		printf("%lld
",dp[N-1][K]);
	}
	
	return 0;
}
