from sys import stdin,stdout
t = int(stdin.readline())
def noo(n):
	k = 0
	for i in bin(n)[2:]:
		if i == '1': k += 1
	return k
for _ in xrange(t):
	n,a,b = map(int,stdin.readline().split())
	n1 = noo(a)
	n2 = noo(b)
	if n1+n2 <= n:
		s = '1'*(n1+n2)+'0'*(n-n1-n2)
	else:
		s = '1'*(2*n-n1-n2)+'0'*(n1+n2-n)
	stdout.write(str(int(s,2)))
	stdout.write('
')
