#include<bits/stdc++.h>
#include<cstdio>
using namespace std;

int main(){
int t,n,a[100005],b[100005];
cin>>t;
for(int i=0;i<t;i++)
{
    cin>>n;
    long long x=0,y=0;
    for(int i=0;i<n;i++)
       cin>>a[i];

    for(int i=0;i<n;i++)
    {
        cin>>b[i];
        if(b[i]==1)
          x++;
        else if(b[i]==2)
          y++;
    }

    sort(a,a+n);
    sort(b,b+n);

    double kiss=0;
    for(int count=0,i=0;i<n;i++)
    {
    if(a[i]!=1)
    {
        count=1;
        while(i+1<n&&(a[i]==a[i+1]))
        {
				i++;
				count++;
        }
         kiss=kiss+count*x;

         if(a[i]==2&&(b[n-1]>4))
         {
            int z=upper_bound(b,b+n,4)-b;
            z=n-z;

            kiss=kiss+(long long)count*z;
			}
          if(a[i]!=2&&a[i]<b[n-1]){
				int z=upper_bound(b,b+n,a[i])-b;
				z=n-z;
				kiss=kiss+(long long)count*z;
			}
            if(a[i]==3)
             kiss=kiss+count*y;

    }
    }
        printf("%lf
",kiss/n);

}
return 0;
}
