#include <iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;

int main()
{
   int T;
  
   unsigned long long int N;
   cin>>T;
   while(T--)
   {
       ////a->x && b->y a^b > b^a 
       cin>>N;
       unsigned long int temp,i,j;
       double a[N],b[N];
       for(i=0;i<N;i++)
           {
              
               cin>>temp;
               a[i]=log(temp)/temp;    
           }
       for(i=0;i<N;i++)
           {
              
              cin>>temp;
               b[i]=log(temp)/temp;        
           }
       sort(a,a+N);
       sort(b,b+N);
       i=0;j=0;
        double ans=0.0,count=0.0;
       while(i<N)
        {
            while(a[i]>b[j] && j<N)
            {
                count=count+1.0;
                j++;
                //cout<<"YNWA"<<endl;
            }
            ans=ans+count/N*1.0;
            i++;
        }
        printf("%.6f
",ans);
       
   }
   
   return 0;
}
