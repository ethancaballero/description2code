#include <cstdio>
#include <algorithm>

using namespace std; 

int A[100000];
int B[100000];

int main()
{	
	int T;
	scanf("%d",&T);
	
	while(T--)
	{
		long long int N;
		scanf("%lld", &N);

		long long int i, lenA = 0, lenB = 0, count = 0;

		for(i = 0; i < N; i++)
		{
			int x;
			scanf("%d", &x);
			if(x == 1) continue;
			if(x == 2) x = 4;
			A[lenA++] = x;
		}

		for(i = 0; i < N; i++)
		{
			int x;
			scanf("%d", &x);
			if(x == 1) continue;
			if(x == 2) x = 4;
			B[lenB++] = x;
		}

		if(lenA == 0){printf("0.000000
"); continue;}
		count += lenA*(N-lenB);
		double expt = ((double)count)/N;
		if(lenB == 0){printf("%lf
", expt); continue;}

		sort(A, A+lenA);
		sort(B, B+lenB);

		int flag = 0;
		long long int j = 0;
		for(i = 0; i < lenA; i++)
		{
			for(; j < lenB; j++)
			{
				if(B[j] > A[i])
				{
					count += lenB - j;
					break;
				}
				else if(A[i] >= B[lenB - 1])
				{
					flag = 1;
					break;
				}
				else
				{
					if(A[i] >= B[(lenB+j)/2])
						j = (lenB+j)/2;
				}				
			}
			if(flag) break;
		}

		expt = ((double)count)/N;
		printf("%lf
", expt);
	}

	return 0;
}
