#include <bits/stdc++.h>

using namespace std;

vector<int> a, b;

int main() {
	int t;
	scanf("%d", &t);
	while(t--) {
		a.clear();
		b.clear();
		int n, one = 0, two = 0;
		scanf("%d", &n);
		for(int i = 0; i<n ;i++) {
			int x;
			scanf("%d", &x);
			if(x != 1) {
				a.push_back(x);
			}
		}
		for(int i = 0; i<n ;i++) {
			int x;
			scanf("%d", &x);
			b.push_back(x);
			if(x == 1) one++;
			if(x == 2) two++;
		}
		sort(a.begin(), a.end());
		sort(b.begin(), b.end());
		double r = 0;
		for(int i = 0; i<(int)a.size(); i++) {
			vector<int> :: iterator it;
			int aux;
			if(a[i] == 1) {
				//printf("oxi");
				continue;
			}
			else if(a[i] == 2) {
				it = upper_bound(b.begin(), b.end(), 4);
				//printf(">>%d
", aux);
				aux = b.end() - it;
				aux += one;
			}
			else if(a[i] == 3) {
				it = upper_bound(b.begin(), b.end(), 3);
				//printf(">>%d
", aux);
				aux = b.end() - it;				
				aux += two + one;
			}
			else {
				it = upper_bound(b.begin(), b.end(), a[i]);
				//printf(">>%d
", aux);
				aux = b.end() - it;	
				aux += one;
			}
			r += double(aux)/(double)(b.size());
		}
		printf("%.7lf
", r);
	}
	return 0;
}
