#include <cstdio>
#include <algorithm>
 
using namespace std; 
 
int A[100000];
int B[100000];

inline void scanint(int* x)
{
	register char c = getchar_unlocked();
	*x = 0;
	for(; (c<48)||(c>57);c = getchar_unlocked());
	for(; (c>47)&&(c<58);c = getchar_unlocked())
		*x = (int)((((*x)<<1) + ((*x)<<3)) + c - 48);
}

int main()
{	
	int T;
	scanint(&T);
	
	while(T--)
	{
		int N;
		scanint(&N);
 
		long long int i, lenA = 0, lenB = 0, count = 0;
 
		for(i = 0; i < N; i++)
		{
			int x;
			scanint(&x);
			if(x == 1) continue;
			if(x == 2) x = 4;
			A[lenA++] = x;
		}
 
		for(i = 0; i < N; i++)
		{
			int x;
			scanint(&x);
			if(x == 1) continue;
			if(x == 2) x = 4;
			B[lenB++] = x;
		}
 
		if(lenA == 0){printf("0.000000
"); continue;}
		count += lenA*(N-lenB);
		double expt = ((double)count)/N;
		if(lenB == 0){printf("%lf
", expt); continue;}
 
		sort(A, A+lenA);
		sort(B, B+lenB);
 
		int flag = 0;
		long long int j = 0;
		for(i = 0; i < lenA; i++)
		{
			for(; j < lenB; j++)
			{
				if(B[j] > A[i])
				{
					count += lenB - j;
					break;
				}
				else if(A[i] >= B[lenB - 1])
				{
					flag = 1;
					break;
				}				
			}
			if(flag) break;
		}
 
		expt = ((double)count)/N;
		printf("%lf
", expt);
	}
 
	return 0;
}
