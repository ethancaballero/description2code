int quicksort(long a[], int i, int j )
{	
	if(i>j)
	{	return 0;
	}
	long pivot=a[i];
	int frst=i;int last=j;int pos=-1;
	i++;
	while(j>=i)
	{	if(a[j]>pivot)
		{	j--;
		}
		else 
		{	int swap = a[i];
			a[i]=a[j];
			a[j]=swap;
			i++;
		}

	}
	i--;
	a[frst]=a[i];
	a[i]=pivot;
	i--;
	j++;
	quicksort(a,frst,i);
	quicksort(a,j,last);
}
