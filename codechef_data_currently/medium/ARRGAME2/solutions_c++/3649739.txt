double a[100010],b[100010];
 
int bin_search(int n,int r)
{
        int low=0,high=n, mid,flag=0;
        b[0]=0;b[n+1]=1000000100;
        while(low <= high)
        {
                mid = (low + high)/2;
                if( b[mid] <= a[r] && b[mid+1] > a[r] )
                {
                    return mid; 
                }
                else if( b[mid] <= a[r] && b[mid+1] <= a[r] )
                {
					low=mid+1;	
				}
                else if(b[mid] > a[r])
                {
 
                    high = mid;
                }
        }
}
