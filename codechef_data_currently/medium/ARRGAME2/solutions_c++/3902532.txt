    inline static int get_i(){
    static int noRead,p;
    noRead=0;
    do{
    p=getchar_unlocked();
    }while(p<48);
    while(p>47){
    noRead= (noRead<<3)+(noRead<<1)+p-'0' ;
    p=getchar_unlocked();
    }
    return noRead;
    }
     
     
    void max_heapify(int *x,int n){
    int *a=x-1;
    int i,j,k,t;
    for(i=2;i<=n;i++){
    j=i;
    k=(j>>1);
    while(j!=1&&a[j]>a[k]){
    t=a[j];
    a[j]=a[k];
    a[k]=t;
    j=k;
    k=j>>1;
    }
    }
     
    }

    void sort_heap(int *x,int n){
    int i,t;
    for(i=n;i>=2;i--){
    t=pop_heap(x,i);
    x[i-1]=t;
    }
    }
