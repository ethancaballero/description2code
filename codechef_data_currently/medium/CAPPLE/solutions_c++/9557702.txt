#include<bits/stdc++.h>
    using  namespace std;
    int a[100001];
    int main()
    {
       int t,i,n,count;
       cin>>t;
       while(t--)
       {
           count=1;
           cin>>n;
           for(i=0;i<n;i++)
            cin>>a[i];
           sort(a,a+n);
           for(i=0;i<n-1;i++)
           {
               if(a[i]!=a[i+1])
                count++;
           }
           cout<<count<<"
";

       }

        return 0;
    }
