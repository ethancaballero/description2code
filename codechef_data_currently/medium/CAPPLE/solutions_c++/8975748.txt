#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,i,b,sum=0;
		scanf("%lld",&n);
		long long int a[100002]={0};
		for(i=0;i<n;i++)
		{
			scanf("%lld",&b);
			if(a[b]==0)
			{
				a[b]++;
				sum++;
			}
		}
		printf("%lld
",sum);
	}
	return 0;
}
