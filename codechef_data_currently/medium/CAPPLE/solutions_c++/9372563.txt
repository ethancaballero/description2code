#include<bits/stdc++.h>

using namespace std;
int main()
{
	int t,n,a[100005];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(int i=0;i<n;i++)scanf("%d",&a[i]);
		int count=0;
		int prev=0,curr=0;
		sort(a,a+n);
		//for(int i=0;i<n;i++)printf("%d ",a[i]);
		for(int i=0;i<n;i++)
		{
			curr=a[i];
			if(prev!=curr)count++;
			
			prev=curr;
		}
		

		
		printf("%d
",count);		
	}
	return 0;
}
