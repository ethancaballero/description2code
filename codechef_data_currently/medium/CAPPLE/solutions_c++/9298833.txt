#include<bits/stdc++.h>
using namespace std;
int a[100001];
int main()
{
	int t;
	cin>>t;
	long long int n,i,count;
	while(t--)
	{
		count=1;
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		for(i=1;i<n;i++)
		{
			if(a[i]!=a[i-1])
			count++;
			
		}
		cout<<count<<"
";
	}
	return 0;
}
