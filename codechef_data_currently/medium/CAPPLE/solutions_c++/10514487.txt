#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        int num,b[100001]={0},c=0;
        for(i=0;i<n;i++)
        {
            scanf("%d",&num);
            b[num]++;
        }
        for(i=1;i<=100000;i++)
            if(b[i]>0)c++;
        printf("%d
",c);
    }
    return 0;
}
