ll power(ll a,ll b,ll mod)
{
	ll ans=1;
    while(b)
    {
    	if(b&1)
    	{
    		ans=mul(ans,a,mod);
		}
		a=mul(a,a,mod);
    	b>>=1;
    	
	}
	return ans;
}
