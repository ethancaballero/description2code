int main(){   
    int T,R,C,i,j,k,l,count,L,Rt,Tp,B,minm,mon;
    char map[505][505];
    scanf("%d",&T);
    while(T--){
    	count=L=Rt=Tp=B=0;
    	scanf("%d%d",&R,&C);
    	for(i=0;i<R;i++)
    	   for(j=0;j<C;j++)
    	     cin>>map[i][j];      
    	for(i=2;i<R-2;i++){
    		for(j=2;j<C-2;j++){
    			L=Rt=Tp=B=0;
    			if(map[i][j]!='#'){
    				minm=min(i,j);
    				minm=min(minm,R-1-i);
    				minm=min(minm,C-1-j);
    				k=1;
    				while(k<=minm){
    					if(map[i][j-k]!='^'||map[i][j+k]!='^'||map[i-k][j]!='^'||map[i+k][j]!='^')
    					   break;
    					else
						  k++;   
    				}
    				k--;
    				//printf("min=%d
",minm);
    				for(l=0;prime[l]<=k;l++);
                    mon=l;
                    //printf("mon=%d
",mon);
    	            count+=mon;
    			}
    		}
    	}
		printf("%d
",count);    
    }

}
