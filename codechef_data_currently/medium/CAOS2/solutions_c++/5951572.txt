{
	primecount.clear();

	primecount[0]=0;

		primecount[i] = c;

		if(!not_prime[i])
		{
			++c;
			primecount[i] = c;

			for(int j=2*i;j<MAX;j+=i)
			{

				not_prime[j] = true;
			}

		}

	}



}
