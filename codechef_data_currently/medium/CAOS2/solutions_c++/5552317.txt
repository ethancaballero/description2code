#include<iostream>
#include<cstdio>
#define max 999999
using namespace std;
char ar[505][505];
bool ar1[505];
int prime[505];
int n,m;
int ans;
void sieve()
{
    int idx=0;
    for(int i=2;i<=500;i++)
    {
        if(ar1[i]==0)
        {
            prime[idx]=i;
            ++idx;
            for(int j=(i*i);j<=505;j+=i)
            ar1[j]=1;
        }
    }
}
int min(int a,int b,int c,int d)
{
    if(a<=b&&a<=c&&a<=d)
    return a;
    if(b<=a&&b<=c&&b<=d)
    return b;
    if(c<=a&&c<=b&&c<=d)
    return c;
    if(d<=a&&d<=b&&d<=c)
    return d;
}
void fun(int i,int j)
{
        int len = min(i,j,(n-1-i),(m-1-j));
        int val=1;
        while(val<=len)
        {
            if(ar[i-val][j]!='^'||ar[i+val][j]!='^'||ar[i][j-val]!='^'||ar[i][j+val]!='^')
            break;
            ++val;
        }
        for(int k=0;k<=100;k++)
        {
            //cout<<"
 val="<<val<<" prime[k]="<<prime[k]<<" k="<<k;
            if(prime[k]<val)
            ++ans;
            else
            break;
        }
        //cout<<" ans="<<ans;
}
int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    sieve();
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ans=0;
        scanf("%d%d",&n,&m);
        for(int i=0;i<n;i++)
        cin>>ar[i];
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                if(ar[i][j]!='#')
                fun(i,j);
            }
        }
        printf("%d
",ans);
    }
    return 0;
}
