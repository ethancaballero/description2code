int min(int a,int b,int c,int d)
{
	return (a<=b?(a<=c?(a<=d?a:d):(c<=d?c:d)):(b<=c?(b<=d?b:d):(c<=d?c:d)));
}
int compute(char *m[],int i,int j,int x,int y)
{

	a=i;
	b=j;

	d=y-j-1;
	small=min(a,b,c,d);
	int k=1;
	if(m[i][j]!='#')
	{

	
	while(k<=small)
	{
		if(m[i-k][j]!='^'||m[i][j-k]!='^'||m[i+k][j]!='^'||m[i][j+k]!='^')
		{
			break;
		}
            
		else
			k++;

	}
	return (k-1);
	}
	else
		return 1;
}
