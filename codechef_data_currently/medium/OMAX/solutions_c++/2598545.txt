    #include<stdio.h>
    int dp[77][77][77][77];
    int sum[77][77];
    int min(int a,int b)
    {
    if(a<b)return a;
    return b;
    }
    int max(int a,int b)
    {
    if(a>b)return a;
    return b;
    }
    int main()
    {
    int n,m,i,j,k,l,ans,tmp,x1,y1,x2,y2;
    while(scanf("%d %d",&n,&m)!=EOF)
    {
    if(n==0 && m==0)break;
    for(i=0;i<n;i++)
    {
    for(j=0;j<m;j++)
    {
    scanf("%d",&tmp);
    if(i>0 && j>0)
    sum[i][j]=sum[i-1][j]+sum[i][j-1]-sum[i-1][j-1]+tmp;
    else if(i>0)
    sum[i][j]=sum[i-1][j]+tmp;
    else if(j>0)
    sum[i][j]=sum[i][j-1]+tmp;
    else
    sum[i][j]=tmp;
    //printf("%d ",sum[i][j]);
    }
    //printf("
");
    }
    ans=-1e9;
    for(i=n-3;i>=0;i--)
    for(j=m-3;j>=0;j--)
    for(k=i+2;k<n;k++)
    for(l=j+2;l<m;l++)
    {
    x1=i+1;y1=j+1;x2=k-1;y2=l-1;
    dp[x1][y1][x2][y2]=sum[x2][y2]-sum[x2][y1-1]-sum[x1-1][y2]+sum[x1-1][y1-1];
    if(x2-x1>0)
    {
    dp[x1][y1][x2][y2]=min(dp[x1][y1][x2][y2],dp[x1+1][y1][x2][y2]);
    dp[x1][y1][x2][y2]=min(dp[x1][y1][x2][y2],dp[x1][y1][x2-1][y2]);
    }
    if(y2-y1>0)
    {
    dp[x1][y1][x2][y2]=min(dp[x1][y1][x2][y2],dp[x1][y1+1][x2][y2]);
    dp[x1][y1][x2][y2]=min(dp[x1][y1][x2][y2],dp[x1][y1][x2][y2-1]);
    }
    //printf("dp[%d][%d][%d][%d]=%d
",x1,y1,x2,y2,dp[x1][y1][x2][y2]);
    if(i>0 && j>0)
    ans=max(ans,sum[k][l]-sum[k][j-1]-sum[i-1][l]+sum[i-1][j-1]-dp[x1][y1][x2][y2]);
    else if(i>0)
    ans=max(ans,sum[k][l]-sum[i-1][l]-dp[x1][y1][x2][y2]);
    else if(j>0)
    ans=max(ans,sum[k][l]-sum[k][j-1]-dp[x1][y1][x2][y2]);
    else
    ans=max(ans,sum[k][l]-dp[x1][y1][x2][y2]);
    }
    printf("%d
",ans);
    }
    return 0;
    }
