#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<iostream>

using namespace std;

int x[600],y[600];

double r,R;
int nod;
double dist[505][505];
double radius[505];
int adj[505][505];
bool visited[505];

double distfunc(int x,int y,int x1,int y1)
{
    double ans= sqrt((x-x1)*(x-x1) + (y-y1)*(y-y1));
    return ans;
}

int max(int a,int b)
{
    if(a>b)return a;
    return b;
}


int min(int a,int b)
{
    if(a<b)return a;
    return b;
}


void dfs(int a)
{
    if(visited[a]==true)
    {
        return;
    }
    visited[a]=true;

    for(int i=0;i<adj[a][0];i++)
    {
        if(visited[adj[a][i+1]]==false)
        {
            dfs(adj[a][i+1]);
        }
    }
    return;
}

bool isPos(double c)
{
    for(int i=0;i<=(nod+1);i++)
    {
        adj[i][0]=0;
        visited[i]=false;
    }

    for(int i=0;i<nod;i++)
    {
        if((radius[i] - r) < c)
        {
            adj[i][adj[i][0]+1]=nod;
            adj[nod][adj[nod][0]+1]=i;
            adj[i][0]+=1;
            adj[nod][0]+=1;
        }
        if((R-radius[i]) < c)
        {
            adj[i][adj[i][0]+1]=nod+1;
            adj[nod+1][adj[nod+1][0]+1]=i;
            adj[i][0]+=1;
            adj[nod+1][0]+=1;
        }

        for(int j=i+1;j<nod;j++)
        {
            if(dist[i][j]<c)
            {
                adj[i][adj[i][0]+1]=j;
                adj[j][adj[j][0]+1]=i;
                adj[i][0]+=1;
                adj[j][0]+=1;
            }
        }
    }
    dfs(nod);
    if(visited[nod+1]==true)
    {
        return false;
    }
    else return true;
}

int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
        scanf("%lf%lf",&r,&R);

        scanf("%d",&nod);

        for(int i=0;i<nod;i++)
        {
            scanf("%d%d",&x[i],&y[i]);
            radius[i] = sqrt(x[i]*x[i] + y[i]*y[i]);
        }

        for(int i=0;i<nod;i++)
        {
            for(int j=i+1;j<nod;j++)
            {
                dist[i][j]= distfunc(x[i],y[i],x[j],y[j]);
            }
        }

        double l,h,mid;
        l=0;
        h=R-r;

        while(1)
        {
            int check;

            check = (int)((h-l)*10000);
            mid=(l+h)/2;

            if(check==0) break;

            if(isPos(mid))
            {
                l=mid;
            }
            else
            {
                h=mid;
            }
        }
        printf("%.3f
",mid);
    }
    return 0;
}
