 #include<cstdio>
 #include<iostream>
 #include<algorithm>
 //typedef long long unsigned int l;
 using namespace std;

 int main()
 {
     int t,n;
     int p[10005],ans,i,j,r;
     int cm,sum;
     for(i=0;i<=10005;i++)
        p[i]=1;
     scanf("%d",&t);
     p[1]=0;

     for(i=2;i<=10000 && p[i]==1;i++)
     {
         j=2*i;
         while(j<=10000)
         {
             p[j]=0;
             j=j+i;


         }
      }

    while(t--)
     {
         ans=2000000000;
         scanf("%d",&n);
         int a[n];
         for(i=0;i<n;i++)
            scanf("%d",&a[i]);


       for(i=2;i<=10000;i++)
         {
             cm=0;
             sum=0;
             if(p[i]==0)
                continue;
             for(j=0;j<n;j++)
             {
                 if(a[j]>cm)
                    {
                        //r=(a[j]-1)/i;
                        //cm=(r+1)*i;
                        cm=((a[j]+i-1)/i)*i;

                    }

                 sum=sum+cm-a[j];

             }
          ans =min(ans,sum);
         }
         printf("%d
",ans);
     }


 }
