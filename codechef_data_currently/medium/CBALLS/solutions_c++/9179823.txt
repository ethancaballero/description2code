#include<cstdio>
using namespace std;
int a[10004],prm[10004],p[2048];
int main()
{
    int t,i,j,k,mx,ians,ans,pn,n,pr,tans,tmp;
    pn=0;
    for(i=2;i<10000;i++)
    {
        if(prm[i]==0)
        {
            p[pn]=i;
            pn++;
            for(j=i*i;j<10000;j+=i)
            {
                prm[j]=1;
            }
        }
    }
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        mx=a[0];
        ians=0;
        for(i=1;i<n;i++)
        {
            if(a[i]>mx)
                mx=a[i];

            ians+=mx-a[i];
            a[i]=mx;
        }
        ans=n;
        for(i=0;i<pn;i++)
        {
            pr=p[i];
            tans=0;
            for(j=0;j<n;j++)
            {
                if(a[j]%pr!=0)
                {
                        tmp=pr-(a[j]%pr);
                        tans+=tmp;
                }
            }
            if(tans<ans)
                ans=tans;
        }
        ans+=ians;
          printf("%d
",ans);
    }
    return 0;
}
