#include <iostream>
#include <algorithm>
#include <cstdlib>
#include <cstdio>
using namespace std;
typedef long long ll;
ll a[10005];
bool isprime[10005];
void sieve(int n){
for(int i=0;i<=n;i++)
    isprime[i]=true;
isprime[0]=false;
isprime[1]=false;
for(int i=2;i*i<=n;i++){
    if(isprime[i]==true)
        for(int j=i*i;j<=n;j+=i)
        isprime[j]=false;
}
}
int main()
{
    int t;
   scanf("%d",&t);
    while(t--){
    int n;
    scanf("%d",&n);
    sieve(10000 );
    ll  ans=10000000;
    for(int i=0;i<n;i++)
     scanf("%d",&a[i]);
    for(ll i=2;i<=1000;i++){// numbers from 2 to 1000 as potential gcd
            if(isprime[i]==false)
            continue;
    else{
        ll curr=0;
        ll diff=0;
        for(ll j=0;j<n;j++){
            if(a[j]>curr ){
                curr=((a[j]+i-1)/i)*i;
                //if(i==3)
                 //  cout<<"curr is "<<curr<<endl;

            }
            diff+=curr-a[j];
           // if(i==3)
              //  cout<<"diff is "<<diff<<endl;
        }
        // cout<<" diff is "<<diff<<endl;
        ans=min(ans,diff);
    }
    }
    printf("%lld 
",ans);
    }

    }
