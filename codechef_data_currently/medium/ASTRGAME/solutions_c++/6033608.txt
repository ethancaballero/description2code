int rec(int s,int e)
{
    //printf("s=%d e=%d
",s,e);
    if(s>e)
    {
        //printf("s=%d e=%d ret=0
",s,e);
        return 0;
    }
    if(dp[s][e]!=-1)
    {
        //printf("s=%d e=%d ret=%d
",s,e,dp[s][e]);
        return dp[s][e];
    }
    int i,j;
    vector<int> v;
    F(i,s,e+1)
    {
        F(j,i,e+1)
        {
            if(sub[i][j])
            {
                int cal=rec(s,i-1)^rec(j+1,e),num;
                v.push_back(cal);
            }
        }
    }
    sort(v.begin(),v.end());
    int r=0;
    F(i,0,v.size())
    {

        if(v[i]!=r)
            break;
        if(i==v.size()-1||v[i]!=v[i+1])
            r++;
    }
    dp[s][e]=r;
    //printf("s=%d e=%d ret=%d
",s,e,dp[s][e]);
    return dp[s][e];
}
