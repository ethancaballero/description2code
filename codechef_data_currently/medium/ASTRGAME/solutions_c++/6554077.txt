string a[35];
int n;
map < string , int > dp;
int memoize( string str )
{
    if( dp.count(str) == 1 )
        return dp[str];
    int ans = 0;
    si myset;
    REP(i,0,n - 1)
    {
        int len = a[i].length();
        int siz = str.length();
        REP(j,0,siz - 1)
        {
            int q = j;
            int k = 0;
            while( k < len && q < siz )
            {
                if( str[q] == a[i][k])
                {
                    k++;
                    q++;
                }
                else
                    break;
            }
            if( k == len )
            {
                string p = "";
                REP(s,0,j - 1)
                    p += str[s];
                string r = "";
                REP(s,q,siz - 1)
                    r += str[s];
                //cout << p << " " << r << endl;
                int sg = memoize( p ) ^ memoize( r );
                myset.insert(sg);
            }
        }
    }
    int i = 0;
    while( 1 )
    {
        if( myset.count(i) == 0 )
            return dp[str] = i;
        i++;
    }
}
