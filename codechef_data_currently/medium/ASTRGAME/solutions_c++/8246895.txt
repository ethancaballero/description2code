int solve(int i, int j)
{
	if (i > j) return 0;
	if (dp[i][j] != -1)
		return dp[i][j];
	bool g[100000];
	clr(g, 0);
	int w = 0;
	bool ok = 0;
	rep(u, ss.size()){
		ok = 0;
		int len = ss[u].length();
		rep(v, pos[u].size()){
			if (i <= pos[u][v] && pos[u][v] + len <= j){
				
				w = solve(i, pos[u][v]) ^ solve(pos[u][v] + len, j);
				g[w]=1;
			}
		}
	}
	int ret = 0;
	while (g[ret])
		ret++;
	return dp[i][j] = ret;
}
