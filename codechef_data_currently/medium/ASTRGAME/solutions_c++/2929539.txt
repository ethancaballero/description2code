#include <cstdio>
#include <cstring>

int main(){
  int T;
  for(scanf("%d", &T); T--; ){
    char word[50];
    char dictionary[50][50];
    int N;
    scanf("%s %d", word, &N);
    int len=strlen(word);
    int dlen[50];
    for(int i=0; i<N; i++){
      scanf("%s", dictionary[i]);
      dlen[i]=strlen(dictionary[i]);
    }
    int grundy[50][50];
    for(int i=0; i<len; i++)
      grundy[i][i]=0;
    for(int k=1; k<=len; k++){
      for(int i=0; i+k<=len; i++){
        int mex[50]={0};
        for(int d=0; d<N; d++){
          for(int p=i; p+dlen[d]<=i+k; p++){
            if(memcmp(dictionary[d], word+p, dlen[d])==0)
              mex[grundy[i][p]^grundy[p+dlen[d]][i+k]]=1;
          }
        }
        grundy[i][i+k]=0;
        while(mex[grundy[i][i+k]])
          grundy[i][i+k]++;
      }
    }
    puts(grundy[0][len] ? "Teddy" : "Tracy");
  }
  return 0;
}
