int grundy[35][35],n;
char a[35];
char dict[35][35];
bool same(char first[],char second[])
{
	int j,stop=strlen(second);
	for(j=0;j<stop;j++)
	{
		if(first[j]!=second[j])
		break;
	}
	if(j==stop)
	return true;
	return false;
}
int solve(int start,int end)
{
	if(end<start)
	return 0;
	if(grundy[start][end]!=-1)
	return grundy[start][end];
	bool mex[200];
	int i,j,k;
	rep(i,0,200) mex[i]=false;
	for(k=0;k<n;k++)
	{
		int sz=strlen(dict[k]);
		for(i=start;i<=end-sz+1;i++)
		{
			
			if(same(a+i,dict[k])==true)
			{
				int x=solve(start,i-1);
				int y=solve(i+strlen(dict[k]),end);
				int p=x^y;
				mex[p]=true;
			}
		}
	}
	for(i=0;i<200;i++)
	if(mex[i]==false)
	break;
	grundy[start][end]=i;
	return i;
}
int main()
{
	int t;
	sc(t);
	while(t--)

		scs(a);
		int len=strlen(a);
		int i,j;
		for(i=0;i<=34;i++)
		for(j=0;j<=34;j++)
		grundy[i][j]=-1;
		sc(n);
		rep(i,0,n)
		scs(dict[i]);
		int res=solve(0,len-1);
		if(res==0)
		cout<<"Tracy"<<endl;
		else cout<<"Teddy"<<endl;
	}
}
