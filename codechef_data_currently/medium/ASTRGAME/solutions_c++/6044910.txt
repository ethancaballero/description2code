int n;
string s;
vs v;
map<string, int> m;

void init() {
  cin >> s >> n;
  v = vs(n);
  for (int i = 0; i < n; i++) cin >> v[i];
}

int SG(string ss) {
  if (!ss.size()) {
    return 0;
  }
  
  if (m.find(ss) != m.end()) {
    return m[ss];
  }
  
  set<int> have;
  for (int i = 0; i < n; i++) {
    int pre = 0;
    while (true) {
      size_t found = ss.find(v[i], pre);
      if (found == string::npos) {
        break;
      }
      string s1 = ss.substr(0, found);
      string s2 = ss.substr(found + v[i].size());
      // cout << "Case" << endl;
      // cout << ss << " " << v[i] <<endl;
      // cout << s1 << " * " << s2 << endl;
      int sg = SG(s1) ^ SG(s2);
      have.insert(sg);
      pre = found + 1;
    }
  }
  for (int i = 0; i <= 10000000; i++) {
    if (!have.count(i)) {
      return m[ss] = i;
    }
  }
}

void solve() {
  m.clear();
  int res = SG(s);
  printf("%s
", res ? "Teddy" : "Tracy");
}
