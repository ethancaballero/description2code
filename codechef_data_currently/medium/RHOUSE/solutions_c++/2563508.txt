void union_(int a , int b)
{
	if(sizee[a]>=sizee[b])
	{
		par[b]=par[a];
		sizee[a]+=sizee[b];
        // if b had any R , so now a will also have
		isR[a]=isR[a]|isR[b];
	}
	else
	{
		par[a]=par[b];
		sizee[b]+=sizee[a];

		isR[b]=isR[b]|isR[a];
	}
}
