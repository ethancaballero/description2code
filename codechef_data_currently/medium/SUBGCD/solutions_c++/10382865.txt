#include<stdio.h>
 
#define g getchar_unlocked
#define ll int
 
inline ll inp()
{
ll x=0,sign=1;
char c;
while(((c=g())<'0'||c>'9')&&c!='-');
if(c=='-')	sign=-1,c=g();
while(c>='0'&&c<='9')
x=(x<<3)+(x<<1)+c-48,c=g();
return sign*x;
}
 
int gcd(int a,int b)
 {
 if(a==0)
 return b;
 return(gcd(b%a,a));
 }
 
int main()
{
 int i,n,temp,a[100000],t;
 t=inp();
 while(t--)
 {
 scanf("%d",&n);
 for(i=0;i<n;i++)
 {a[i]=inp();
  temp=(i==0)?a[i]:gcd(temp,a[i]);
 }
 
 if(temp==1)
 printf("%d
",n);
 else
 printf("-1
");
 }
return 0;
}
