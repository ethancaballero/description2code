// METHOD 2 (by two pointers)
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,t,n,ans,j,sum,d;
	scanf("%d",&t);
	while(t--)
	{
		ans=sum=0;
		scanf("%d",&n);
		vector<int> v(3001);
		for(i=0;i<n;i++)
		{
			scanf("%d",&j);
			sum+=j;
			v[j]++;
		}
		if(n>1)
		{
			if(sum%n)
			{
				ans=-1;
			}
			else
			{
				sum/=n;
				for(i=0,j=n;(i<sum)||(j>sum);)
				{
					while((i<sum)&&(!v[i]))
						i++;
					while((j>sum)&&(!v[j]))
						j--;
					d=(j-i+1)/2;
					if(!d)
						break;
					v[i+d]++;
					v[j-d]++;
					v[i]--;
					v[j]--;
					ans++;
				}
			}
		}
			printf("%d
",ans);
	}
	return 0;
}
