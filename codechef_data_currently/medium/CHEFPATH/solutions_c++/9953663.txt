inline int fi() { //fast input function
    register int c = getcx();
    int x = 0;
    int neg = 0;
    for (;
        ((c < 48 || c > 57) && c != '-'); c = getcx());
    if (c == '-') {
        neg = 1;
        c = getcx();
    }
    for (; c > 47 && c < 58; c = getcx()) {
        x = (x << 1) + (x << 3) + c - 48;
    }
    if (neg) x = -x;
    return x;
}
