int solve(string s,string m,int k){
	
	int pl=-1;
	int pe=0;
	for(int i=0;i<L(m);i++){
		int npl=-1,npe=0;
		for(int j=9;j>=0;j--){
			if(i>k-1 and i<k+L(s) and !cp[s[i-k]-'0'][j]){
				continue;
			}
			if(npl<pl*10+j) npl=pl*10+j;
			if(j<m[i]-'0' and npl<pe*10+j) npl=pe*10+j;
			if(j== m[i]-'0' and npe<pe*10+j)npe=pe*10+j;
		}
		pe=npe,pl=npl;
	}
	return max(pe,pl);
}
