		mini = 100000000000LL;
		for (i = 0; i < n - 1; i++) {
			for (j = i + 1; j < n; j++) {
				if (abs(a[i] + a[j] - k) < mini) {
					mini = abs(a[i] + a[j] - k);
				}
			}
		}
		
		count = 0;
		for (i = 0; i < n - 1; i++) {
			for (j = i + 1; j < n; j++) {
				if (abs(a[i] + a[j] - k) == mini) {
					count++;
				}
			}
		}
