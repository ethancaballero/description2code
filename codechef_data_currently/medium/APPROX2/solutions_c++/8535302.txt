#include<iostream>
#include<stdlib.h>
#include<cstdio>
#include<climits>
using namespace std;
int main()
{
    int t,n,k,arr[1001];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);
        for(int i=0; i<n; i++)
            scanf("%d",&arr[i]);
        int mn = INT_MAX;
        for(int i=0; i<n-1;i++)
        {
            for(int j=i+1; j<n;j++)
            {
                int tmp = arr[i] + arr[j] - k;
                if(tmp < 0)
                    tmp *= -1;
                mn = min(mn,tmp);
            }
        }
        int cnt =0;
        for(int i=0; i<n-1; i++)
        {
            for(int j=i+1; j<n; j++)
            {
                int tmp = arr[i] + arr[j] - k;
                if(tmp < 0)
                    tmp*=-1;
                if(tmp == mn)
                    cnt++;
            }
        }
        printf("%d %d
",mn,cnt);
    }
   return 0;
}
