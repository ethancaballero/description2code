#include<bits/stdc++.h>
using namespace std;
#define for1(i,n) for(i=0;i<n;i++)
#define for2(i,n) for(i=n-1;i>=0;i--)
#define ll long long int 
#define CLEAR(array, value) memset(array, value, sizeof(array));
#define si(a)     scanf("%d", &a)
#define sl(a)     scanf("%lld", &a)
#define sc(a)     scanf(" %c", &a)
#define ss(a)     scanf("%s", a)
#define pi(a)     printf("%d
", a)
#define pl(a)     printf("%lld
", a)
#define pn        printf("
")

#define M 1299709
ll gcd(ll a,ll b)
{
if(b==0)
return a;
else
return gcd(b,a%b);
}


int main()
{
	ll t;
	sl(t);
	while(t--)
	{
		ll n,i,j,k,min,p1,p2,temp,o=0,counter=0;
		sl(n);sl(k);
		ll a[n];
		for1(i,n)
		{
			sl(a[i]);
		}
		sort(a,a+n);
		p1=0;p2=n-1;
		while(p1<p2)
		{
			temp=a[p1]+a[p2]-k;
			
			if(temp>0)
			{
				p2--;
			}
			else if(temp<0)
			p1++;
			else
			{
				min=0;
				break;
			}
			if(o==0)
			{
				min=temp<0?-temp:temp;
			
			}
			else
			{
				temp=temp<0?-temp:temp;
				if(min>temp)
				min=temp;
			}
			o++;
			//	printf("temp=%lld min=%lld 
",temp,min);
		}
		//pl(min);
		for1(i,n)
		{
			for1(j,i)
			{
				temp=a[i]+a[j]-k;
				if(min<temp)
				break;
				if(min==temp || min==temp*-1)
				counter++;
			//	printf("%lld ",temp);
			}
			//pn;
		}
		printf("%lld %lld
",min,counter);
	}
  return 0;
}
