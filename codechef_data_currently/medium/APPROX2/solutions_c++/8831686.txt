#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,k,d,c=0,j;
		scanf("%lld%lld",&n,&k);
		long long int a[n],i,min;
		for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
		sort(a,a+n);
		min=abs((a[0]+a[1])-k);
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				if(min>(abs((a[i]+a[j])-k)))
				{
					min=abs((a[i]+a[j])-k);
				}
				
			}
		}
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				d=abs((a[i]+a[j])-k);
				if(d==min)
				{
					c++;
				}
				
			}
		}
		printf("%lld %lld
",min,c);
		
		
		
	}
	return 0;
}
