#include <stdio.h>
#include <limits.h>
#include<stdlib.h>
int main()
{
    int T;
    scanf("%d", &T);
    while(T--)
    {
        int N; int K;
        scanf("%d %d", &N, &K);
        int farr[N], i,j;
        for(i=0; i<N; i++)
            scanf("%d", &farr[i]);
        int count = 0;
        int mindif = INT_MAX, diff;
        for(i=0; i<N-1; i++)
        {
            for(j= i+1; j<N; j++)
            {
                diff = abs(farr[i]+farr[j]-K);
                if(diff == mindif)
                    count++;
                if(diff<mindif)
                {
                    mindif = diff;
                    count = 1;
                }
            }
        }

        printf("%d %d
", mindif, count);

    }
    return 0;
}
