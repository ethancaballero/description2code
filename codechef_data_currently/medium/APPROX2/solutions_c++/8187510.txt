#include<stdio.h>
#include<stdlib.h>
int main() {
	long long int t,i,j,a[1005],sum[1000005],k,d,mini,n,ans,temp;
	scanf("%lld",&t);
	while(t--) {
		scanf("%lld %lld",&n,&d);
		for(i=0;i<n;i++) {
			scanf("%lld",&a[i]);
		}
		k=0;
		for(i=0;i<n;i++) {
			for(j=i+1;j<n;j++) {
				sum[k++] = a[i]+a[j];
			}
		}
		mini = abs(sum[0]-d);
		ans=1;
		for(i=1;i<k;i++) {
			temp = abs(sum[i]-d);
			if(temp==mini) {
				ans++;
			}
			else if(mini > temp) {
				mini = temp;
				ans=1;
			}
		}
		printf("%lld %lld
",mini,ans);
	}
	return 0;
}
