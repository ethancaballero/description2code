#include <cstdio>
using namespace std;

#define N 1003

long long a[N];

long long mn, cnt = 0;

int main()
{
	int tc;
	scanf("%d",&tc);
	
	while(tc--)
	{
		long long n, k;
		mn = 9999999999LL;
		cnt = 0;
		
		scanf("%lld%lld",&n,&k);
		
		for(int i = 0; i < n; ++i)
			scanf("%lld",&a[i]);
		
		// All duplets
		int p = 0;
		for(int i = 0; i < n; ++i)
		 for(int j = i + 1; j < n; ++j)
		 	{
		 		long long s = a[i] + a[j] - k;
		 		if(s < 0)s *= -1;
		 		if(s < mn)
		 		{
		 			mn = s;
		 			cnt = 1;
				}
				else if(s == mn)
				{
					cnt++;
				}
			}
		printf("%lld %lld
", mn, cnt);
	}
	return 0;
}
