#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	int t,k,n,i,d,h,r,s,x,j,f;
	cin>>t;
	while(t--)
	{
		d=h=f=0;
		cin>>n;
		int H[n],D[n];
		for(i=0;i<n;i++)
		cin>>H[i];
		for(i=0;i<n;i++)
		cin>>D[i];
		if(n==2)
		{
			r=H[0]-D[0];
			if(r==D[1]-H[1])
			cout<<fabs(r)<<"
";
			else
			cout<<"-1
";
			continue;
		}
		for(i=0;i<n;i++)
		{
			d+=D[i];
			h+=H[i];
		}
			if((d-h)%(n-2)==0)
			{
				s=(d-h)/(n-2);
				if(s>=0)
				{
					for(j=0;j<n;j++)
					{
						x=H[j]+s-D[j];
						if(x%2!=0||x<0)
						{
							f=1;
							break;
						}
					}
					if(!f)
					{
						cout<<s<<"
";
						continue;
					}
				}
			}
		cout<<"-1
";
	}
	return 0;
}
