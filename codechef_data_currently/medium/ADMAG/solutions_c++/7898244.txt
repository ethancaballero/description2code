#include <iostream>
#include <stdio.h>

using namespace std;

long long int func(long long int n, long long int arr[])
{
    if(arr[n]!=0)
    {   return arr[n];}

    arr[n] = func(n-1, arr) + func(n-2, arr);
    return arr[n];
}

int main()
{
    long long int arr[1001];
    long long int n=0, trav=0;
    int t=0;

    for(int i=0; i<1001; i++)
    {   arr[i]=0;}
    arr[0]=1;
    arr[1]=1;
    func(1000,arr);

    cin>>t;
    while(t--)
    {   scanf("%lld", &n);
        trav = 0;
        while(arr[trav]<=n || trav==0)
        {   trav++;}
        printf("%lld
", trav-1);
    }
    return 0;
}
