#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define MX 1000009
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m,a[MX]={},i;
		scanf("%d %d",&n,&m);
		for(i=1;i<=n;i++)
			a[i]=1;
		for(i=0;i<m;i++)
		{
			int w,x,y,z;
			scanf("%d %d %d %d",&w,&x,&y,&z);
			if(w==1)
			{
				a[x]+=z;
				if(y<n)
					a[y+1]-=z;
			}
			else
			{
				a[x]-=z;
				if(y<n)
					a[y+1]+=z;
			}
		}
		ll max_sum=INT_MIN,cur_sum=0;
		for(i=2;i<=n;i++)
		{
			if(cur_sum>0)
				cur_sum+=a[i];
			else
				cur_sum=a[i];
			if(cur_sum>max_sum)
				max_sum=cur_sum;
		}
		cur_sum=0;
		for(i=2;i<=n;i++)
		{
			if(cur_sum>0)
				cur_sum-=a[i];
			else
				cur_sum=-a[i];
			if(cur_sum>max_sum)
				max_sum=cur_sum;
		}
		printf("%lld
", max_sum);
	}
	return 0;
}
