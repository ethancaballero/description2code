#include<bits/stdc++.h>

bool isSubsetSum(int set[], int n, int sum)
{
   if (sum == 0)
     return true;
   if (n == 0 && sum != 0)
     return false;
     
   if (set[n-1] > sum)
     return isSubsetSum(set, n-1, sum);


   return isSubsetSum(set, n-1, sum) || isSubsetSum(set, n-1, sum-set[n-1]);
}

int main()
{
  int t,n,m;
  scanf("%d",&t);
  while(t--)
  {
      scanf("%d %d",&n,&m);
      int arr[n];
      for(int i=0;i<n;i++)
        scanf("%d",&arr[i]);
      if(isSubsetSum(arr, n, m) == true)
        printf("Yes
");
      else
        printf("No
");
  }
  return 0;
}
