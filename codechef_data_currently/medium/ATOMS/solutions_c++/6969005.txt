#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#define pb push_back
using namespace std;

int p[100],rank[100];

void create()
{
int i;
for(i=0;i<100;i++) {p[i] = i; rank[i] = 1;}
}

int find(int x)
{
if(x!=p[x]) p[x] = find(x);
return p[x];
}

void merge(int x,int y)
{
int px = find(x);
int py = find(y);
if(rank[px] >= rank[py]) rank[px]+=rank[py] , p[py] = px;
else rank[py]+=rank[px] , p[px] = py;

}
