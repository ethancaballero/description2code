#include <stdio.h>
int table[100];

int main(){
	register int T, N, M, L, tmp, bitFlag, i, j;
	scanf("%d", &T);

	while(T--){
		scanf("%d %d", &N, &M);
		for(i=0; i<N; ++i) table[i] = 0;

		for(i=0; i<M; ++i){
			bitFlag = 1 << i;
			scanf("%d", &L);
			while(L--){
				scanf("%d", &tmp);
				table[tmp] |= bitFlag;
			}
		}

		for(i=0; i<N; ++i) for(j=0; j<i; ++j) if(table[i] < table[j]) table[i] ^= table[j] ^= table[i] ^= table[j];
		tmp = 1;
		for(i=1; i<N; ++i) if((table[i] ^ table[i-1]) != 0) ++tmp; 

		printf("%d
",tmp);
	}
	return 0;
}
