#include<cstdio>
#include<cstdlib>
#include<iostream>
using namespace std;
int diff(const void*a,const void*b)
{
	return *(int*)a-*(int*)b;
}
int main()
{
	int t,m,n;
	int i,j,v;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		
		int s[n],c;
		for(i=0;i<n;i++)
		{
			s[i]=0;
		}
		i=0;
		while(m--)
		{
			for(i=0;i<n;i++)
		{
			s[i]=s[i]*2;
		}
			scanf("%d",&c);
		
			for(j=0;j<c;j++)
			{
				scanf("%d",&v);
				s[v]=s[v]+1;
			}
			
		}
		qsort(s,n,sizeof(int),diff);
		int count=1;
		for(i=1;i<n;i++)
		{
			if(s[i-1]!=s[i])
			{
				count++;
			}
		}
		printf("%d
",count);
	}
	return 0;
}
