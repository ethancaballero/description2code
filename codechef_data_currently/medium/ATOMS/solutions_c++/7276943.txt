#include<stdio.h>
 
int main() 
{
    int n,m,t,i,j,k,l;
    scanf("%d",&t);
    int s[100];
    while (t--) 
    {
        scanf("%d%d",&n,&m);
        for (i = 0; i < n; ++i) 
            s[i] = 0;
        for (i = 0; i < m; ++i) 
        {
            scanf("%d",&k);
            while (k--) 
            {
                scanf("%d",&l);
                s[l] |= (1<<i);
            }
        }
        int count=n;
        for(i=0;i<n;i++)
        {
            for(j=0;j<i;j++)
                if(s[i]==s[j])
                {
                    count--;
                    break;
                }
        }
        printf("%d
",count);
    }
    return 0;
}
