#include<iostream>
#include<vector>
#include<cstring>
using namespace std;

 vector<int> v[100];
 int mark[100];

void mt(int x)
{
      vector<int> ::iterator t=v[x].begin();
                  while(t!=v[x].end())
                  {
                      if(mark[*t]==0)
                      {
                          mark[*t]=1;
                          mt(*t);
                       }
                       t++;
                  }
}

int main()
 {
    int T;

     cin>>T;
     while(T--)
     {  int n,m,x,y,i,j;
        int a[100]={0};
         cin>>n>>m;
         memset(mark,0,sizeof(mark));
         for(i=0;i<m;i++)
         {
             cin>>x;
             for(j=0;j<x;j++)
             {
                 cin>>y;
                 a[y]|=1<<i;
             }
         }

         for(i=0;i<n;i++)
            for(j=i+1;j<n;j++)
               {
                    if((a[i]^a[j])==0)
                        {
                            v[i].push_back(j);
                        }

                }
                int count=0;
        for(i=0;i<n;i++)
        {
            if(mark[i]==0)
              {
                   count++;
                  mark[i]=1;
                  vector<int> ::iterator t=v[i].begin();
                  while(t!=v[i].end())
                  {
                      if(mark[*t]==0)
                      {
                          mark[*t]=1;
                          mt(*t);
                       }
                       t++;
                  }

              }

        }
        for(i=0;i<100;i++)
            v[i].clear();
        cout<<count<<"
";
     }
 }
