#include<stdio.h>
int main()
{
    int i,j,k,t,n,m;
    int count=0;
    scanf("%d",&t);
    while(t--)
    {
        int S[35][105]={{0}};
        count=0;
        int data[105]={0};
        scanf("%d %d",&n,&m);
        // m is no of sets.. & n is there range.. !!
        for (i=0;i<m;i++)
        {
            scanf("%d",&S[i][102]);  // S[i][102] will store the cardinality of ith set..
            for(j=1;j<=S[i][102];j++)
            {
                scanf("%d",&k);
                S[i][k]=1;
            }
        } // we have scanned..

        for(i=0;i<n;i++)
        {
            if(data[i]==1)
            continue;
            data[i]=1;
            for(j=i+1;j<n;j++)
            {
                // checking compatibility of i & j
                for(k=0;k<m;k++)
                {
                    if(S[k][i]==0 && S[k][j]==0)
                    {
                        continue;
                    }
                    else if (S[k][i]==1 && S[k][j]==1)
                    {
                        continue;
                    }
                    else
                    {
                        break;
                    }
                }
                if(k==m && ((S[k-1][i]==0 && S[k-1][j]==0) || (S[k-1][i]==1 && S[k-1][j]==1)))
                {
                    data[j]=1;
                }
            }
            count+=1;
        }
        printf("%d
",count);
    }
    return 0;
}
