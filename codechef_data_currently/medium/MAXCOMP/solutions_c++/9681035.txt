#include<stdio.h>
int m[49][49],dp[49];
int maxm(int a,int b)
{
    if(a>b) return a;
    return b;
 
}
int main()
{
    int t,n,s,e,i,j;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d%d%d",&s,&e,&j);
            m[s][e]=maxm(m[s][e],j);
        }
        for(e=0;e<49;e++)
        {
            dp[e]=0;
            for(s=0;s<e;s++)
            {
                dp[e]=maxm(dp[e],dp[s]+m[s][e]);
                m[s][e]=0;
            }
        }
        printf("%d
",dp[48]);
    }
    return 0;
}
