#include <bits/stdc++.h>
using namespace std;

int gcd(int a,int b) {
    if(b == 0) {
        return a;
    }
    else {
        return gcd(b,(a%b));
    }
}

int main() {
    int t;
    cin >> t;

    while(t--) {
        int n;
        cin >> n;

        int a[n],i,j;
        for(i = 0;i < n;i++) {
            cin >> a[i];
        }

        int g = 0;
        for(i = 0;i < n;i++) {
            g = gcd(a[i],g);
        }

         int ans = g;

        if(g == 1) {
            cout << "-1" << "
";
        } else {
            for(i = 2;(i*i) <= g;i++) {
                if((g % i) == 0) {
                    ans = i;
                    break;
                }
            }
            cout << ans << "
";
        }
    }
    return 0;
}
