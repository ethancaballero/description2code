#include<cstdio>
#include<iostream>
using namespace std;
#define ll long long
ll rep_sqr(int base,int exp,int mod)
{
	ll res = 1;
	while(exp!=0)
	{
		if(exp%2==1)	res = (res*base)%mod;
		base =  ((ll)base*base)%mod;
		exp/=2;
	}
	return res%mod;
}
int main()
{
    int k=1;
	int p[25];
	int flag[101];
	for(int i=0;i<101;i++)
		flag[i]=false;
	p[0]=2;
	for(int i=3;i<=100;i+=2)
	{
		if(!flag[i])
		{
			p[k++]=i;
			for(int j=3*i;j<=100;j+=2*i)
				flag[j]=true;
		}
	}
	int n;
	scanf("%d",&n);
	int *arr = new int[n+1];
	for(int i=1;i<n+1;i++)
		scanf("%d",&arr[i]);
	int **cf = new int*[n+1];
	for(int i=0;i<n+1;i++)
		cf[i] = new int[25];
	for(int i=0;i<n+1;i++)
		for(int j=0;j<25;j++)
			cf[i][j] = 0;
	for(int i=1;i<n+1;i++)
	{
		int t = arr[i];
		for(int j=0;j<25;j++)
		{
			cf[i][j]+=cf[i-1][j];
			while(t%p[j]==0)
			{
				t/=p[j];
				cf[i][j]++; 
			}
		}
	}
	int q;
	scanf("%d",&q);
	while(q--)
	{
		int l,r,m;
		scanf("%d%d%d",&l,&r,&m);
		ll result = 1;
		for(int j=0;j<25;j++)
		{
			int exp = cf[r][j]-cf[l-1][j];
			result = (result*rep_sqr(p[j],exp,m))%m;
		}
		printf("%d
",result);
	}
	return 0;
}
