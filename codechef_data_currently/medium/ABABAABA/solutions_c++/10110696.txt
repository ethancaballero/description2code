#define rf freopen("in.in", "r", stdin)
#define wf freopen("out.out", "w", stdout)
#define rep(i, s, n) for(int i=int(s); i<=int(n); ++i)
using namespace std;
const int mx = 1e5 + 10, mod = 1e9+7;

int main()
{
	//rf;// wf;
	ios::sync_with_stdio(0);

	int t; cin >> t;
	while(t--)
	{
		string s;
		cin >> s;
		int n = s.size();

		int cnt_a = 0, cnt_b = 0;
		rep(i, 0, n-1)
		{
			cnt_a += (s[i] == 'A');
			cnt_b += (s[i] == 'B');
		}

		if(cnt_a == 2)
		{
			cout << "A
";
			continue;
		}
		else if(cnt_b == 2)
		{
			cout << "B
";
			continue;
		}
		else if(n < 4)
		{
			cout << "-1
";
			continue;
		}

		string ans = "";
		s = '.' + s + '.';
		for(int i = 0; i <= n-2; ++i)
		{
			if(s[i+2] != s[i+3] and s[i+1] == s[i+2] and s[i] != s[i+1])
			{
				rep(j, 1, n)
				{
					if(j!=i+1)
						ans += s[j];
				}
				break;
			}
		}

		cout << ((ans == "")? "-1": ans) << endl;
	}
			
	return 0;
}
