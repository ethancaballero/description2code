#include<bits/stdc++.h> 
    using namespace std;
     
     
     
    typedef long long LL;
     
     
    LL T,n,k,ans;
    const LL mod=1e9+7;
    LL powmod(LL a,LL b,LL c)
    {
    	LL ans=1;
    	while (b)
    	{
    		if (b%2==1) ans=ans*a%c;
    		b/=2;
    		a=a*a%c;
    	}
    	return ans;
    }
     
    int main()
     
    {
    	cin>>T;
    	while (T--)
    	{
    		cin>>n>>k;
    		ans=powmod(k-1,n-1,mod);
    		ans=ans*k%mod;
    		cout<<ans<<endl;
    	}
     
    	return 0;
     
    }
