#include<stdio.h>
int main()
{
    int t,n;
    char x1,x2,x3;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        if(360%n==0)
            x1='y';
        else
            x1='n';
        if(n<=360)
            x2='y';
            else x2='n';
        if (n<=26)
            x3='y';
        else x3='n';
        printf("%c %c %c
",x1,x2,x3);
    }
    return 0;
}
