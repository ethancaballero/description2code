#include<bits/stdc++.h>
using namespace std;
#define N 100001
int st[4*N];
char str[N];
char ans[N];
int get(int x,int y)
{
    if(x== -1) return y;
    if(y== -1) return x;
	return str[x]<=str[y] ? x : y ;
}
void constr(int in,int low,int high)
{
	if(low==high)
	{
	  st[in]=low;
	  return;
	}
	int mid=low+((high-low)/2);
	constr(2*in+1,low,mid);
	constr(2*in+2,mid+1,high);
	st[in]=get(st[2*in+1],st[2*in+2]);
}
int getmin(int in,int low,int high,int qs,int qe)
{
	if(qs<=low && qe>=high)
	{
	return st[in];
	}
	if(qs>high || qe<low || qs>qe)
	{
	return -1;
	}
	int mid=low+((high-low)/2);
	int left=getmin(2*in+1,low,mid,qs,qe);
	int right=getmin(2*in+2,mid+1,high,qs,qe);
	return get(left,right);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
	 int k;
	 scanf("%s",str);
	 scanf("%d",&k);
	 int l=strlen(str);
	 constr(0,0,l-1);
	 int i=0,j;
     int prev=0;
     for(j=0;j<k;j++)
     {
       int in=getmin(0,0,l-1,prev,l-k+j);
       prev=in+1;
       ans[j]=str[in];     
     }
     for(i=0;i<k;i++)
     {
     printf("%c",ans[i]);
     }
     printf("
");
	}
	return 0;
}
