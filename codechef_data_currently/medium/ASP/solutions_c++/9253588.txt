#include <cstdio>

using namespace std;

int main()
{
   int t,n;
    scanf("%d",&t);
   for(int i=0;i<t;i++){
	scanf("%d",&n);
	 int a[n];
	for(int j=0;j<n;j++)
		scanf("%d",&a[j]);
	int flag=0;
	for(int j=0;j<n-2;j++){
		if(a[j]>a[j+2])
			{flag=1;
			break;}
	}
	if(flag==1)
		printf("NO
");
	else
		printf("YES
");}
   
   return 0;
}
