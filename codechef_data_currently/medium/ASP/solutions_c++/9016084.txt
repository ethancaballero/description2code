#include<bits/stdc++.h>
using namespace std;

typedef long long int ll;
ll a[1000001];
int main()
{
    int t;
    ll n,i,temp,c;
    scanf("%d",&t);
    while(t--){
        c=0;
        int flag = 1;
        scanf("%lld",&n);
        for(i=0;i<n;i++)
                scanf("%lld",&a[i]);
        temp=a[0];
        for(i=1;i<n;i++){
            if(temp > a[i])
                    ++c;
            else
                {
                    c=0;
                    temp=a[i];
                }
            if(c>1)
            {
                flag=0;
                break;
            }
        }
        if(flag == 1)
            printf("YES
");
        else
            printf("NO
");
        }
    return 0;
}
