#include <bits/stdc++.h>
using namespace std;
typedef long long lld;
lld a[1000010];
int main() {
    int t,f; lld n,i,tmp;
    scanf("%d",&t);
    while(t--){
        scanf("%lld",&n);
        for(i=0;i<n;++i){
            scanf("%lld",&a[i]);
        }
        if(n==1||n==2) f=1;
        else {
            f = 1;
            for (i = 0; i < n-1; i++) {
                if (a[i] > a[i + 1]) {
                    tmp = a[i];
                    a[i] = a[i + 1];
                    a[i + 1] = tmp;
                    ++i;
                }
            }
            for(i=0;i<n-1;++i){
                if(a[i]>a[i+1]){
                    f=0; break;
                }
            }
        }
        if(f==1) printf("YES
");
        else printf("NO
");
    }
 
    return 0;
}
