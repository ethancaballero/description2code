#include<stdio.h>
#include<algorithm>
using namespace std;
int a[1000002];
int main()
{
    int t,n,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        scanf("%d",&a[0]);
        int p;
        for(i=1;i<n;i++)
        {
            scanf("%d",&a[i]);
            if(a[i]<a[i-1])
            {   if(i==p+1)
                {
                    break;
                }
                swap(a[i],a[i-1]);
                p=i;
            }

        }
        if(i!=n)
        {   i++;
            for(;i<n;i++) scanf("%d",&a[i]);
            printf("NO
");
            continue;
        }
        int flag=1;
        for(i=1;i<n;i++)
        {
            if(a[i]<a[i-1])
            {
                flag=0;
                printf("NO
");
                break;
            }
        }
        if(flag==1) printf("YES
");
    }
    return 0;
}
