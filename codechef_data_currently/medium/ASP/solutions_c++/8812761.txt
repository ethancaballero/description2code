#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define cl ios_base::sync_with_stdio(false); cin.tie(NULL)
int main() 
{
 cl;
 pair<int,int>p[100001];
 int t;
 cin>>t;
 while(t--)
 {
 	int n,flag=0;
 	cin>>n;
 	for (int i = 0; i <n; ++i)
 	{
 		cin>>p[i].first;
 		p[i].second=i;
 	}
 	sort(p,p+n);
 	for (int i = 0; i < n; ++i)
 	{
 		if (p[i].second == i || abs(p[i].second-i)<=1)
 		{
 			flag++;
 		}
 	}
 	if (flag==n)
 	{
 		printf("YES
");
 	}
 	else
 		printf("NO
");
 }
 return 0;
}
