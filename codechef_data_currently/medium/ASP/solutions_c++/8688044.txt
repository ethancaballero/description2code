#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i,ans=1,a,b,c;
		scanf("%d",&n);
		scanf("%d",&a);
		if(n>1)
		{
			scanf("%d",&b);
			for(i=2;i<n;i++)
			{
				scanf("%d",&c);
				if(c<a)
					ans=0;
				a=b;
				b=c;
			}
		}
		if(ans)
			printf("YES
");
		else
			printf("NO
");
	}
	return 0;
}
