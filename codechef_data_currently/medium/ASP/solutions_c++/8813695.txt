#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#include <vector>
using namespace std;

const int MAXN = 100000;

int a[MAXN], n;

int main()
{
    int tests;
    for (scanf("%d", &tests) == 1; tests --; ) {
        scanf("%d", &n);
        for (int i = 0; i < n; ++ i) {
            scanf("%d", &a[i]);
        }
        for (int i = 1; i < n; ++ i) {
            if (a[i] < a[i - 1]) {
                swap(a[i], a[i - 1]);
		++i;
            }
        }
        bool flag = true;
        for (int i = 1; i < n; ++ i) {
            if (a[i] < a[i - 1]) {
                flag = false;
                break;
            }
        }
        printf("%s
", flag ? "YES" : "NO");
    }
    return 0;
}
