#include<stdio.h>
#include<stdlib.h>

int main()
{
	int i,x,flag,n,t,no1,no2;
	scanf("%d",&t);
	while(t--)
	{
		flag=0;
		scanf("%d",&n);
		no1=no2=0;
		for(i=1;i<=n;i++)
		{
			scanf("%d",&x);
			if(x>=no1)
			{
				no2=no1;
				no1=x;
			}
			else
			{
				if(x>=no2)
				no2=x;
				else
				flag=1;
			}
			
		}
		if(flag==0)
		printf("YES
");
		else
		printf("NO
");
	}
	return 0;
}
