#include<bits/stdc++.h>

using namespace std;

#define MAXM 20000+5

int T;
int N, M;

int a[MAXM];

int main() {
	scanf("%d", &T);
	while (T--) {
	    scanf("%d%d", &N, &M);
		for (int i = 0; i < M; i++) {
			scanf("%d", &a[i]);
		}
		sort(a, a+M);
		int ni = 0, sum = 0;
		for (int i = 0; i < M; i++) {
			sum += (a[i] + 1);
			if (sum >= M) {
			   ni = i;
			   break;
            }
        }
	    printf("%d
", M-ni-1);
	}
	return 0;	
}
