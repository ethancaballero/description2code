ll fast_expo(ll a,ll b) {
	if(b == 0) return 1;
	ll p;
	if(b % 2 == 0) {
		p = fast_expo(a,b / 2);
		p = (p * p) % MOD;
	}
	else p = (a * fast_expo(a,b - 1)) % MOD;
	return p;
}
