typedef long long ll;

#define dim 3

struct matrix {
    int a[dim][dim];
};

#define MOD 1000000007

matrix mul(matrix x, matrix y) {
    matrix res;
    FOR(a,0,dim) FOR(b,0,dim) res.a[a][b] = 0;
    FOR(a,0,dim) FOR(b,0,dim) FOR(c,0,dim) {
	res.a[a][c] += (x.a[a][b] * ll(y.a[b][c])) % MOD;
	res.a[a][c] %= MOD;
    }
    return res;
}




matrix power(matrix m, int n) {
    if(n == 1) return m;
    matrix u = mul(m, m);
    u = power(u, n/2);
    if(n&1) u = mul(u, m);
    return u;
}
