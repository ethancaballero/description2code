//FILE *fin = freopen("in","r",stdin);

const int N =int(2e5)+10;
int P[N],C[N];
int main()
{
	DRT()
	{
		int n,m,q;
		si(n);si(m);si(q);
		int p,r,k;
		si(p);si(r);si(k);
		P[0]=q;
		C[0]=n;
		int wand=0;
		for(int i=r;i<min(k+1,m*r+1);i+=r)
		{
			P[i]+=1;
			wand++;
		}
		int ans=0;
		for(int i=0;i<k+1;i++)
		{
			int go = min(P[i],C[i]);
			if(i+p<k+1)
				ans+=go;
			P[i]-=go;
			C[i]-=go;
			P[i+1]+=P[i];
			C[i+p]+=go;
			C[i+1]+=C[i];
		}
		printf("%d %d %d %d
",C[k],ans,m-wand,P[k]);
		SET(C,0);
		SET(P,0);
	}
	return 0;
}
