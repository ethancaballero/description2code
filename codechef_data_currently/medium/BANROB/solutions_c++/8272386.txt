#include <bits/stdc++.h>

using namespace std;

double power(long long int a,double p)
{
    double ans=1.000000;
    while(a)
    {
        if(a&1)
            ans=ans*p;
        p=p*p;
        a>>=1;
    }
    return ans;
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        double a=1000000000.000000,p;
        long long int m;
        scanf("%lld %lf",&m,&p);
        double ans=0.000000;
        ans=(m%2==0)?(-1)*power(m-1,p):power(m-1,p);
        //printf("%lf
",ans);
        double result=(a*p*(1-ans))/(1+p);
        printf("%.2lf %.2lf
",a-result,result);
    }
    return 0;
}
