#include <cstdio>
#include <cmath>

using namespace std;

int main()
{
	int T, M;
	double p,cash = 1000000000;
	scanf("%d", &T);
	while (T--)
	{
		scanf("%d%lf", &M, &p);
		double fst = cash * (1 - pow(-p, M)) / (p + 1);
		double snd = cash - fst;
		printf("%.4lf %.4lf
", fst, snd);
	}
	return 0;
}
