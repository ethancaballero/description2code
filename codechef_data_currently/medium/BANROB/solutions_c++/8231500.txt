#include <bits/stdc++.h>
#define MAX 1000000000.0
using namespace std;

double power(double p, int m)
{
    if(m==0)    return 1;
    double y = power(p,m/2);
    if((m%2)==0)    return y*y;
    else {
        if(m>0) return y*y*p;
        else    return (y*y)/p;
    }
}

int main()
{
    int t,i,m;
    double p;
    scanf("%d", &t);

    while(t--) {
        scanf("%d %lf", &m, &p);
        double x=power(p,m);
        if((m&1))   x*=-1;
        double ans = ((1-x)/(1+p))*MAX;
        printf("%.2lf %.2lf
", ans, MAX-ans);
    }

    return 0;
}
