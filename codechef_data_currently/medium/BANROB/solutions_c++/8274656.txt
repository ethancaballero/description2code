#include <iostream>
#include <stdio.h>
using namespace std;
 
double pow(double a, long b)
{
    double result = 1.0;
 
    while (b)
    {
        if (b%2==1)
        {
            result *= a;
        }
        b /= 2;
        a *= a;
    }
 
  return result;
}
 
int main()
{
    int t;
    cin>>t;
    long m;
    double p, result;
    while(t--)
    {
        cin>>m>>p;
        result = (1-pow(-p, m))/(1+p);
        result = 1000000000*result;
        printf("%.11lf %.11lf
", result, 1000000000-result);
    }
    return 0;
}
