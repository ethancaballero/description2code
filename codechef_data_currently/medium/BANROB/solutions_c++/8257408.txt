struct sort_pred                        //Function to sort according to the second element in a pair!
{
    bool operator()(const pair<int,int> &left, const pair<int,int> &right)
    {
    	return left.second < right.second;
    }
};
double POW(double a, int b)
{
	double ret=1,Exp=a;
	
	while(b)
	{
		if(b&1)
		{
			ret = ret*Exp;
		}
		Exp = Exp*Exp;
		b/=2;
	}
	return ret;
}
