#include<stdio.h>
#include<math.h>
float power(float p,long int n)
{
	if(n==0)
	return 1;
	if(n==1)
	return p;
	float temp=power(p,n/2);
	if(n%2==0)
	return temp*temp;
	else
	return temp*temp*p;
	
}
int main()
{
	int t;
	double p;
	long int m;
	double ans;
	scanf("%d",&t);
	while(t)
	{
		scanf("%ld%lf",&m,&p);
		if(m%2==0)
	    ans=1000000000*((1-pow(p,m))/(1+p));
	    else
	    ans=1000000000*((1+pow(p,m))/(1+p));
		
		printf("%0.2lf %0.2lf
",ans,1000000000.0-ans);
		t--;
	}
}
