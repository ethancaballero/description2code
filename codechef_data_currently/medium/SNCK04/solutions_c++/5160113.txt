#include<stdio.h>
#define N 1000003
 
int ph[N];
long long s[N+1];
 
int main()
{
    int i,j,t,n;
    for(i=2;i<N;i++)
                    ph[i]=i;
    for(i=2;i<N;i++)
    {
                    if(ph[i]==i)
                    {
                    for(j=i;j<=N;j+=i)
                                    ph[j]=(ph[j]/i)*(i-1);
                                    }
                    s[i]=s[i-1]+ph[i];
    }
    scanf("%d",&t);
    while(t--)
    {
              scanf("%d",&n);
              printf("%lld
",s[n]);
    }
    return 0;     
}
