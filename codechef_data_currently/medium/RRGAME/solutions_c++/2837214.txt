template< class T > inline T _gcd(T a,T b)
{
    a=_abs(a);
    b=_abs(b);
    if(!b) return a;
    return _gcd(b,a%b);
}
template< class T > inline T _lcm(T a,T b)
{
    a=_abs(a);
    b=_abs(b);
    return (a/_gcd(a,b))*b;
}

template< class T > inline T _extended(T a,T b,T &x,T &y)
{
    a=_abs(a);
    b=_abs(b);
    T g,x1,y1;
    if(!b)
    {
        x=1;
        y=0;
        g=a;
        return g;
    }
    g=_extended(b,a%b,x1,y1);
    x=y1;
    y=x1-(a/b)*y1;
    return g;
}

template< class T,class X, class Y > inline T _bigmod(T n,X m,Y mod)
{
    ull ret=1, a = n%mod ;
    while(m)
    {
        if(m&1)ret=(ret*a)%mod;
        m>>=1;
        a=(a*a)%mod;
    }
    ret%=mod;
    return (T)ret;
}
template< class T, class Y > inline T _modinv(T n,Y mod)
{
    return _bigmod(n,mod-2,mod);
}
