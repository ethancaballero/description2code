#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main()
{
	int n,t;
	scanf("%d%d",&n,&t);
	int arr[n];
	char arr1[n+10];
	scanf("%s",arr1);
	for(int i=0;i<n;i++)
	{
		arr[i] = arr1[i] - '0'; // converting char into integer
	}

	int mat[10][n];

	for(int i=0;i<10;i++)
	for(int j=0;j<n;j++)
	mat[i][j]=0;

	for(int i=0;i<10;i++)
	{
		for(int j=0;j<n;j++)
		{
			if(j==0)
			{
				if(arr[j]==i)
				mat[i][j] = 1;
			}
			else
			{
				if(arr[j]==i)
				mat[i][j] = mat[i][j-1] + 1;
				else
				mat[i][j] = mat[i][j-1] ;
			}
		}
	}	
	while(t--)
	{
		int v;
		scanf("%d",&v);
		int pos=0,neg=0,ans;
		v--;
		if(v==0)
		ans = 0;
		else
		{
			int elem = arr[v];
			int ind = v-1;
			
			for(int i=0;i<10;i++)
			{
				if(i<elem)
				pos += (elem - i) * mat[i][ind] ;
				else
				neg += (i - elem) * mat[i][ind] ;			
			}
			ans = pos + neg ;
		}
		printf("%d
",ans);
	}
	
	return 0;
}
