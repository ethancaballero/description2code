#include<bits/stdc++.h>
using namespace std;
#define ll int
int temp[100002][10];
void precompute(int *a,int n)
{
	int i,j;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=10;j++)
			temp[i][j]=temp[i-1][j];
		temp[i][a[i]+1]=temp[i-1][a[i]+1]+1;
		//cout<<temp[a[i]][i]<<endl;
	}
}
int main()
{
	char c;
	int n,q,i,x,j;
	cin>>n>>q;
	int a[n+1];
	for(i=1;i<=n;i++)
	{
		cin>>c;
		a[i]=c-48;
	}
	/*for(i=1;i<=n;printf("%d ",a[i++]));
	cout<<"
";*/
	precompute(a,n);
	/*for(i=0;i<=n;i++){
		for(j=0;j<=10;cout<<temp[i][j++]<<" ");
			cout<<"
";
		}*/
	while(q--)
	{
		cin>>x;	
		int count1=0,sum1=0,count2=0,sum2=0;
		for(i=1;i<=a[x]+1;i++)
		{
			count1+=temp[x][i];
			sum1+=(i-1)*temp[x][i];
		}
		count1--;
		sum1-=a[x];
		for(i=a[x]+2;i<=10;i++)
		{
			count2+=temp[x][i];
			sum2+=temp[x][i]*(i-1);
		}
		//printf("%d %d %d %d ",count1, count2, sum1,sum2);
		int res=(sum2-sum1)+(count1-count2)*a[x];
		cout<<res<<endl;
	}
	return 0;
}
