#include<bits/stdc++.h>
#define gc getchar
using namespace std;
typedef long long ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    ll n,m,i,b1,b2,j,e,p,q;
    n=read_int();
    m=read_int();
    char s[n+1];
    cin>>s;
    ll a[10][n+1];
    for(i=0;i<10;i++)
    {
        a[i][0]=0;
    }
    memset(a,0,sizeof(a));
    for(i=0;i<=9;i++)
    {
        for(j=1;j<=n;j++)
        {
            a[i][j]=a[i][j-1];
            ll e=s[j-1]-48;
            if(e==i)
                a[i][j]++;
        }
    }
    /*for(i=0;i<10;i++)
    {
        for(j=0;j<n;j++)
            cout<<a[i][j]<<"  ";
        cout<<'
';
    }*/
    for(i=1;i<=m;i++)
    {
        p=read_int();
        q=s[p-1]-48;
        b1=0;
        for(j=0;j<q;j++)
        {
            b1+=a[j][p]*(q-j);
        }
        b2=0;
        for(j=q+1;j<=9;j++)
        {
            b2+=a[j][p]*(q-j);
        }
        cout<<b1-b2<<'
';
    }
    return 0;
}
