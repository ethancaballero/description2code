#include<iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
int main()
{   int n,m,x,y,i,j,k,b1,b2;
    char str[100002];
   cin>>n>>m;
    int a[n+1];
    int b[10][n+1];
   //getchar();// remove 

   scanf("%s",str);
   
   for(i=1;i<=n;i++)
     a[i]=(str[i-1]-'0');
   //cout<<getchar()-'0'<<endl;
   for(i=0;i<=9;i++)
   {  b[i][0]=0;
     for(j=1;j<=n;j++)
     {
        b[i][j]=b[i][j-1];
         if(a[j]==i)
            b[i][j]++;
     }
   }
   while(m--)
   {
      cin>>k;// index value
      b1=b2=0;
      for(i=0;i<a[k];i++)// greater than 0
      {
        b1+=b[i][k-1]*(a[k]-i);
//        cout<<(b[i][k-1]*(a[k]-i))<<" ";
      }
   //   cout<<"
";
      for(i=a[k]+1;i<=9;i++)// smaller than 0
      {
        b2+=b[i][k-1]*(a[k]-i);
     //     cout<<(b[i][k-1]*(a[k]-i))<<" ";
      }
//      cout<<"
";
      cout<<(b1-b2)<<endl;
//      cout<<"
";
   }

   return 0;
}
