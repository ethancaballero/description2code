#include<bits/stdc++.h>
#define lli long long int
char str[100005];
using namespace std;
int main()
{
        lli i,j,k,m,n,t,l,arr[10]={0},c,p;
        scanf("%lld%lld",&n,&m);
        scanf(" %s",str);
        l=strlen(str);
        lli b1[l],b2[l];

        for(i=0;i<l;i++)
        {
                b1[i]=0;
                b2[i]=0;
                c=str[i]-'0';
                arr[c]++;

                for(j=0;j<c;j++)
                        b1[i]+=(arr[j]*(c-j));

                for(j=c+1;j<10;j++)
                        b2[i]-=(arr[j]*(c-j));
        }
        for(i=0;i<m;i++)
        {
                scanf("%lld",&p);
                printf("%lld
",b1[p-1]+b2[p-1]);
        }
}
