
#include <iostream>
#include <map>
#include <cstdio>
using namespace std;
const int sz=200200;
int B[sz],N,K,ans;
map<int,bool> hm;
 
int main()
{
    int f;
 
    scanf("%d%d",&N,&K);
    for(int i=1;i<=N;i++){
        scanf("%d",&f);
        if(!hm[f]){hm[f]=true; ans++; B[i+K-1]=f;}
        hm[B[i]]=false;
    }
 
    printf("%d
",ans);
 
    return 0;
}
