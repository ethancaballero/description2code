#include<bits/stdc++.h>
using namespace std;
#define mod 1000000
#define ll long long
int main()
{
	int i,a,b,c,d,t,n,turn;
	ll temp,cal,mini,maxi,sum1,sum2;
	vector<int> cnum(1000001);
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d%d%d",&n,&a,&b,&c,&d);
		cnum[d]++;
		temp=d;
		mini=d;
		maxi=d;
		for(i=1;i<n;i++)
		{
			cal=(a*temp*temp+b*temp+c)%mod;
			cnum[(int)cal]++;
			temp=cal;
			mini=min(mini,cal);
			maxi=max(maxi,cal);
		}
		sum1=sum2=0;
		turn=1;
		for(i=mini;i<=maxi;i++)
		{
			while(cnum[i])
			{
				if((turn&1))
				{
					sum1+=i;
					turn=0;
				}
				else
				{
					sum2+=i;
					turn=1;
				}
				cnum[i]--;	
			}
		}
		if(sum1-sum2<0)
			printf("%lld
",sum2-sum1);
		else
			printf("%lld
",sum1-sum2);
	}
	return 0;
}
