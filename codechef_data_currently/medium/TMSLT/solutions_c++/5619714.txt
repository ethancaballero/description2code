#include<cstdio>
#include<algorithm>
#define M 1000000
using namespace std;
int y[1000000];
int main()
{
	long long int t,n,a,b,c,d,i,s;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld%lld%lld%lld",&n,&a,&b,&c,&d);
		for(i=0;i<1000000;i++)
		y[i]=0;
		s=0;
		y[d]++;
		s=d;
		for(i=1;i<n;i++)
		{
			y[s = (a * s * s + b * s + c) % 1000000]++;
		}
		s=0;
		int f=0;
		for(i=0;i<1000000;i++)
		{
			if(y[i]>0)
			{
				if(y[i]%2 && f==0)
				{
				f=1;
				s=s+i;
		     	}
				else if(y[i]%2 && f==1)
				{
				f=0;
				s=s-i;
		     	}
			}
		}
		printf("%lld
",abs(s));
		
	}
	return 0;
}
