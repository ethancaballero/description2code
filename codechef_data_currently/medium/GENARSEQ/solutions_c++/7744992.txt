#define Max 5000000

typedef long long LL;
typedef unsigned long long ULL;

using namespace std;

int visit[Max],CS,ara[1010];

int main()
{
    int T,t,a,b,n;
    si(T);
    rep(t,T)
    {
        siii(a,b,n);
        CS++;
        ara[1] = 1;
        visit[1] = CS;
        int tmp = 1*a - 1*b;
        if(tmp > 0)visit[tmp] = CS;

        int cnt = 1,cur = 1,i;
        while(1)
        {
            if(cnt == n)break;
            cur++;
            if(visit[cur] == CS)continue;
            ara[++cnt] = cur;
            for(i=1;i<=cnt;i++)
            {
                int my = ara[cnt] * a - ara[i] * b;
                if(my > 0)visit[my] = CS;

                my = ara[i]*a - ara[cnt]*b;
                if(my > 0)visit[my] = CS;
            }
        }
        rep(i,n)
        {
            if(i>1)printf(" ");
            printf("%d",ara[i]);
        }
        NL;
    }
    return 0;
}
