string cityn[50];
int n,m;
int dist[50][50];
int t,k;
bool flag[50];
int error;


int main()
{
	sf(n);
	for(int i=0;i<n;i++)
		cin>>cityn[i];

	sf(m);
	for(int i=0;i<m;i++)
	{
		string a,b;
		cin>>a;
		cin>>b;

		int f=0;
		int j=0;
		int r,c;
		while(f!=2)
		{
			if(cityn[j]==a)
			{
				f++;
				r=j;
			}

			else if(cityn[j]==b)
			{
				f++;
				c=j;
			}
			j++;
		}

		sf(dist[r][c]);
	}
	scanf("%d",&t);
	while(t--)
	{
		error=0;
		int route[50];
		scanf("%d",&k);
		LL s=0;
		for(int i=0;i<k;i++)
		{
			string a;
			cin>>a;
			int f=0;
			for(int j=0;j<n;j++)
			{
				if(cityn[j]==a)
				{
					if(flag[j])
						error=1;

					flag[j]=1;
					route[i]=j;
					f=1;
					break;
				}
			}
			if(!f)
				error=1;

		}

		if(!error)
		{
			for(int i=1;i<k;i++)
			{
				int r=route[i-1];
				int c=route[i];

				if(r==c)
				{
					error=1;
					break;
				}

				if(dist[r][c])
					s+=dist[r][c];

				else
				{
					error=1;
					break;
				}
			}
		}

		if(error)
			printf("ERROR
");

		else
			printf("%lld
",s);
		mem(flag,false);
	}
	return(0);
}
