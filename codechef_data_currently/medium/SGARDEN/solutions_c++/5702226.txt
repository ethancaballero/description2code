#include <iostream>
#include <cstdio>
#include <bits/stdc++.h>
 
#define gc getchar_unlocked
#define MOD 1000000007
#define ll long long
#define MAXN 100001
#define NPRIME 9592
bool svisit[MAXN];
int primes[NPRIME];
int exp_arr[NPRIME];
ll lcm;
 
using namespace std;
 
inline int read_int() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  int ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
 
inline ll mod(ll x)
{
    if (x < MOD)
        return x;
    else
        return x % MOD;
}
 
inline ll multiply(ll x, ll y)
{
    x = mod(x);
    y = mod(y);
    return mod(x*y);
}
 
inline ll power(ll a, ll b)
{
    ll ans = 1;
    if ((a == 1) || (b == 0))
        return 1;
    if (a == 0)
        return 0;
    if (b == 1)
        return a;
    while(b)
    {
        if(b & 1) {
            // If b is odd, multiply current ans by that
           ans=multiply(ans,a);
        }
        a=multiply(a,a);
        b>>=1;
    }
    return ans;
}
 
void seive()
{
    int i,j,k=0;
    int root = sqrt(MAXN);
    root+=3;
    primes[k++] = 2;
    for(i=3;i<MAXN;i+=2)
    {
        if(!svisit[i])
        {
            primes[k++] = i;
            for(j=i*i;i<root && j<MAXN;j+=(i<<1)) {
                svisit[j] = true;
            }
        }
    }
}
 
inline void init()
{
    for(int i=0; i < NPRIME; i++)
    {
        exp_arr[i] = 0;
    }
    lcm = 1;
}
 
void print(std::vector<int > data)
{
    int temp;
    for (ll i = 0; i < data.size(); ++i)
    {
        temp = data.at(i);
        printf("%d
", temp);
    }
}
 
int main()
{
    int t;
    seive();
    t = read_int();
    while(t-- > 0)
    {
        init();
        int max_index = 0;
        vector<int>v;
        std::vector<int> visited;
        std::vector<int> length;
        int n = read_int();
        v.resize(n+2);
        visited.resize(n+2);
        for(int i=1; i <=n; i++) {
            v[i]=read_int();
            visited[i] = false;
        }
        for (int i= 1; i <= n; i++)
        {
            if (!visited[i]) {
                int j = v[i];
                int count = 1;
                visited[i] = true;
                while(j!=i)
                {
                    visited[j] = true;
                    count++;
                    j = v[j];
                }
                if ((count!=0) && (count!=1))
                    length.push_back(count);
            } else {
                continue;
            }
        }
        // Factorize the lengths
        for(int i=0; i < length.size(); i++)
        {
            int cur_count = length.at(i);
            for(int j=0; cur_count >= primes[j]; j++) {
                int pow_p = 0;
                while((cur_count % primes[j]) == 0) {
                    pow_p++;
                    cur_count = cur_count / primes[j];
                }
                if(exp_arr[j] < pow_p)
                    exp_arr[j] = pow_p;
                if (max_index < j)
                    max_index = j;
            }
        }
        for(int i=0; i <= max_index; i++)
        {
            if(exp_arr[i]!=0) {
                lcm = multiply(lcm, power(primes[i], exp_arr[i]));
                exp_arr[i] = 0;
            } else {
                continue;
            }
        }
        printf("%lld
", lcm);
    }
    return 0;
}
