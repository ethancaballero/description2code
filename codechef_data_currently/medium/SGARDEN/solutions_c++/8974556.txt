ll modpow(ll a, ll b)// a^b % MOD
{
  a %= MOD;
  ll result = 1;
  while( b > 0)
  {
    if(b & 1)
      result = (result*a)%MOD;
    a = (a * a)%MOD;
    b/=2;
  }
  return result%MOD;
}
