#include<iostream>
#include<cstdio>
#include<math.h>
#include<string.h>
using namespace std;

int scanint(){
	register int c = getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	return x;
}

int main(){
	int t,n,sg[10001],temp[10001];
	sg[0] = sg[1] = 0 ;
	for(int i=2; i<=10000 ; i++){
		memset(temp,0,10001*sizeof(int));
		for(int x=0,y=i-2;x<=y;x++,y--)
			temp[sg[x] ^ sg[y]] = 1;
		for(int j=0;j<=i;j++){
			if(!temp[j]){
				sg[i] = j;
				break;
			}
		}
	}
	
	t = scanint();
	while(t--){
		n = scanint();
		if(sg[n])
			cout<<"Arjuna
";
		else
			cout<<"Bhima
";
	}
}
