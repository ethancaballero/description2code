int g[10010];

void pre() {
	
	g[0] = g[1] = 0; // losing positions
	for(int n = 2; n < 10010; n++) {
		bool vis[10010] = {0};
		for(int x = 0, y = n-2; x <= y; x++, y--) {
			vis[g[x]^g[y]] = 1;
		}
		int t = 0;
		while(vis[t]) t++;
		g[n] = t;
	}
	
}

int main() {
	
	
	pre();
	
	int nt; scanf("%d", &nt); while(nt--) {
		int n; scanf("%d", &n);

	}

}
