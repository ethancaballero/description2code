#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#define size 1000
#define ull long long
ull min(ull a,ull b)
{
    return (a>b?b:a);
}
int main()
{
    ull t,s,k,m,n,i,low,j;
    double p,q,ans;
    double comb[size][size];
    for(i=0;i<size;i++)
    for(j=0;j<size;j++)
    comb[i][j]=0.0;
    for(i=0;i<size;i++)
    {
                       comb[i][i]=1.0;
                       comb[i][0]=1.0;
                       for(j=1;j<i;j++)
                       {
                                          comb[i][j]=comb[i-1][j-1]+comb[i-1][j];
                       }
    }
   /* for(i=0;i<size;i++)
    for(j=0;j<size;j++)
    printf("%.1f ",comb[i][j]);
    printf("
");*/
    //printf("%f
",comb[9][9]);
    scanf("%lld",&t);
    for(i=0;i<t;i++)
    {
                    scanf("%lld %lld %lld %lld",&s,&n,&m,&k);
                    low=min(n-1,m-1);
                    p=0;
                    for(j=k;j<=low;j++)
                    {
                                      p+=comb[m-1][j]*comb[s-m][n-1-j];
                    }
                    q=comb[s-1][n-1];
                    if(q==0)
                    ans=0;
                    else
                    ans=p/q;
                    printf("%f
",ans);
    }
    return 0;
}
