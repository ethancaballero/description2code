int mini[1000010]={0};
int main()
{
	int n,k,x;
	vector<int> vec;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++){
		scanf("%d",&x);
		vec.push_back(x);
		mini[vec[i]]=INT_MAX;
	}
	int val;
	for(int i=0;i<n;i++)
	{
		val=min(i+1,n-i);
		if(mini[vec[i]]>val)
			mini[vec[i]]=val;
	}
	int mi=INT_MAX,temp,flag=0;
	for(int i=0;i<n;i++)
	{
		val=k-vec[i];
		if(val>0){
			if(vec[i]!=val)
			{
			
				if(mini[val]!=0)
				{
					flag=1;
					temp=max(mini[vec[i]],mini[val]);
					if(mi>temp)
						mi=temp;
				}
			}
		}
	}
	if(flag)
		printf("%d
",mi);
	else
		printf("-1
");
	
	
	return 0;
}
