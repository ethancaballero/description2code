int getmoves1(char *str,int a,int b)
{
 
int i,j;
int count=0;
 
i=a;
 
// first half ......
if( str[i++] != 'l' )
count++;
 
if( str[i++] != 'u' )
count++;
 
if( str[i++] != 'c' )
count++;
 
if( str[i++] != 'k' )
count++;
 
if( str[i++] != 'y' )
count++;
 
// second half .......
j=b;
 
if( str[j--]!='l' )
count++;
 
if( str[j--]!='u' )
count++;
 
if( str[j--]!='c' )
count++;
 
if( str[j--]!='k' )
count++;
 
if( str[j]!='y'  &&  j!=(i-1) )
count++;
 
 
 
// rest of the string .......
i=0;
j=len-1;
    while(  i < j )
    {
        if( i==a )
        {
          i=i+5;
          j=j-5;
          continue;
        }
 
      if( str[i]!=str[j] )
      count++;
 
      i++;
      j--;
    }
 
return count;
}
