#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
	char str[33];
	scanf("%s", str);
	while(str[0] != '~')
	{
		bool flag;
		int answer = 0, count;
		while(str[0] != '#')
		{
			count = strlen(str);
			if(count < 3)
			{
				flag = 2 - count;
			}
			else
			{
				count -= 2;
				while(count--)
				{
					answer <<= 1;
					answer += flag;
				}
			}
			scanf("%s", str);
		}
		scanf("%s", str);
		printf("%d
", answer);
	}
    return 0;
}
