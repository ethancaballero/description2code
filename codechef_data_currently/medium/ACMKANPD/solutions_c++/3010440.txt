#include<stdio.h>
#include<math.h>

int duzina,i,cifra;
char c;
int temp[40];

unsigned int binstrtodec(int length){
  unsigned int br = 0;
  unsigned int dva = 1;
  for (i=length; i>0; i--) {
    if (temp[i]==1) {br += dva;}
    dva *= 2;
  }
  return(br);
}

int main(){
  duzina = 0;
  while (true) {
    scanf("%c", &c);
    int brojac = 0;
    while (c == '0') {
      brojac++;
      scanf("%c", &c);
    }
    if (brojac>0){
      if (brojac<3){
         cifra = (brojac==1)? 1 : 0;
         brojac = 0;
      }
      else {
         for (i=1; i<=brojac-2; i++) {duzina++; temp[duzina] = cifra;}
         brojac = 0;
      }
    }
    if (c=='~'){return(0);}
    if (c=='#'){
       printf("%d
", binstrtodec(duzina));
       duzina = 0;
    }
  }
}
