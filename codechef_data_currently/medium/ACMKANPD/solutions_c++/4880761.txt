#include <cstdio>
using namespace std;
#define nmax 35

char ch, str[nmax];
int ans, p, c, f;

int main(){
	while(true){
		p=0;
		c=0;
		while(ch=getchar()){
			if(ch==' ' && c==0) continue;
			else if(ch=='#') break;
			else if(ch=='~') return 0;
			else if(ch=='"'){
				while(ch!='>') ch=getchar();
				continue;
			}
			else if(ch=='0') c++;
			else{
				if(c==1) f=1;
				else if(c==2) f=0;
				else if(c>2){
					c-=2;
					while(c--) str[p++]=(char)(f+'0');
				}
				c=0;
			}
		}
		ans=0;
		for(int i=0;i<p;i++) ans=(ans<<1)+str[i]-'0';
		printf("%d
",ans);
	}
}
