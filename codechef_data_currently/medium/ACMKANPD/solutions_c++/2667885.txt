#include<stdio.h>
#include<string.h>
#include<vector>
using namespace std;
int main()
{
	char s[36];
	while(scanf("%s",&s)==1 && s[0]!='~')
	{
		vector<int> A;
		int flag=0,len;
		do{
		
		len=strlen(s);
		if(len==1)
		flag=1;
		else if(len==2)
		flag=0;
		else
		for(int i=0;i<len-2;i++)A.push_back(flag);
		}while(scanf("%s",&s)==1 && s[0]!='#');	
		unsigned int x=0;
		for(int i=A.size()-1,base=1;i>=0;i--,base<<=1)
		{
			x+=A[i]*base;
		}
		printf("%u
",x);
	}
}
