#include <iostream>
#include <string>
using namespace std;
string cmd;
int main() {
	int now=0, flag=0;
	while(1){
		cin>>cmd;
		if(cmd=="#"){
			cout<<now<<"
";
			now=0;
		}else if(cmd=="~"){
			break;
		}else{
			int sz=cmd.size();
			if(sz==1) flag=1;
			else if(sz==2) flag=0;
			else{
				sz-=2; now<<=sz;
				if(flag) now+=(flag<<sz)-1;
			}
		}
	}
	return 0;
}
