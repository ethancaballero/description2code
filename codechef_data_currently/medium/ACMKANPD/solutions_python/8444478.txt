#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;
int main(){
	
	string s;
	int index, flag, number;
	while( 1 ){
		index = 0;
		number = 0;
		cin >> s;
		if( s == "~" )
			return 0;
		flag = 0;
		while( s != "#") {
			if( s == "0" )
				flag = 1;
			else if( s == "00" )
				flag = 0;
			else{
				int l = s.length() -2;
				while( l-- ){
					number = number*2 + flag;
				}
			}
			cin >> s;
			if( s == "~" )
				return 0;
		}

		printf("%d
", number);
	}
	return 0;
}
