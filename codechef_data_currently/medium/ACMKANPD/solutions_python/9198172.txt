#include<stdio.h>
#include<string.h>
 
int main()
{
	char block[35];
	unsigned int num;
	int flag,length;
	scanf("%s",block);
	while(block[0]!='~')
	{
		num=0;
		while(block[0]!='#')
		{
			length=strlen(block);
			if(length==1)
				flag=1;
			else if(length==2)
				flag=0;
			else
			{
				length-=2;
				if(flag==0)
					num<<=length;
				else
					num=(num<<length)|((1<<length)-1);
			}
			scanf("%s",block);
		}
		printf("%d
",num);
		scanf("%s",block);
	}
	return 0;
}
