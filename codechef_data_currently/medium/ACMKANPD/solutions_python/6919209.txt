#include<stdio.h>
#include<string.h>
 
int main()
{
	while(1)
	{
		int n=0;
		short flag;		
		char c[100];
 
		while(1)
		{
			scanf("%s",c);
 
			if(c[0]=='~')
				return 0;
			else if(c[0]=='#')
				break;
		
			short times=strlen(c);
 
			if(times==1)
				flag=1;
			else if(times==2)
				flag=0;
			else{
				times-=2;
				n<<=times;
				if(flag)
					n|=((1<<times)-1);
			}
		}
		printf("%d
",n);
	}
	return 0;
}
