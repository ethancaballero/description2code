#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
int captain[100001],rank[100001];
int find(int x)
{
	if(captain[x] == x)
		return x;
	captain[x] = find(captain[x]);
	return captain[x];
}

void Union(int a,int b)
{
	int ca = find(a);
	int cb = find(b);
	if(ca!=cb)
	{
		if(rank[ca]>rank[cb])
		{
			captain[cb] = ca;
			rank[ca] += rank[cb];
		}
		else 
		{
			captain[ca] = cb;
			rank[cb] += rank[ca];
		}
	}
}

int main()
{
	int t,m,n,i,j;
	long long int ways;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&m);
		for(i=1;i<=n;i++)
		{
			captain[i] = i;
			rank[i] = 1;
		}
		while(m--)
		{
			scanf("%d%d",&i,&j);
				Union(i,j);
		}
		int ctr = 0;
		ways = 1;
		for(i=1;i<=n;i++)
		{
			if(captain[i] == i)
			{
				ctr+=1;
				ways = (ways * rank[i])%MOD;
			}
		}
		printf("%d %lld
",ctr,ways);
	}
	return 0;
}
