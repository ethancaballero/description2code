#include<stdio.h>
#include<math.h>
 int prime[5000003];
 int res[5000003];
int main()
{
	long long int max = 5000000,i,j,n,t;
	for (i = 2; i <= max / 2; i++)
	{
		prime[i * 2] = 1;
	}
	for (i = 3; i*i <= max; i+=2)
	{
		if (prime[i] == 0)
		{
			for (j = i*i; j <= max; j +=i)
			{
				prime[j] = 1;
				if ((i % 4) == 1 )
				{
					res[j] = 1;
				}
			}
		}
	}
	for (i = 5; i <= max; i += 4)
	{
		if (prime[i] == 0)
		{
			for (j = 1; j < i*i&&i*j<=max; j++)
			{
				res[i*j] = 1;
			}
		}
	}
	res[5] = 1;
	scanf("%lld", &t);
	while (t--)
	{
		scanf("%lld", &n);
		if (res[n] == 1)
			printf("YES
");
		else
			printf("NO
");
	}
	getchar();
	getchar();
	return 0;
}
