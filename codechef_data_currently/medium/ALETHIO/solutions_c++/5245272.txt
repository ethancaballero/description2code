string comp(string a,string b)
{   int i;
	for(i=0;i<a.length();i++)
    {
    	if(a[i]!='0')
    	break;
    }
    if(i==a.length()) a="0";
    else a=a.substr(i,a.length()-i);
    //cout<<a<<"
";
    for(i=0;i<b.length();i++)
    {
    	if(b[i]!='0')
    	break;
    }
    if(i==b.length()) b="0";
    else b=b.substr(i,b.length());
   // cout<<b<<"
";
	int al=a.length(),bl=b.length();
	if(al<bl) return b;
	if(al>bl) return a;
	if(al==bl)
	{
		for(i=0;i<a.length();i++)
		{
			if(a[i]>b[i]) return a;
			if(b[i]>a[i]) return b;
		}
		return a;
	}
}
