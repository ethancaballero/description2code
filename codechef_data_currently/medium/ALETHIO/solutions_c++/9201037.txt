#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
#define sz(_v) (int)(_v).size()
#define all(_v) (_v).begin(),(_v).end()
#define pii pair<int,int>
typedef long long ll;
const int INF = 1000000000;
const ll MOD = 1e9 + 7;

const int maxn = 1000010;

ll fast_expo(ll a,ll b) {
	if(b == 0) return 1;
	ll p;
	if(b % 2 == 0) {
		p = fast_expo(a,b / 2);
		p = (p * p) % MOD;
	}
	else p = (a * fast_expo(a,b - 1)) % MOD;
	return p;
}

int main() {
	ios_base::sync_with_stdio(false);
	
    string s; cin >> s;
	string ans = "";
	bool flag = true;
	
	for (int i = 0; i < sz(s); i ++) {
		if(s[i] >= '0' and s[i] <= '9') continue;
		flag = false;
		
		string temp = "9";
		for (int j = i - 1; j >= 0; j --) {
			if(isalpha(s[j])) break;
			temp += s[j];
		}
		reverse(all(temp));
		for (int j = i + 1; j < sz(s); j ++) {
			if(isalpha(s[j])) break;
			temp += s[j];
		}
		
		string tmp = "";
		int k;
		for (k = 0; temp[k] == '0'; k ++);
		for (int j = k; j < sz(temp); j ++) tmp += temp[j];
		//cout << i << ' ' << temp << ' ' << tmp << ' ' << ans << "
";

		if(sz(tmp) > sz(ans)) ans = tmp;
		else if(sz(ans) == sz(tmp)) {
			for (int j = 0; j < sz(tmp); j ++) {
				if(ans[j] == tmp[j]) continue;
				if(ans[j] < tmp[j]) ans = tmp;
				break;
			}
		}
	}
	
	if(flag) {
		int ii;
		for (ii = 0; s[ii] == '0'; ii ++);
		for (int i = ii; s[i]; i ++) ans += s[i];
	}

	if(ans == "") ans = "0";
	cout << ans << "
";
	return 0;
}
