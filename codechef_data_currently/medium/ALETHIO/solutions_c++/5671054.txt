#include<iostream>
#include<cstring>

char str[1005];
int ans[1005][1005];

using namespace std;
int main(){
	
	
	cin>>str;
	
	int maxlength=0;
	int maxindex=-1;
	
	
	for(int i=0;i<strlen(str);i++){
		int j=i+1;
		int flag=0;
		int sum=0;
		int length=0;
		//cout<<i<<"
";
		
		if(str[i]>='A'&&str[i]<='Z'){
			ans[i][length]=9;
			length++;
			
			while(flag<1&&j<strlen(str)){
				if(str[j]>='0'&&str[j]<='9'){
					ans[i][length]=str[j]-'0';
					length++;
				}
				
				else {
					flag++;
				}
				
				j++;
			}
		}
		
		else{		
			if(str[i]=='0')
			continue;
				
			ans[i][length]=str[i]-'0';
			length++;			
			
			while(flag<2&&j<strlen(str)){
				if(str[j]>='0' && str[j] <='9'){
					ans[i][length]=str[j]-'0';
					length++;
				}
				
				else if(flag<2){
					flag++;
					if(flag<2){
						
						ans[i][length]=9;
						length++;
					}
				}
				j++;
				
				//cout<<sum<<" ";
				
			}
		}
		
		if(length>maxlength){
			maxlength=length;
			maxindex=i;
		}
		
		else if(length==maxlength){
		//	cout<<i<<"
"<<maxindex<<"
";
			for(int k=0;k<length;k++){
				
				if(ans[i][k]>ans[maxindex][k]){					
			//	cout<<i<<" "<<maxindex<<" "<<k<<"
";
					maxindex=i;
					break;
				}
				
				else if(ans[i][k]<ans[maxindex][k]){
					break;
				}
				
				
			}
			
		}
		
//		cout<<"
";
	}
	
//	for(int i=0;i<strlen(str);i++){
//		for(int j=0;j<strlen(str);j++){
//			cout<<ans[i][j]<<" ";
//		}
//		cout<<"
";
//	}
	
	if(maxindex==-1){
		cout<<0<<"
";
	}
	else{
	
		for(int i=0;i<maxlength;i++){
			cout<<ans[maxindex][i];
		}
	}
	cout<<"
";
}
