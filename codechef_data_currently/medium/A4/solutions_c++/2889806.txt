#include <cmath>
#include <iostream>
#define ll long long int
using namespace std;
ll Calc(ll x,ll y,ll mad)
{
	ll tmp=x,ans=1;
	while(y)
	{
		if(y&1)
			ans=(ans*tmp)%mad;
		tmp=(tmp*tmp)%mad;
		y>>=1;
	}
	return ans;
}
int main()
{
	ios_base::sync_with_stdio(0);
	ll test,mad,i,ans,len,kk,n,k;
	long double t,x;
	cin>>test;
	while(test--)
	{
		cin>>n>>k;
		mad=1;
		for (i=1;i<=k;++i)
			mad*=10;
		t=n*1.0*log10l(n*1.0);
		len=t+1;
		ans=Calc(n,n,mad);
		kk=len-k;
		x=pow(10,t-kk);
		cout<<(int)x<<" ";
		for (mad=mad/10;mad>1 && ans<mad;mad=mad/10)
			cout<<"0";
		cout<<ans<<"
";
	}
	return 0;
}
