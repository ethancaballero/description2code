#include <cmath>
#include <iostream>
using namespace std;
int n,k;
inline int Calc(int x,int y,int mad)
{
	int tmp=x,ans=1;
	while(y)
	{
		if(y&1)
			ans=((long long)ans*tmp)%mad;
		tmp=((long long)tmp*tmp)%mad;
		y>>=1;
	}
	return ans;
}
int main()
{
	ios_base::sync_with_stdio(0);
	int test,mad,i,ans;
	long double t,x;
	long long len,kk;
	cin>>test;
	while(test--)
	{
		cin>>n>>k;
		mad=1;
		for (i=1;i<=k;++i)
			mad*=10;
		t=n*1.0*log10l(n*1.0);
		len=t+1;
		ans=Calc(n,n,mad);
		kk=len-k;
		x=pow(10,t-(long double)kk);
		cout<<(int)x<<" ";
		for (mad=mad/10;mad>1 && ans<mad;mad=mad/10)
			cout<<"0";
		cout<<ans<<"
";
	}
	return 0;
}
