inline int rnd(int x,int y,int pos)
{
int tmp=x,lastk=1;
for (;y;y>>=1)
{
if (y&1) lastk=((long long)lastk*tmp)%pos;
tmp=((long long)tmp*tmp)%pos;
}
return lastk;
}
