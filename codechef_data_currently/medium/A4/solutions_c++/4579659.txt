#include<iostream>
#include<cmath>
using namespace std;

   int last(long long n,int k)
    {
    long long int factor,result =1,x;
    x=pow(10,k);
    factor =n%x;
    while(n)
    {
    if(n&1)
    result=(result*factor)%x;
    factor=(factor*factor)%x;
    n>>=1;
    }
    printf("%0*lld
",k,result);
    }
    
   int firstdigits(long long n,int k)
    {
    long double x,y;
    x = n*log10l(n);
    y = powl(10,x-(int)x);
    long long z = y*powl(10,k-1);
    printf("%lld ",z);
    }

int main()
{
  long long n;
    int t,k;
    scanf("%d",&t);
    while(t--)
    {
    scanf("%lld %d",&n,&k);
    firstdigits(n,k);
    last(n,k);
    }
	
}
