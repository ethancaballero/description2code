#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int maxn = 1e5 + 10;
const ll mod = 1e9 + 7;
ll power(ll a,ll b){
	ll ans = 1;
	while(b>0){
		if(b&1) ans = (ans*a);
		a = (a*a);
		b >>= 1;
	}
	return ans;
}
ll modpower(ll a,ll b, ll mod){
	ll ans = 1;
	a %= mod;
	while(b>0){
		if(b&1) ans = (ans*a)%mod;
		a = (a*a)%mod;
		b >>= 1;
	}
	return ans;
}
int main(){
	int t;
	cin>>t;
	ll n,k;
	while(t--){
		cin>>n>>k;
		ll ans1 = modpower(n,n,power(10,k));
		long double x = n*log10l(n);
		ll ans2 = pow(10,x - floor(x) + k -1);
		cout<<ans2<<" ";
		printf("%0*lld
",(int)k,ans1);
	}
	return 0;
}
