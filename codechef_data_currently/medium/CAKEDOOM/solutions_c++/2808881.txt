string solve(Tsingle_test a)
{
	string ss=a.s;
	int n=ss.size();
	int k=a.K;

	char p;

	FOR(r,'0','0'+k)
	{
	string s=ss;

	bool ok=true;

		FOR(i,0,n)
		{
			if (s[i]!='?')
			{
				if ((i>0 && s[i-1]==s[i]) 
				|| (i<n-1 && s[i]==s[i+1])
				|| (i==n-1 && s[i]==s[0] && n!=1)
				|| (s[i]>='0'+k))
				{
					ok=false;
					break;
				}
			}
			else
			{
			p='0';
			   while (p<'0'+k && 
				  ((i>0 && s[i-1]==p) 
				 || (i<n-1 && s[i+1]==p)
		  		 || (i==0 && s[n-1]==p) 
				 || (i==n-1 && s[0]==p))
				 )
		   		   p++;

				   if (p=='0'+k) 
				   {
					   ok=false;
					   break;
				   }
			   s[i]=p;
			}
		}
	if (ok) return s;
	if (ss[0]!='?') 
		return "NO";
	}
	return "NO";
}
