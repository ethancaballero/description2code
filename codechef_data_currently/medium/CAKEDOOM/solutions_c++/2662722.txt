#include<iostream>
using namespace std;

int K;
int length;
string str;
void calculate()
{
	length = str.length();
	if(length == 1)
	{
		if(str[0] == '?')
		{
			str[0] = '0';
			cout<<str<<endl;
			return;
		}
		else if(str[0] - '0'< K)
		{
			cout<<str<<endl;
			return;
		}
		cout<<"NO
";
		return;
	}
	for(int i = 0; i< length; i++)
	{
		if((str[i]!= '?') && (str[i] == str[(i+1)%length]))
		{
			cout<<"NO
";
			return;
		}
	}

	if(K == 1)
	{
		if(length == 1)
		{
			if(str[0] == '?')
			{
				str[0] = '0';
			}
			cout<<str<<endl;
			return;
		}
		else
		{
			cout<<"NO
";
			return;
		}
	}
	else if(K == 2)
	{
		if(length%2 == 1)
		{
			cout<<"NO
";
			return;
		}
		
		int setup_1 = 1; //01010101
		for(int i = 0; i< length; i++)
		{
			if(str[i] != '?')
			{
				if(str[i]!= '0' + i%2)
				{
					setup_1 = 0;
					break;
				}
			}
		}
		if(setup_1 == 1)
		{
			for(int i = 0; i<length; i++)
			{
				str[i] = '0' + i%2;
			}
			cout<<str<<endl;
			return;
		}
		int setup_2 = 1;
		for(int i = 0; i< length; i++)
		{
			if(str[i] != '?')
			{
				if(str[i]!= '0' + 1 - i%2)
				{
					setup_2 = 0;
					break;
				}
			}
		}
		if(setup_2 == 1)
		{
			for(int i = 0; i<length; i++)
			{
				str[i] = '0' + 1 - i%2;
			}
			cout<<str<<endl;
			return;
		}
		if(setup_1 == 0 && setup_2 == 0)
		{
			cout<<"NO
";
			return;
		}
	}
	else
	{
		int left, right;
		for(int i = 0; i<length; i++)
		{
			left = (i - 1)%length;
			if(left < 0)
				left+= length;
			right = (i + 1)%length;
			if(str[i] == '?')
			{
				for(char j = '0'; j<='2'; j++)
				{
					if(str[left]!= j && str[right]!= j)
					{
						str[i] = j;
						break;
					}
				}
			}
		}
	}
	cout<<str<<endl;
}
int main()
{
	int test_cases;
	cin>>test_cases;
	for(int i = 1; i<= test_cases; i++)
	{
		cin>>K;
		cin>>str;
		calculate();
	}
	return 0;
}
