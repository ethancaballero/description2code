using namespace std;
const int MX = 100+10;
char A[MX];
char ans[MX];
bool check2( int val,int len){
    bool ok = true;
    for(int i=1;i<=len && ok;i++){
        if(A[i]=='?' || A[i]=='0'+val){}
        else{
            ok = false;
        }
        ans[i] = '0'+val;
        val = 1-val;
    }
    return ok;
}
int main(){
    //freopen("data.in","r",stdin);
    int T,K,len;
    bool ok;
    scanf("%d",&T);
    while(T--){
        scanf("%d %s",&K,A+1);
        len = strlen(A+1);
        A[0]=A[len];
        A[len+1] = A[1];
        //cout<<"A=[";for(int i=0;i<len+2;i++)cout<<A[i];cout<<"]"<<" len="<<len<<endl;
        ok = true;
        if(K==2){
            if(len%2==0 || len==1){
                ok = check2(0,len);
                if(!ok) ok = check2(1,len);
            }
            else ok = false;
        }
        else for(int i=1;i<=len &&ok;i++){
            if(A[i]!='?' && len>1 && (A[i]==A[i-1] || A[i]==A[i+1])){
                ok = false;
            }
            ans[i] = A[i];
            if(ok &&A[i]=='?'){
                ok = false;
                int v = 0;
                char cv;
                for(;v<K && !ok;v++){
                    cv = '0'+v;
                    ok = cv!= ans[i-1] && cv!=A[i+1];
                    if(ok && i==1 && len>1) ok = cv!=A[len];
                    if(ok && i==len && len>1) ok = cv!=ans[1];
                    if(ok){
                        ans[i] = cv;
                    }
                }
            }
        }
        if(ok){
            for(int i=1;i<=len;i++){
                printf("%c",ans[i]);
            }
            printf("

}
