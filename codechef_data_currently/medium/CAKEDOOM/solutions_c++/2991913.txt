int main()
{
    int t=scan_d();
    char s[102];
    while(t--)
    {
        int k=scan_d();
        scanf("%s",s);
        int len=strlen(s);
        if(k==1&&len>1)
        {
        	printf("NO
");
    		continue;
        }
        bool flag=0;
        for(int i=0;i<len-1;i++)
        {
        	if(s[i]==s[i+1]&&s[i]!='?')
        	{
        		flag=1;
        		break;
        	}
        }
        if(len==1)
    	{
    		if(s[0]=='?')
    			printf("0
");
    		else
    			printf("%s
",s);
    		continue;
    	}
    	if(flag||(s[0]==s[len-1]&&s[0]!='?')||(k==2&&len&1))
    	{
    		printf("NO
");
    		continue;
    	}
    	if(k==2)
    	{
    		bool all=1;
    		for(int i=0;i<len;i++)
    		{
    			if(s[i]!='?')
    				all=0;
    		}
    		if(all)
    		{
    			for(int i=0;i<len;i++)
    				putchar_unlocked('0'+(i&1));
    			putchar_unlocked('
');
    			continue;
    		}
    		int i;
    		for(i=0;i<len;i++)
    		{
    			if(s[i]!='?')
    				break;
    		}
    		bool zero_pos;
    		if(s[i]=='1')
    			zero_pos=!(i&1);
    		else
    			zero_pos=i&1;
    		bool fl=0;
    		for(i=0;i<len;i++)
    		{
    			if((s[i]=='1'&&((i&1)==zero_pos))||(s[i]=='0'&&((i&1)!=zero_pos)))
    			{
    				fl=1;
    				break;
    			}
    		}
    		if(!fl)
    		{
    			string s1;
    			for(int i=0;i<len;i++)
    			{
    				if((i&1)==zero_pos)
    					s1+='0';
    				else
    					s1+='1';
    			}
    			cout<<s1<<endl;
    		}
    		else
    			printf("NO
");
    		continue;
    	}
    	for(int i=0;i<len;i++)
    	{
    		if(s[i]=='?')
    		{
    			if(!i)
    			{
    				for(int j=0;j<k;j++)
    				{
    					if(s[1]!=(char)(j+'0')&&s[len-1]!=(char)(j+'0'))
    					{
    						s[0]=(char)(j+'0');
    						break;
    					}
    				}
    			}
    			else if(i==len-1)
    			{
    				for(int j=0;j<k;j++)
    				{
    					if(s[len-2]!=(char)(j+'0')&&s[0]!=(char)(j+'0'))
    					{
    						s[len-1]=(char)(j+'0');
    						break;
    					}
    				}
    			}
    			else
    			{
    				for(int j=0;j<k;j++)
    				{
    					if(s[i+1]!=(char)(j+'0')&&s[i-1]!=(char)(j+'0'))
    					{
    						s[i]=(char)(j+'0');
    						break;
    					}
    				}
    			}
    		}
    	}
	    printf("%s
",s);
    }
    return 0;
}
