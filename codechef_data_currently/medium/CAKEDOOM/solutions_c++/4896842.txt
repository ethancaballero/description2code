#include<stdio.h>
#include<string.h>
 
int main()
{
 int t,k,len,i,flag;
 char left,right;
 char s[102];
 
 scanf("%d",&t);
 
 while(t--)
 {
  flag=0;
  scanf("%d",&k);
  scanf("%s",s);
  len=strlen(s);
 
  if(len==1)
  {
	 if(s[0]=='?')
	  s[0]='0';
 
	 printf("%s
",s);
  }
  else if(k==1)
	printf("NO
");
 
  else
  {
	 if(k==2 && s[0]=='?')
	 {
	  i=0;
	  while(s[++i]=='?');
	  if((i%2==0 && s[i]=='1')||(i%2==1 && s[i]=='0'))
		 s[0]='1';
	 }
 
	 for(i=0;i<len;i++)
	 {
	  if(i==0)
	  {
		left=s[len-1];
		right=s[1];
	  }
	  else if(i==len-1)
	  {
		left=s[len-2];
		right=s[0];
	  }
	  else
	  {
		left=s[i-1];
		right=s[i+1];
	  }
 
	  if(s[i]!='?' &&( s[i]==left || s[i]==right) ) //Already invalid
	  {
		printf("NO
");
		flag=1;
		break;
	  }
 
	  if(s[i]=='?')
	  {
		if(left!='0' && right!='0')
			s[i]='0';
		else if(left!='1' && right!='1')
			s[i]='1';
		else
		{
			if(k>2)
				s[i]='2';
			else          //Insufficient no. of colors
			{
				printf("NO
");
				flag=1;
				break;
			}
		}
	  }
	 }
	 if(!flag)
	  printf("%s
",s);
  }
 }
return 0;
}
