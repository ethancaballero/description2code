        char a[500];

                if ((a[0]-48)>_kr)

                    cout<<"NO
";goto end;

         }

                {
                    next = previous = (a[(i+1)%l]-48);
                    int j=0;
                    while (j==next)
                    {
                         j++;
                    }
                    if (j>_kr)
                    {
                        cout<<"NO
";
                        goto end;

                    a[i]=(j+48);

                else if (a[i]>47 && a[i]<58)
                    {
                        if (a[i]==a[(i+1)%l] || (a[i]-48)>_kr)
                        {
                        cout<<"NO
";
                        goto end;

                }
                 
            }
         
        }
    else
    {
        if (_kr==1 && a[0]=='?')

                cout<<"NO
";

                {

    for (int i=0;i<l;i++)

        next = a[(i+1)%l]-48;

        if (a[i]=='?')

";
