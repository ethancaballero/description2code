int main(){
    ios_base::sync_with_stdio(0);
    int T;
    cin >> T;
    while (T --> 0){
          int n,k;
          cin >> k;
          string s;
          getline(cin,s);getline(cin,s);
          n = s.size();
          if (n == 1){
             if (s[0] == '?') cout << "0
";
             else
                 if (s[0] >= '0' + k)
                    cout << "NO
";
                 else
                 cout << s << endl;
             continue;
          }
          if (k == 2){
             if (n % 2 == 1) cout << "NO
";
             else{
                  bool zero = true, one = true;
                  for (int i = 0; i < n ;i ++)
                      if (s[i] != '?'){
                         if (i % 2 == 0){
                            if (s[i] == '0') 
                               one = false;
                            else
                                zero = false;
                         }
                         else{
                              if (s[i] == '0')
                                 zero = false;
                              else
                                  one = false;     
                         }
                      }
                  if (!zero && !one)
                     cout << "NO
";
                  else{
                      s[0] = (zero?'0':'1');
                      for (int i = 1; i < n ;i ++)
                          s[i] = '0' + (1 - (s[i - 1] - '0'));
                      cout << s << endl;
                  }
             }
             continue;
          }
          bool fail = false;
          for (int i = 1;i < n ;i ++){
              if (s[i - 1] == s[i] && s[i] != '?')
                 fail = true;
              if (s[i] != '?' && (s[i] < '0' || s[i] >= '0' + k))
                 fail = true;
          }
          if (s[0] == s[n - 1] && s[0] != '?')
             fail = true;
          if (fail){
                    cout << "NO
";
                    continue;
          }
          if (s[0] == '?'){
              char x = '0';
              if (s[1] == '0' || s[n - 1] == '0')
                 x ++;
              if (x == '1' && (s[1] == '1' || s[n - 1] == '1'))
                 x ++;
              if (x - '0' >= k) fail = true;
              s[0] = x;
          }
          s += s[0];
          for (int i = 1;i < n ;i ++){
              if (s[i] != '?') continue;
              char x = '0';
              if (s[i + 1] == '0' || s[i - 1] == '0')
                 x ++;
              if (x == '1' && (s[i + 1] == '1' || s[i - 1] == '1'))
                 x ++;
              if (x - '0' >= k) fail = true;
              s[i] = x;
          }
          for (int i = 1;i < n ;i ++){
              if (s[i - 1] == s[i] && s[i] != '?')
                 fail = true;
          }
          if (s[0] == s[n - 1] && s[0] != '?')
             fail = true;
          if (fail){
                    cout << "NO
";
                    continue;
          }
          for (int i = 0 ;i < n ;i ++)
              cout << s[i];
          cout << endl;
    }
    
    return 0;
}
