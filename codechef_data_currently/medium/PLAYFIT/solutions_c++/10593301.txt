#include <iostream>
#include <cstdio>
#include<limits.h>
#include <algorithm>
#define gc getchar_unlocked
#define pc putchar_unlocked
using namespace std;
inline unsigned long int fastread()
    {
      unsigned long int i=0;
      char c=gc();
      while(c<'0' || c>'9')
        c=gc();
      for(;c>='0' && c<='9';c=gc())
        i=(i<<3)+(i<<1)+c-'0';
      return i;
    }
int main()
{
    int t;
    t=fastread();
    //scanf("%d",&t);
    while(t--)
    {
        int n;
       n=fastread();
      // scanf("%d",&n);
       long int a[n],b[n],c[n];
       int max=0;
       int min=INT_MAX;
        long int mdif=0;
        for(int i=0;i<n;i++){
            a[i]=fastread();
            //scanf("%d",&a[i]);
        }
        for(int i=n-1;i>=0;i--)
            {
                if(a[i]>max)
            max=a[i];
            b[i]=max;
    }
      for(int i=0;i<n;i++){
            if(a[i]<min)
                min=a[i];
            c[i]=min;
             if(b[i]-c[i]>mdif)
                mdif=b[i]-c[i];
            }

        if(mdif>0)
            printf("%li
",mdif);
            else
                printf("UNFIT
");
    }
    return 0;
}
