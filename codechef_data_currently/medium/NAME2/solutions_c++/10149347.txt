#include<iostream>
#include<string.h>
using namespace std;
int main()
{
 int t,i,j,l1,l2,flag=0,c;
 char s[25500],p[25500];
 cin>>t;
 while(t--)
 { j=0;c=0;flag=0;
   cin>>s>>p;
   l1=strlen(s);
   l2=strlen(p);
   if(l1==l2)
   {
   if(strcmp(s,p)==0)
   {
     cout<<"YES
";
     flag=1;
   }
   }
   else
   if(l1<l2)
   {
     for(i=0;i<l2;i++)
     {
       if(p[i]==s[j])
	{
       c++;j++;
	}
	if(j==l1)
	break;
      }
      if(c==l1)
      {
      cout<<"YES
";
      flag=1;
      }
    }
    else
    {
      for(i=0;i<l1;i++)
     {
       if(p[j]==s[i])
       {
       c++;j++;
       }
       if(j==l2)
       break;
      }
    if(c==l2)
    {
    cout<<"YES
";
    flag=1;
    }
   }
   if(flag==0)
   cout<<"NO
" ;
 }
 return 0;
 }
