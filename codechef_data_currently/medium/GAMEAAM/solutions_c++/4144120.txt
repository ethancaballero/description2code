bool f(int a, int b)
{
    assert(a>0 && b>0);
    if(a==b)return false;
    if(a>b)swap(a,b);
    if(b%a==0) return true;
    if(b/a > 1) return true;
    return !f(b%a,a);
}

int grundy(int a, int b)
{
    assert(a>0 && b>0);
    if(a>b)swap(a,b);
    if(!f(a,b))return 0;
    if(b%a==0)return b/a - 1;
    int ret=b/a;
    int x = grundy(b%a,a);
    if(b/a<=x)ret--;
    return ret;
}
