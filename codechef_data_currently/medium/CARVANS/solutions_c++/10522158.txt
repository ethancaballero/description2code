#include<cstdio>
using namespace std;
int main()
{
	int t,n,i,a[10000],count,min;
	scanf("%d",&t); //no of test cases
	while(t--)
	{   
		count=1;
		scanf("%d",&n);  //no of cars
		scanf("%d",&a[0]);
		min=a[0];
		for(i=1;i<n;i++)
		{
			scanf("%d",&a[i]);  //max speed of each car
			if(a[i]<min)
			{
				count++;
				min=a[i];
			}
		}
		printf("%d
",count );

	}
	return 0;
}
