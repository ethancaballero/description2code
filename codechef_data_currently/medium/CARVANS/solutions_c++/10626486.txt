    int readInt(){
    int t=0;
    char c;
    c=getchar_unlocked();
    while(c<'0' || c>'9')
    c=getchar_unlocked();
    while(c>='0' && c<='9'){
    t=(t<<3)+(t<<1)+c-'0';
    c=getchar_unlocked();
    }
    return(t);
    }
