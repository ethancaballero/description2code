#include<stdio.h>
#include<stdlib.h>
int main()
{
	int t,n,i,j;
	scanf("%d",&t);
	while(t--)
	{
		int count=0;
		scanf("%d",&n);
		int a[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		int min=a[0];
		for(i=0;i<n;i++)
		{
			if(a[i]<=min)
			{
				min=a[i];
				count++;
			}
		}
		
		printf("%d
",count);
	}
	return 0;
}
