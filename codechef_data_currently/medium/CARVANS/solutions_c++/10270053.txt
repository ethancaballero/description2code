#include<bits/stdc++.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i,count=1;
        scanf("%d",&n);
        int a[10001];
        scanf("%d",&a[0]);
        int max=a[0];
        for(i=1;i<n;i++)
        {
            scanf("%d",&a[i]);
            if(a[i]<max)
            {
                max=a[i];
                count++;
            }
        }
        printf("%d
",count);
    }
}
