#include <ctime>
#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
#include<cassert>
using namespace std;
int main(){
    int T,n,i,j,k,sumn=0;
    scanf("%d",&T);
    while(T--){
        scanf("%d",&n);
        int pl,pr;
        string ans;
        for(i=0;i<n;i++){
            int l,r;
            scanf("%d%d",&l,&r);
            if(i){
                if(l>pl){
                    while(pl!=pr-1){
                        ans+="L+";
                        pl++;
                        if(pl==l) break;
                    }
                    while(pl!=l){
                        ans+="R+L+";
                        pl++;pr++;
                    }
                    while(pr!=r){
                        if(pr>r)
                            ans+="R-",--pr;
                        else
                            ans+="R+",++pr;
                    }
                }
                else if(l<=pl){
                    while(pl!=l){
                        ans+="L-";
                        pl--;
                    }
                    while(pr!=r){
                        if(pr>r)
                            ans+="R-",--pr;
                        else
                            ans+="R+",++pr;
                    }
                }
            }
            pl=l;pr=r;
        }
        int l=(int)ans.size();
        printf("%d
",l/2);
        printf("%s
",ans.c_str());
    }
    return 0;
}
