#include <bits/stdc++.h>
 
int gcd(int a,int b)
{
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}
 
int mod(char b[],int a,int n)
{
    int i,r;
    r=0;
    for(i=0;i<n;i++)
    {
                    r=(r*10)+(b[i]-48);
                    r=r%a;
    }
    return r;
}

int main()
{
    int t,a,i,m,n,res;

}
