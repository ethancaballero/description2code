#include<iostream>
#include<bits/stdc++.h>
#define F(i,x,y) for(i=x;i<y;i++)
#define S(n) scanf("%lld",&n)
#define s(n) scanf("%d",&n)
#define P(n) printf("%lld
",n)
#define p(n) printf("%d
",n)
#define LLI long long int
#define R return 0
using namespace std;
LLI gcd(LLI a,LLI b)
{
	if(a%b==0)
	return b;
	else
	return(gcd(b,a%b));
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i; s(n);
		LLI a[n],hcf;
		F(i,0,n) S(a[i]);
		sort(a,a+n);
		hcf=a[n-1];
		if(a[0]==1)
		hcf=1;
		else
		{
			for(i=0;i<n-1;i++)
			{
		    	hcf=gcd(a[i],a[i+1]); 
                a[i+1]=hcf;
	    	}
	    }
		P(hcf);
	}
	R;
}
