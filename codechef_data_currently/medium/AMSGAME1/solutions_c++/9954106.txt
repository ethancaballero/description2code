#include <iostream>
#include <cstdio>

using namespace std;

long gcd(long a, long b) 
{
    long rem;

    if(b == 0)
        return a;

    while(true)
    {
        rem = a%b;
        if(rem == 0)
            break;
        a = b;
        b = rem;
    }

    return b;
}

long gcdOfnNumbers(int size, long *array, int p, int q)
{
    if(size == 1)
        return array[p];

    if(size == 2)
        return gcd(array[p], array[q]);

    int n = size/2;
    return gcd(gcdOfnNumbers(n, array, p, p+n-1), gcdOfnNumbers(size-n, array, p+n, q));

}


int main(int argc, char const *argv[])
{
    int t, n, i;
    long result;

    cin>>t;

    while(t--)
    {
        cin>>n;

        long *ptr = new long[n];
        for(i = 0; i < n; ++i)
            cin>>ptr[i];

        result = gcdOfnNumbers(n, ptr, 0, n-1);
        cout<<result<<"
"; 
    }
    
    return 0;
}
