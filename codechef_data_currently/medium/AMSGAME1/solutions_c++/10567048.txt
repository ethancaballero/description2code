#include<stdio.h>
#include<string>
#include<limits.h>
#include<math.h>
#define S(n) scanf("%li",&n);
#include<iostream>
#define s(n) scanf("%d",&n);
#define p(n) printf("%d
",n);
using namespace std;
int gcd(int a, int b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}
int main()
{
   int t;
   s(t);
   while(t--)
   {
       int n;
       s(n)
       long int a[n];
       for(int i=0;i<n;i++)
        S(a[i])
        int g=a[0];
       for(int i=1;i<n;i++)
       {
           g=gcd(g,a[i]);
       }
       p(g)
   }
   return 0;
}
