struct num
{
	int len;

		for(len=0; n>0 ;n/=p)
            it[len++]=n%p;
	}
};

{
	int i,dif=0;
	for(i=0;i<a.len;i++)
	{
		dif+=a.it[i]-(i<b.len?b.it[i]:0)+2*p;
		a.it[i]=dif%p;
		dif=dif/p-2;
	}
	for(;i-->=0 && a.it[i]==0;);
	a.len=i+1;
}


num operator*(num a,num b)
{
	num prod(0);
	memset(prod.it,0,sizeof prod.it);   //its a must dont forget it
	if(b.len==0 || a.len==0)
        return prod;
	int i,j,div;
	for(i=0;i<a.len;i++)
	{
		if(a.it[i]>0)
		{
			div=0;
			for(j=0;j<b.len;j++)
			{
				div+=prod.it[i+j]+a.it[i]*b.it[j];
				prod.it[i+j]=div%p;
				div/=p;
			}
			if(div>0) prod.it[i+j]=div;
		}
	}
	prod.len=a.len+b.len;
	if(prod.it[prod.len-1]==0) prod.len--;
	return prod;
}
