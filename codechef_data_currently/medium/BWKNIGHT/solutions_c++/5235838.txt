        a = b/base;
        b %= base;
        a *= M;
        b *= M;

        b %= base;
        if (M>1&&N>2) {
            b -= 4*(M-1)*(N-2);
            if (b<0) {
                b += base;

            }
        }
        if (M>2&&N>1) {
            b -= 4*(M-2)*(N-1);
            if (b<0) {
                b += base;
                a--;
            }
        }

            printf("%lld",a);
