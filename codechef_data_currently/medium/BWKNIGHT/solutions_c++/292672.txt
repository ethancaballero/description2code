int base=10000000;

vector <int> operator +(const vector <int> & a,const vector <int> & b)
{
	vector <int> res;
	int r,p=0,l1=a.size(),l2=b.size();
	for(int i=0;i<l1||i<l2;i++)
	{
		r=p;
		if (i<l1) r+=a[i];
		if (i<l2) r+=b[i];
		p=r/base;
		res.push_back(r%base);
	}
	if (p>0) res.push_back(p);
	return res;
}

vector <int> operator *(const vector <int> & a,ll b)
{
	vector <int> res;
	ll r,p=0,l1=a.size();
	for(int i=0;i<l1;i++)
	{
		r=p;
		r+=b*a[i];
		p=r/base;
		res.push_back(r%base);
	}
	while(p>0)
	{
		res.push_back(p%base);
		p/=base;
	}
	return res;
}

vector <int> Solve(ll a,ll b)
{
	ll A=8*(a*b-4)+2*(a+b-8)*(2*a*b-12)+4*(2*a*b-8),C=(a-4),D=(b-4),B=(a*b-9);
	vector <int> v,v1;
	v.clear();
	v1.clear();
	while(C>0)
	{
		v.push_back(C%base);
		C/=base;
	}
	while(A>0)
	{
		v1.push_back(A%base);
		A/=base;
	}
	v=v*B;
	v=v*D;
	v=v+v1;
	return v;
}
