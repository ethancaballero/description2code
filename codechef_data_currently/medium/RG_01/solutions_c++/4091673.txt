    typedef struct {
    unsigned short v1, v2;
    unsigned int w;
    } edge_t;

    typedef struct node_t {
    node_t *head;
    node_t *next;
    } node_t;


    node_t *node_head(node_t *node){
        if(node->head == NULL)
        return node;
        

        return node_head(node->head);
    }

    void node_append(node_t *head_last, node_t *tail_first){
        

        node_t *head_head = node_head(head_last);
        tail_first->head = head_head;
        head_last->next = tail_first;
        
    }
    
    bool edge_cmp_max(edge_t t1, edge_t t2)
    {
        return t1.w > t2.w;
    }

    static char *fmt_uint(int x, char *s){
        char b[16], *p = b+16;
        if(x < 100){
        if(x < 10) { s[0] = '0' + x; s[1] = 0; return s+1; }
        else { s[0] = digit_lut[x*2]; s[1] = digit_lut[x*2+1]; s[2] = 0; return s+2; }
        }
        do {
        char *d = digit_lut + (x%100)*2;
        *(--p) = d[1]; *(--p) = d[0];
        x /= 100;
        } while(x >= 10);
        if(x){ *(s++) = '0' + x; }
        while(p != b+16){ *(s++) = *(p++); *(s++) = *(p++); }
        *s = 0;
        return s;
    }


    void mat_print(int A[1000][1000], int n, int m)
    {
        int i, j;
        char buf[20000];
        for(i = 0; i < n; i++){
        char *ptr = buf;
        for(j = 0; j < m; j++){
        if(j != 0){
        *ptr = ' ';
        ptr++;
        }

        ptr = fmt_uint(A[i][j], ptr);
        }
        *ptr = 0;
        printf("%s
", buf);
        }
    }


    void max_score_pairs(int n_nodes, node_t *nodes, int n_edges, edge_t *edges,
    int C[1000][1000])
    {
    int i, j;
    sort(edges, edges+n_edges, edge_cmp_max);
    j = 0;
    for(i = 0; i < n_edges; i++){
    int v1 = edges[i].v1;
    int v2 = edges[i].v2;
    int w = edges[i].w;
    node_t *n1 = &nodes[v1];
    node_t *n2 = &nodes[v2];
    node_t *h1 = node_head(n1);
    node_t *h2 = node_head(n2);
    
    if(h1 == h2)
        continue;
    
    C[v1][v2] = C[v2][v1]=w;
    
    node_t *cur1 = h1;
    
    while(1){
    
        node_t *cur2 = h2;
         int i1 = cur1 - nodes;
        while(cur2){
           
            int i2 = cur2 - nodes;
            int c;
                    c = w;
                    
                    C[i1][i2] = C[i2][i1]=c;
            
            cur2 = cur2->next;
        }
        
        if(cur1->next==NULL){
            node_append(cur1, h2);
            break;
        }
        cur1 = cur1->next;
    }
         
    }
