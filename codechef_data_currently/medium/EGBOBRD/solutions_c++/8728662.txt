int main() {
	
	long long int tc, n, k, po;
	long long int i, j;

	cin >> tc;

	while (tc--) {
		cin >> n >> k;
		long long int eat[n];
		for (i = 0; i < n; i++) {
			cin >> eat[i];
		}


		long long int packets = 0, rem = 0;
		for (i = 0; i < n; i++) {
			if (rem >= eat[i]) {
				rem -= eat[i];
				if (rem > 0) {
					rem--;
				}
				continue;
			} else {
				eat[i] -= rem;
				//rem = 0;
			}
			if (eat[i] % k == 0) {
				packets += (eat[i] / k);
				rem = 0;
			} else {
				po = packets;
				packets += (eat[i] / k) + 1;
				rem = (packets - po) * k - eat[i];
			}
			if (rem > 0) {
				rem--;
			}
			//debbb(eat[i], packets, rem);
		}
		cout << packets << endl;
	}

	return 0;
}
