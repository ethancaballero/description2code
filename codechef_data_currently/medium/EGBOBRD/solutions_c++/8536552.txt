#include<iostream>
#include<cmath>
using namespace std;
long long int dp[100002];
long long int Left[100002];
long long int n, k, i, t, temp;
long long int a[100002];
int main(void)
{
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(i = 0; i < n; i++)
            cin>>a[i];
        if(a[0]%k == 0)
        {
            dp[0] = a[0]/k;
            Left[0] = 0;
        }
        else
        {
            dp[0] = 1 + (a[0]/k);
            Left[0] = k - (a[0]%k);
        }
        for(i = 1; i < n; i++)
        {
            /*if((a[i]%k)==0)
            {
                dp[i] = dp[i-1]+(a[i]/k);
                Left[i] = Left[i-1];
                continue;
            */
            temp = a[i];
            if(Left[i-1]!=0)
            {
                a[i] = a[i]-(Left[i-1]-1);
            }
            if(a[i]<=0)
            {
                dp[i] = dp[i-1];
                Left[i] = (Left[i-1]-1)-temp;
                continue;
            }
            if((a[i]%k) == 0)
            {
                dp[i] = dp[i-1] + (a[i]/k);
                Left[i] = 0;
            }
            else
            {
                dp[i] = dp[i-1] + 1 + (a[i]/k);
                Left[i] = k - (a[i]%k);
            }

        }
        cout<<dp[n-1]<<"
";
    }
}
