ll solve(int st, int end)
{
	if( st >= end)
		return 1; 
	ll &r = ans[st][end]; 
	if(r!=-1)
		return r; 
	ll ans1 =0 ; 
	r = solve(st+1 , end); 
	for(int i = st+1 ; i <= end ; i++)
	{
		if(a[st] == -a[i])
		{
			ans1 =( solve(st+1 , i-1) * solve(i+1 , end) )%MOD;
			r = (r + ans1)%MOD;
		}
	}
	return r; 
}
