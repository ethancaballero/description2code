#include <string.h>

#include <math.h>
#define MOD 1000000007
#define MAX 5000


long long int Bio[MAX][MAX];
long long int ans[MAX];

void ZFunction(char* s, int* z, int N){
	int i;
	int l = 0;
	int r = 0;
	z[0] = N;
	for(i = 1;i<N;i++){
		if(i>r){
			int j = i;
			while(j<N && s[j]==s[j-i]){
				j++;
			}
			z[i] = j-i;
			l = i;
			r = j-1;
		}else{
			if(z[i-l]<r-i+1){
				z[i] = z[i-l];
			}else{
				z[i] = r-i+1;
				int j = r+1;
				while(j<N && s[j]==s[j-i]){
					j++;
					z[i]++;
				}
				l = i;
				r = j-1;
			}

		}

	}
}

void cnt(int* c,int k, int* z, int N){
	int j;
	int d[N+1];
	for(j=0;j<N+1;j++){
		d[j] = 0;
	}
	for(j=k;j<N;j++){
		d[z[j]]++;
	}
	for(j=N-1;j>=0;j--){
		d[j] = d[j] + d[j+1];
	}
	for(j = 0;j<N-k;j++){
		c[d[j+1]-1]++;
	}
}
