int depth[101];
bool visited[101];
int su[101];
list<int> q;
int n;
vector<vector<int> > fr;

void bfs(int a) {
	q.clear();
	q.push_back(a);
	for(int i=1; i<=n; i++) {
		depth[i] = visited[i] = 0;
	}
	visited[a] = 1;
	while(q.size()>0) {
		int x = q.front();
		q.pop_front();
		for(int i=0; i<fr[x].size(); i++) {
			if(!visited[fr[x][i]]) {
				visited[fr[x][i]] = 1;
				depth[fr[x][i]] = depth[x]+1;
				q.push_back(fr[x][i]);
			}
		}
	}
}
