#include<iostream>
#include<stdio.h>
#include<string.h>
#include<vector>
using namespace std;
inline void input(int &x)
{
    int c = getchar_unlocked();
    x = 0;
    int neg = 0;
    
    for(; ((c<48 || c>57) && c != '-'); c = getchar_unlocked());
    
    if(c=='-') {
        neg = 1;
        c = getchar_unlocked();
    }
    
    for(; c>47 && c<58 ; c = getchar_unlocked()) {
        x = (x<<1) + (x<<3) + c - 48;
    }
    
    if(neg)
        x = -x;
}
 
vector<int> v[105];
int visited[105];
int j=1;
void dfs(int i)
{
    visited[i]=j;
    for(int j=0;j<v[i].size();j++)
       if(!visited[v[i][j]])
            dfs(v[i][j]);
}
int main()
{
    int t;
    int n,m,a,b,q;
    input(t);
    while(t--)
    {
        input(n);
        input(m);
        for(int i=0;i<n;i++)
            v[i].clear() , visited[i]=0;
        while(m--)
        {
            input(a);
            input(b);
            v[a].push_back(b);
            v[b].push_back(a);
        }
        dfs(0);
        for(int i=0;i<n;i++)
        {
            if(visited[i]==0)
            {
                j++;
                dfs(i);
            }
        }
        cin>>q;
        while(q--)
        {
            input(a);
            input(b);
            if(visited[b]==visited[a])
                printf("YO
");
            else
                printf("NO
");
        }
    }
    return 0;
}
