#include <iostream>
#include <cmath>

using namespace std;

int main(){
	int t;
	int A[30];
	scanf("%d", &t);
	while(t--){
		bool flag=false;
		int n, m;
		scanf("%d", &n);
		scanf("%d", &m);
		for(int i=0;i<n;i++){
			scanf("%d", &A[i]);
		}
		for(int i=0;i<pow(2,n);i++){
			int sum=0;
			for(int j=0;j<n;j++){
				if(i&(1<<j)){
					sum+=A[j];
				}
			}
			if(sum==m){
				flag=true;
				break;
			}
		}
		if(flag){
			printf("Yes
");
		}
		else
			printf("No
");
	}
	return 0;
}
