#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int maxn = 1e5 + 10;
bool dp[30][30000];
int A[30];
int main(){
	int t;
	cin>>t;
	int n,m;
	while(t--){
		cin>>n>>m;
		for(int i = 1;i <= n;++i)
			cin>>A[i];
		memset(dp,0,sizeof(dp));
		dp[0][0] = true;
		for(int i = 1; i <= n;++i){
			dp[i][0] = true;
			for(int j = 1;j <= m; ++j){
				dp[i][j] = dp[i-1][j];
				if(A[i] <= j){
					dp[i][j]  |= dp[i-1][j-A[i]];
				}
			}
		}
		if(dp[n][m]){
			cout<<"Yes
";
		}else {
			cout<<"No
";
		}
	}
	return 0;
}
