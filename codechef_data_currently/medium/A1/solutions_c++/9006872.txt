#include <bits/stdc++.h>
using namespace std;
int n,m,a[100000];
bool verify(int m)
{
	int posum = 0; int dp[100000]={0},i,j;
	for(i=0;i<n;i++)
		posum+=a[i];
	dp[0]=1;
	for(i=0;i<n;i++)
	{
		for(j=posum;j>=a[i];j--)
			dp[j]|=dp[j-a[i]];
	}
	
	return dp[m]; 
}
int main()
{
	int t,i;
	cin>>t;
	while(t--)
	{
	
		cin>>n>>m;
		for(i=0;i<n;i++)
			cin>>a[i];
			if(verify(m))
			 cout<<"Yes
";
			 else
			 cout<<"No
";
	}
}
