int i;
int partition(int *a, int l, int h) 
{ 
	int p=a[l];
	int left=l+1,right=h;
	while (left<right) 
	{
		while (a[left] <=p&&left<h) left++;
		while (a[right] >=p&&right>l) right--;
		if(left<right) { int temp=a[right];
		a[right] =a[left];
		a[left] =temp;}
	}
	if (p>a[right])
{	a[l]=a[right];
	a[right] =p;} 
	return right;
}
void qsort (int *a, int l, int h) 
{
	if (l>=h)
	return ;
	int p=partition(a, l, h);
	if (p!=0)
	qsort (a, l, p-1);
	qsort (a, p+1, h);
}
