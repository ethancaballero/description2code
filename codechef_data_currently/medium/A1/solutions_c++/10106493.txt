#include<bits/stdc++.h>
using namespace std;
int n,m;
bool sum(int array[],int target,int s,int i){
	
	if(s==target) {
		//cout<<s<<" "<<target<<endl;
		return true;
	}
	if(s>target||i>=n) return false;
	//cout<<s+array[i]<<" "<<array[i]<<" "<<i<<" "<<endl;
	return sum(array,target,s+array[i],i+1)||sum(array,target,s,i+1);
	
}
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
	
		//int n,m;
		scanf("%d%d",&n,&m);
		int i;
		int array[n+10];
		for(i=0;i<n;i++){
			scanf("%d",&array[i]);
		}
		
		bool ans=sum(array,m,0,0);
		if(ans)
		printf("Yes
");
		else printf("No
");
	}
	return 0;
}
