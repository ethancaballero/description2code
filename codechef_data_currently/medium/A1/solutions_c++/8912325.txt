#include<iostream>
using namespace std;

int payup()
{
    int m,n;
    cin >> n >> m;
    int a[n],i;
    for(i = 0;i < n;i++)
        cin >> a[i];

    int flag = 0;
    for(i = 0;i < n;i++)
    {
        if (a[i] == m)
            flag=1;
    }
    int sum,j;
    for(i = 0;i < n;i++)
    {
        sum=a[i];
        for(j = 0;j < n;j++)
        {
            if(j!=i)
            {
                if((sum+a[j]) <= m)
                {
                    sum = sum + a[j];
                    if(sum == m)
                        flag = 1;
                }
            }
            if(flag == 1)
                break;
        }
        if(flag == 1)
            break;
    }
    if(flag == 0)
    {
        for(i = n-1;i > -1;i--)
        {
            sum = a[i];
            for(j = n-1;j > -1;j--)
            {
                if(j != i)
                {
                    if((sum + a[j]) <= m)
                    {
                        sum = sum + a[j];
                        if(sum == m)
                            flag  = 1;
                    }
                }
                if(flag == 1)
                    break;
            }
            if(flag == 1)
                break;
        }
    }
    return flag;
}


int main()
{
    int t,i,ans[101];
    cin >> t;
    for (i=0;i<t;i++) {
        ans[i] = payup();
        if(ans[i] == 1)
            cout << "Yes" << "
";
        else
            cout << "No" << "
";
    }
    return 0;
}
