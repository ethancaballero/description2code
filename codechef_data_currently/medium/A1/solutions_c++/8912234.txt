#include<stdio.h>

int payup();
int main()
{
    int T,i,res[101];
    scanf("%d",&T);
    for (i=0;i<T;i++)
        res[i] = payup();

    for (i=0;i<T;i++)
    {
        if(res[i] == 1)
            printf("Yes
");
        else
            printf("No
");
    }
    return 0;
}

int payup()
{
    int m,n,c[21],i,f=0,j,sum;
    scanf("%d%d",&n,&m);
    for(i=0;i<n;i++)
        scanf("%d",&c[i]);

    for(i=0;i<n;i++)
    {
        if (c[i] == m)
            f=1;
    }
    for(i=0;i<n;i++)
    {
        sum=c[i];
        for(j=0;j<n;j++)
        {
            if(j!=i)
            {
                if((sum+c[j]) <= m)
                {
                    sum = sum+c[j];
                    if(sum == m)
                        f = 1;
                }
            }
            if(f==1)
                break;
        }
        if(f==1)
            break;
    }
    if(f==0)
    {
        for(i=n-1;i>-1;i--)
        {
            sum=c[i];
            for(j=n-1;j>-1;j--)
            {
                if(j!=i)
                {
                    if((sum+c[j]) <= m)
                    {
                        sum = sum+c[j];
                        if(sum == m)
                            f = 1;
                    }
                }
                if(f==1)
                    break;
            }
            if(f==1)
                break;
        }
    }
    return f;
}
