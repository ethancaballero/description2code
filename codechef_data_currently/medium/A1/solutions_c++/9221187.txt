#include<cstdio>
#include<math.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	for(int i=0;i<t;i++)
	{
		int n,m;
		scanf("%d",&n);
		scanf("%d",&m);
		int a[n];
		
		for(int j=0;j<n;j++)
		{
			scanf("%d",&a[j]);
		}
		long long int b=pow(2,n);
		int sum=0;
		for(int j=1;j<b;j++)
		{
			for(int k=0;k<n;k++)
			{
				if((j & (1<<k))!=0)
				  sum=sum+a[k];
			}
			if(sum==m)
			  {
			  	printf("Yes
");
			  	break;
			  }
			else 
			  sum=0;  
		     if(j==b-1)
			{
				printf("No
");
			  }  
		}
	}
	
}
