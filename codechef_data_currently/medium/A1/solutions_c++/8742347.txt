#include <iostream>
using namespace std;
bool cal(int *arr,int n,int amount,int sum){
	if(sum==amount || sum+arr[n]==amount )
		return 1;
	if(n<0 || sum>amount)
		return 0;
	if(cal(arr,n-1,amount,sum+arr[n]) || cal(arr,n-1,amount,sum))
		return 1;
	else
		return 0;
}
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,m;
		cin>>n>>m;
		int arr[n];
		for (int i = 0; i < n; ++i)
		{
			cin>>arr[i];
		}
		(cal(arr,n-1,m,0))?cout<<"Yes
":cout<<"No
";
	}

	return 0;
}
