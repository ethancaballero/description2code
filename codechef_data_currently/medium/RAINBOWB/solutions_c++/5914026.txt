void fact()
{
    f[0]=1;

    return;
}
LL fpow(LL a, LL b, LL mod)
{
    LL p=1;
    while(b>0)
    {
        if(b&1) p=(p*a)%mod;
        a=a*a; a%=mod;
        b/=2;
    }
    return p;
}
