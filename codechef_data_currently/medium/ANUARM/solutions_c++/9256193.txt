#include<iostream>
using namespace std;
int main()
{
    int t,n,m,min,max,x,i;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        min=n-1;
        max=0;
        for(i=0;i<m;i++)
        {
            cin>>x;
            if(x>max)
                max=x;
            if(x<min)
                min=x;
        }
        m=(max+min)/2;
        for(i=0;i<=m;i++)
            cout<<max-i<<" ";
        for(;i<n;i++)
            cout<<i-min<<" ";
        cout<<"
";
    }
    return 0;
}
