#include<iostream>
#include<stdio.h>
#include<math.h>
#include<stdlib.h>
using namespace std;
int main()
{
	long int i,j,t,n,m,min,max,y;
	cin>>t;
	while(t--)
	{
	cin>>n;
	cin>>m;
	cin>>min;
	max=min;
	for(i=0;i<m-1;i++)
	{
		cin>>y;
		if(y>max)
		max=y;
		if(y<min)
		min=y;
	}
	for(i=0;i<n;i++)
	{
	    if(abs(i-min)>abs(i-max))
	    cout<<abs(i-min)<<" ";
	    else
	    cout<<abs(i-max)<<" ";
	}
	//for(i=0;i<n;i++)
	
	cout<<"

}
return 0;
}
