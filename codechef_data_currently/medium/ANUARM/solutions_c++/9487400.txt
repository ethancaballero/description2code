#include<stdio.h>
int main()
{
    int n,m,t,x,i,m1,m2,temp;
    scanf("%d",&t);
    while(t--)
    {
        m1=100005;
        m2=-1;
        scanf("%d%d",&n,&m);
        while(m--)
        {
            scanf("%d",&x);
            if(x>m2)
                m2=x;
            if(x<m1)
                m1=x;
        }
        temp=m1+m2;
        if(temp%2==0)
        {
            temp=temp/2;
            for(i=0;i<=temp;i++)
                printf("%d ",m2--);
            m2++;
            for(i=temp+1;i<n;i++)
                printf("%d ",++m2);
        }
        else
        {
            temp=temp/2;
            for(i=0;i<=temp;i++)
                printf("%d ",m2--);
            for(i=temp+1;i<n;i++)
                printf("%d ",++m2);
        }
        printf("
");
    }
    return 0;
}
