#include<stdio.h>
#include<iostream>
int main()
{
    long long int t,n,m,i,x,mi,ma,y;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld",&n,&m);
        scanf("%lld",&x);
        mi=x;
        ma=x;
        for(i=1;i<m;i++)
            {
                scanf("%lld",&x);
                if(x<mi)
                    mi=x;
                if(x>ma)
                    ma=x;
            }
           for(i=0;i<n;i++)
           {
               x=i-mi;
               y=i-ma;
               if(x<0)
                x=-x;
               if(y<0)
                y=-y;
               if(x<y)
                printf("%lld ",y);
               else
               {
                   printf("%lld ",x);
               }
           }
          printf("
");
    }
    return 0;
}
