#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	long int n,m,i,arr[100001],val,a[100001]={0},min,max;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%ld %ld",&n,&m);
		for(i=0;i<m;i++)
	    scanf("%d",&arr[i]);
	    min=arr[0];
	    max=arr[0];
	    for(i=1;i<m;i++)
	    {
	    	if(arr[i]>max)
	    	max=arr[i];
	    	if(arr[i]<min)
	    	min=arr[i];
		}
		for(i=0;i<n;i++)
		{
		if(abs(i-min)>=abs(i-max))
		{
			a[i]=abs(i-min);
		}
		else
		{
			a[i]=abs(i-max);
		}
		}
		for(i=0;i<n;i++)
		{
			printf("%d",a[i]);
			printf(" ");
		}
		printf("
");
	}
	return 0;
}
