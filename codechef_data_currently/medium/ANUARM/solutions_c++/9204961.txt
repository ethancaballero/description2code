#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main(){
	int t;
	long long int n,i,pos,mpos,lpos,pos1,pos2,m;
	scanf("%d", &t);
	while(t--){
		mpos = -1;
		lpos = 1000000;
		scanf("%lld %lld", &n, &m);
	while(m--){
			scanf("%lld", &pos);
			if(pos<lpos){
				lpos = pos;
			}
			if(mpos<pos){
				mpos = pos;
			}
		}
		for(i=0;i<n;i++){
			pos1 = abs(i-lpos);
			pos2 = abs(mpos-i);
			if(pos1 > pos2){
				printf("%lld ", pos1);
			}else{
				printf("%lld ", pos2);
			}
		}
		printf("
");
	}
	return 0;
}
