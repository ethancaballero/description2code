void scanint(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
int main()
{
    int t;
    scanint(t);
    while(t--)
    {
        int n,m,i,pos;
        scanint(n);
        scanint(m);
        scanint(pos);
        int maximum=pos;
        int minimum=pos;
        for(i=1;i<m;i++)
        {
            scanint(pos);
            if(pos>maximum)
                maximum=pos;
            if(pos<minimum)
                minimum=pos;
        }
        for(i=0;i<n;i++)
        {
            if(abs(maximum-i)>=abs(minimum-i))
                printf("%d ",abs(maximum-i));
            else
                printf("%d ",abs(minimum-i));
        }
        printf("
");
    }
    return 0;
}
