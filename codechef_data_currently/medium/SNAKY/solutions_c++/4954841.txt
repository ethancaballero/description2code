# include<stdio.h>
 
 
 
void output(int n, int m, int x, int y, int l, char *move)
{
	char grid[1000][1000],temp;
	int i,j,row,col,trow,tcol,count=0,tk=0;
 
	for(i=0;i<=m-1;i++)
		for(j=0;j<=n-1;j++)
			grid[i][j]='-';
 
	row = m-y;
	col = x-1;
	trow=row;	tcol=col;
 
	grid[row][col]='S';
	for(i=0;i<=l-2;i++)
	{
		if(move[i]=='U')
			grid[--row][col]='S';
		else if(move[i]=='D')
			grid[++row][col]='S';
		else if(move[i]=='R')
			grid[row][++col]='S';
		else if(move[i]=='L')
			grid[row][--col]='S';
	}
 
	temp=move[i-1];
	l--;
 
 
 
	if(temp=='U')
	{
		while(1)
		{
			if(tk==1000095)
				tk=0;
 
			if(move[tk]=='U')
				grid[trow--][tcol]='-';
			else if(move[tk]=='D')
				grid[trow++][tcol]='-';
			else if(move[tk]=='R')
				grid[trow][tcol++]='-';
			else if(move[tk]=='L')
				grid[trow][tcol--]='-';
			tk++;
 
			if(row==0)
			{
				printf("WALL %d
",count);
				break;
			}
			else if(grid[row-1][col]=='S')
			{
				printf("BODY %d
",count);
				break;
			}
			else
			{
				grid[--row][col]='S';
				count++;
 
				if(l==1000095)
					l=0;
				move[l++]='U';
			}
		}
	}
	else if(temp=='D')
	{
		while(1)
		{
			if(tk==1000095)
				tk=0;
 
			if(move[tk]=='U')
				grid[trow--][tcol]='-';
			else if(move[tk]=='D')
				grid[trow++][tcol]='-';
			else if(move[tk]=='R')
				grid[trow][tcol++]='-';
			else if(move[tk]=='L')
				grid[trow][tcol--]='-';
			tk++;
 
			if(row==m-1)
			{
				printf("WALL %d
",count);
				break;
			}
			else if(grid[row+1][col]=='S')
			{
				printf("BODY %d
",count);
				break;
			}
			else
			{
				grid[++row][col]='S';
				count++;
 
				if(l==1000095)
					l=0;
				move[l++]='D';
			}
		}
	}
	else if(temp=='R')
	{
		while(1)
		{
			if(tk==1000095)
					tk=0;
 
			if(move[tk]=='U')
				grid[trow--][tcol]='-';
			else if(move[tk]=='D')
				grid[trow++][tcol]='-';
			else if(move[tk]=='R')
				grid[trow][tcol++]='-';
			else if(move[tk]=='L')
				grid[trow][tcol--]='-';
			tk++;
 
			if(col==n-1)
			{
				printf("WALL %d
",count);
				break;
			}
			else if(grid[row][col+1]=='S')
			{
				printf("BODY %d
",count);
				break;
			}
			else
			{
				grid[row][++col]='S';
				count++;
 
				if(l==1000095)
					l=0;
				move[l++]='R';
			}
		}
	}
	else if(temp=='L')
	{
		while(1)
		{
			if(tk==1000095)
				tk=0;
 
			if(move[tk]=='U')
				grid[trow--][tcol]='-';
			else if(move[tk]=='D')
				grid[trow++][tcol]='-';
			else if(move[tk]=='R')
				grid[trow][tcol++]='-';
			else if(move[tk]=='L')
				grid[trow][tcol--]='-';
			tk++;
 
			if(col==0)
			{
				printf("WALL %d
",count);
				break;
			}
			else if(grid[row][col-1]=='S')
			{
				printf("BODY %d
",count);
				break;
			}
			else
			{
				grid[row][--col]='S';
				count++;
 
				if(l==1000095)
					l=0;
				move[l++]='L';
			}
		}
	}
}
int main()
{
	int t, n,m, x,y, l ;
	int i;
 
	scanf("%d",&t);
 
	for(i=0;i<=t-1;i++)
	{
		char move[1000100];
		scanf("%d %d %d",&n,&m,&x);
		scanf("%d %d", &y, &l);
		scanf("%s",move);
		
		output(n,m,x,y,l,move);
 
	}
	return 0;
}
