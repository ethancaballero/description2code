#include<stdio.h>
#include<iostream>
#include<string>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        string s;
        cin>>s;
        int max=0;
        int c=0;
      for(int i=0;i<s.size();i++)
      {
          if(s[i]=='.')
          {
              int lmax=0;
              while(s[i]=='.')
              {
                lmax++;
                i++;
              }
              if(lmax>max){
                max=lmax;
                c++;
              }
          }
      }
      printf("%d
",c);
    }
    return 0;
}
