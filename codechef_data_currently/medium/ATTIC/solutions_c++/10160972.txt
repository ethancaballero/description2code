#include<iostream>
#include<string.h>
using namespace std;
int main()
{
 long int t,max=0,i,j,k;
 char c[5000000];
 cin>>t;
 while(t--)
 {
   cin>>c;
   k=0;    max=0;
   j=0;
   for(i=0;i<strlen(c);i++)
   {
    j=0;
    if(c[i]=='.')
     while(c[i]!='#')
     {
       j++;
       i++;
       }
       if(j>max)
       {
       max=j;
       k++;
       }

    }
    cout<<k<<"
";
   }
   return 0;
   }
