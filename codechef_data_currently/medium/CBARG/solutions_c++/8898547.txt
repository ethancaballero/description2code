#include <iostream>
#include <vector>
using namespace std;

int main() {
	int tc, t, n, i;
	vector<int> v;
	
	cin >> tc;
	
	while (tc--) {
		cin >> n;
		v.resize(n);
		
		cin >> v[0];
		long long int total = v[0];
		
		for (i = 1; i < n; i++) {
			cin >> v[i];
			if (v[i] > v[i - 1]) {
				total += v[i] - v[i - 1];
			}
		}
		cout << total << endl;
	}
	
	
	return 0;
}
