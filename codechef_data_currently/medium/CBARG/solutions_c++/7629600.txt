#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int main()
{
    int t,n,i,j,l=0,m;
    long long int k;
    sd(t);
    while(t--)
    {
        sd(n);
        sd(i);
        k=i;
        for(m=1;m<n;m++)
        {
            sd(j);
            k+=max(l,j-i);
            i=j;
        }
        printf("%lld
",k);
    }
    return 0;
}
