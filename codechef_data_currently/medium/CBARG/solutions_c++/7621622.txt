#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		ll a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		ll ans=0;
		ll cur=0;
		for(int i=0;i<n;i++)
		{
			if(cur<a[i])
			{
				ans+=a[i]-cur;
			}
			cur=a[i];
		}
		cout<<ans<<"
";
	}
	return 0;
}
