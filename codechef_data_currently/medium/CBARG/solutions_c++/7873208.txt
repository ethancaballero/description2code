#include<bits/stdc++.h>

using namespace std;

typedef long long int lli;

int main(void)
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		lli arr[n],present=0,total=0;
		for(int i=0;i<n;i++)
		  scanf("%lld",&arr[i]);
		
		for(int i=0;i<n;i++)
		{
			if(arr[i]>present)
			{
				total+=arr[i]-present;
			}
			present=arr[i];
		}
		printf("%lld
",total);
	}
	return 0;
}
