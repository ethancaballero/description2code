#include<iostream>
#include<stdio.h>
using namespace std;

typedef long long ll;

int main()
{
    #ifndef ONLINE_JUDGE
        freopen("CBARG_input.txt","r",stdin);
    #endif // ONLINE_JUDGE

    ll t,n,m,prev,sum;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        prev=0;
        sum=0;
        while(n--)
        {
            scanf("%lld",&m);
            if(m>prev)
                sum+=m-prev;
            prev=m;
        }
        printf("%lld
",sum);
    }
}
