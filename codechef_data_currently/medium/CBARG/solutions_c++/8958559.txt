#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int alloc,cur,p;
		int n,i,j;
		cin>>n;
		alloc = 0;
		cur = 0;
		for(i = 0;i<n;i++)
		{
			cin>>p;
			if(cur<p)
			{
				alloc += p-cur;
				cur = p;
			}
			else
			{
				cur = p;
			}
		}
		cout<<alloc<<"
";
	}
	return 0;
}
