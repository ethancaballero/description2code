#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		char s[105];
		scanf("%s %d",s,&n);
		int ans=0;
		int x,y=0,sp=0;
		for(;sp<12*n;sp++)
		{
			y=sp;
			while(true)
			{
				x=y;
				for(int i=0;i<strlen(s);i++)
					x+=s[i]-'R';
				if(x>=12*n)
					break;
				y=x;
				ans++;
			}
		}
		
		printf("%d
",ans);
	}
	return 0;
}
