int t, n, l, scaleLen;
char s[101];

int main(__attribute__((unused)) int argc, __attribute__((unused)) char **argv) {
	scanf("%d", &t);
	while(t--) {
		scanf("%s %d", s, &n);
		l = strlen(s);

		scaleLen = 0;
		for(int i = 0; i < l; i++) {
			if(s[i] == 'T') {
				scaleLen += 2;
			} else {
				scaleLen++;
			}
		}

		int ans = 0;
		for(int i = 0; i < 12 * n; i++) {
			ans += (12 * n - i - 1) / scaleLen;
		}

		printf("%d
", ans);
	}

	return 0;
}
