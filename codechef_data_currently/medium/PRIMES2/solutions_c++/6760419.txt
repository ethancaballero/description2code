#include <bits/stdc++.h>
using namespace std;
#define m 1000000
 
bool prime[m+1];

int main() {
	for(int i=0;i<m;i++)
	{
		prime[i]=true;
	}
	for(int i=2;i*i<m;i++)
	{
		if(prime[i]==true)
		{
			for(int j=i*i;j<m;j+=i)
			{
				prime[j]=false;
			}
		}
	}
	int n;
	while(scanf("%d",&n) && n>0)
	{
		int flag=0;
		for(int i=2;i<=100&&!flag;i++)
		{
			if(i>0 && (i*i*i)<n && prime[i])
			{
			for(int j=2;j<=1000&&!flag;j++)
			{
				if(j>0 && ((i*i*i)+(j*j))<n && prime[j])
				{
				int x=n-(i*i*i)-(j*j);
				if(x>0 && prime[x] && flag==0)
				{
					printf("%d %d %d
",x,j,i);
					flag=1;
					break;
				}
				}
			}
			if(flag==1)
			  break;
			}  
		}
		if(flag==0)
		  cout<<"0 0 0"<<endl;
	}
	return 0;
}
