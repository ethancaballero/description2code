#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <ctype.h>
#include <map>
#include <cmath>
#include <cstring>
using namespace std;
 
int main() {int n,q,a[302][302],c[302][302][10],x1,x2,y1,y2,ans;
            scanf("%d",&n);
            for(int i=1;i<=n;i++)
             for(int j=1;j<=n;j++)
              scanf("%d",&a[i][j]);
            for(int i=1;i<=n;i++)
             for(int j=1;j<=n;j++)
              for(int k=0;k<10;k++)
               c[i][j][k]=0;
            for(int i=1;i<=n;i++)
             for(int j=1;j<=n;j++)
               {c[i][j][a[i][j]-1]++;
               for(int k=0;k<10;k++)         
                c[i][j][k]+=c[i][j-1][k]+c[i-1][j][k]-c[i-1][j-1][k];}
            scanf("%d",&q);
            while(q--)
            {   ans=0;
                scanf("%d %d %d %d",&x1,&y1,&x2,&y2);
                for(int i=0;i<10;i++)
                {if(c[x2][y2][i]+c[x1-1][y1-1][i]-c[x1-1][y2][i]-c[x2][y1-1][i]>0)
                   ans++;}
                 printf("%d
",ans);
            }
                
	
	return 0;
}
