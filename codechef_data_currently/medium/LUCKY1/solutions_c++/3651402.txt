#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <map>
#include <algorithm>
#define MOD 1000000007
typedef long long ll;
using namespace std;
ll work(ll n)
{
	ll d,c4=0,c7=0;
	while(n>0)
	{
		d=n%10;
		n=n/10;
		if(d==7) c7++;
	 	if(d==4) c4++;
	}
	return (c4-c7);
}
map<ll,ll> ma;
ll a[100005];
int main()
{
	ll test,m,n,i,d;
	ll ans;
	scanf("%lld",&test);
	ma.clear();
	d=0;
	ans=0;
	for(i=0;i<=100000;i++)
	{
		d+=work(i);
		if(ma.find(d)!=ma.end())
		{
			ans+=ma[d];
			ma[d]++;
		}	
		else
		{
			ma[d]=1;
		}
		a[i]=ans;
	}
	while(test--)
	{
			scanf("%lld",&n);
		printf("%lld
",a[n]);
	}
	return 0;
}
