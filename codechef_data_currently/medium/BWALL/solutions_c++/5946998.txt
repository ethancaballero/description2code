#include<iostream>
using namespace std;
#define mod 1000000007
void matmul(long long int a[][3],long long int b[][3],long long int c[][3])
{
    int i,j,k;
    for(i=0;i<3;i++)
        for(j=0;j<3;j++)
        {
            c[i][j]=0;
            for(k=0;k<3;k++)
                    {
                        c[i][j]+=a[i][k]%mod*b[k][j]%mod;
                        c[i][j]=c[i][j]%mod;
                    }
        }
}
void matcpy(long long int a[][3],long long int b[][3])
{
    int i,j;
    for(i=0;i<3;i++)
        for(j=0;j<3;j++)
        a[i][j]=b[i][j];
}

long long int solve (int n)
{
    int i;
    long long int a[3][3]={{5,1,1},{0,0,0},{0,0,0}},b[3][3]={{1,1,0},{4,0,1},{2,0,0}},c[3][3]={0},d[3][3];
    for(i=0;i<3;i++)
        c[i][i]=1;
    //matcpy(c,b);
    while(n)
    {
        if(n%2)
        {
            matmul(c,b,d);
            matcpy(c,d);
        }
        matmul(b,b,d);
        matcpy(b,d);
        n=n/2;
    }
    matmul(a,c,d);
    return d[0][0]%mod;
}
