void matmult(long long int a[][3],long long int b[][3],long long int c[][3])
{
    int i,j,k;
    for(i=0;i<3;i++)
    {
        for(j=0;j<3;j++)
        {
            c[i][j]=0;
            for(k=0;k<3;k++)
            {
                c[i][j]+=(((a[i][k])*(b[k][j])));
				c[i][j]=c[i][j]%MOD;
            }
        }
    }
}
