#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define vi vector<int>
#define mp make_pair
#define pb push_back
const int mod = 1e9 + 7;

ll rec[3][3] = {
	{1, 4, 2},
	{1, 0, 0},
	{0, 1, 0}
};

ll tmp[3][3];

void mul(ll a[3][3], ll b[3][3]) {
	for (int i = 0; i < 3; ++i) {
		for (int j = 0; j < 3; ++j) {
			tmp[i][j] = 0;
			for (int k = 0; k < 3; ++k) {
				tmp[i][j] += a[i][k] * b[k][j];
				tmp[i][j] %= mod;
			}
		}
	}
}

void copy(ll a[3][3]) {
	for (int i = 0; i < 3; ++i)
		for (int j = 0; j < 3; ++j)
			a[i][j] = tmp[i][j];
}

int solve(int n) {
	ll sq[3][3];
	ll res[3][3];
	memset(res, 0, sizeof(res));
	for (int i = 0; i < 3; ++i)
		res[i][i] = 1;
	for (int i = 0; i < 3; ++i)
		for (int j = 0; j < 3; ++j)
			sq[i][j] = rec[i][j];
	while (n) {
		if (n & 1) {
			mul(res, sq);
			copy(res);
		}
		mul(sq, sq);
		copy(sq);
		n >>= 1;
	}
	return res[0][0];
}

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	int t; cin >> t;
	while (t--) {
		ll n; cin >> n;
		cout << solve(n) << "
";
	}
	return 0;
}
