int lmb(LL n)
	{
	LL n1=n; LL ret=0; while(n1=n1>>1) ret++; return ret;
	}
	
	
LL** mul(LL** A,LL** B)
	{
	LL i,j,k;
	LL** C=new LL*[3];
	REP(i,3) C[i]=new LL[3];
	REP(i,3)
		REP(j,3)
			{
			C[i][j]=0;
			REP(k,3)
				C[i][j]=(C[i][j]+(A[i][k]*B[k][j])%MOD)%MOD;
			}
	return C;
	}
	
LL** multiply(LL** A,LL n)
	{
	LL i,j,k;
	LL** C=new LL*[3];
	REP(i,3) C[i]=new LL[3];
	if (!n)
		{
		REP(i,3) REP(j,3) C[i][j]=(i==j)?1:0;
		return C;
		}	
	C=multiply(A,n/2);
	if (n&1)
		C=mul(mul(C,C),A);
	else 
		C=mul(C,C);
	return C;
	}
