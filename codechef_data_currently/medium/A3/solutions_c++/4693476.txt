int n, q, nu = 1, k;
int bonus = 0;
struct nod
{
	int x, v, c;
	void in (int a, int b, int k)
	{
		x = a, v = b, c = k;
	}
	const bool operator < (nod a) const 
	{
		if (x < a.x)
			return true;
		return false;
	}
};
nod ot[1000000];
void init()
{
	cin >> n;
	bonus = 0;
	for (int i = 0; i < n; ++i)
	{
		char s;
		string st;
		int a;
		cin >> s >> a >> st;
		if (st[0] == 'Y')
		{
			if (s == '=')
				ot[i].in(a, 1, 0);
			if (s == '>' )
				ot[i].in(a + 1, 0, 1);
			if (s == '<' && a != 1)
			{
				bonus++;
				ot[i].in(a, 0, -1);
			}
		}
		else
		{
			if (s == '=')
			{
				bonus++;
				ot[i].in(a, -1, 0);
			}
			if (s == '>')
			{
				bonus++;
				ot[i].in(a + 1, 0, -1);
			}
			if (s == '<')
				ot[i].in(a, 0, 1);
		}
	}
	ot[n].in(1e9 + 1, -1e9, -1e9);
	ot[n + 1].in(0, -1e9, 0); 
	sort(ot, ot + n + 2);
}
void solve()
{
	int o = bonus;
	int v = 0;
	for (int i = 0; i < n + 2; ++i)
	{
		v += ot[i].v;
		bonus += ot[i].c;
		if (i != n + 1)
			if (ot[i + 1].x == ot[i].x)
				continue;
		o = max(bonus + v, o);
		v = 0;
	}
	printf("%d
", n - o);
}
int main()
{
    #ifndef ONLINE_JUDGE

    #endif
	int t;
	cin >> t;
	for (int i = 0; i < t; ++i)
	{ 
		init();
		solve();

	}
    return 0;
}
