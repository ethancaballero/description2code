inline long long inp()
{
    register long long r=0,c;
    for(c=getchar_unlocked(); c<=32; c=getchar_unlocked());
    if(c=='-')
        return -inp();
    for(; c>32; r=(r<<1)+(r<<3)+c-'0',c=getchar_unlocked());
    return r;
}

}
long long value(char ch)
{
	switch(ch)
	{
		case 'c':
		return 0;
		case 'h':
		return 1;
		case 'e':
		return 2;
		case 'f':
		return 3;
	}
}
char rev(long long p)
